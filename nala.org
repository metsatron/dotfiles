* Repos setup script
#+BEGIN_SRC bash :tangle debian/.local/bin/nala-repos-setup :shebang "#!/usr/bin/env bash"
set -euo pipefail

MAN="${NALA_REPOS_SSV:-$HOME/.nala/manifest/repos.ssv}"

if [ ! -f "$MAN" ]; then
  echo "No repos manifest at $MAN" >&2
  exit 0
fi

# Detect OS
OS_ID="unknown"
if [ -f /etc/os-release ]; then
  # shellcheck disable=SC1091
  . /etc/os-release
  OS_ID="$ID"
fi
IS_DEVUAN=0
[ -f /etc/devuan_version ] && IS_DEVUAN=1
ARCH="$(dpkg --print-architecture)"

scope_matches() {
  local scope="$1"
  case "$scope" in
    all) return 0 ;;
    devuan) [ "$IS_DEVUAN" -eq 1 ] ;;
    ubuntu) [ "$OS_ID" = "ubuntu" ] ;;
    *) return 0 ;;
  esac
}

CHANGED=0

parse_repo_line() {
  local line="$1"
  local -a parts=()

  mapfile -t parts < <(printf '%s\n' "$line" | xargs -n1 printf '%s\n')
  while [ "${#parts[@]}" -lt 6 ]; do
    parts+=("")
  done
  printf '%s|%s|%s|%s|%s|%s\n' \
    "${parts[0]}" "${parts[1]}" "${parts[2]}" \
    "${parts[3]}" "${parts[4]}" "${parts[5]}"
}

while IFS='|' read -r name type key_url repo_spec os_match _extra; do
  [ -z "$name" ] && continue

  if ! scope_matches "$os_match"; then
    echo "  skip $name (os=$os_match, this=$OS_ID)"
    continue
  fi

  repo_spec="${repo_spec//\{\{ARCH\}\}/$ARCH}"

  case "$type" in
    key+sources)
      # key_url is GPG key, repo_spec is URL to .sources file
      keyring="/usr/share/keyrings/${name}-keyring.gpg"
      sources_file="/etc/apt/sources.list.d/${name}.sources"

      if [ -f "$sources_file" ]; then
        echo "  ok $name (already configured)"
        continue
      fi

      echo "  add $name repo..."
      sudo install -d -m 755 "$(dirname "$keyring")" "$(dirname "$sources_file")"

      # Download and install key
      if [ -n "$key_url" ]; then
        if echo "$key_url" | grep -q '\.gpg$'; then
          sudo wget -qO "$keyring" "$key_url"
        else
          wget -qO- "$key_url" | gpg --dearmor | sudo tee "$keyring" >/dev/null
        fi
        sudo chmod go+r "$keyring"
      fi

      # Download and install sources file
      sudo wget -qO "$sources_file" "$repo_spec"
      CHANGED=1
      echo "  done $name"
      ;;

    ppa)
      # Ubuntu PPA — repo_spec is the PPA string
      if [ "$OS_ID" != "ubuntu" ]; then
        echo "  skip $name (PPA, not Ubuntu)"
        continue
      fi
      if grep -rq "$repo_spec" /etc/apt/sources.list.d/ 2>/dev/null; then
        echo "  ok $name (already configured)"
        continue
      fi
      echo "  add $name PPA..."
      sudo add-apt-repository -y "ppa:$repo_spec"
      CHANGED=1
      ;;

    key+list)
      keyring="${_extra:-}"
      [ -n "$keyring" ] || keyring="/usr/share/keyrings/${name}-keyring.gpg"
      list_file="/etc/apt/sources.list.d/${name}.list"

      if [ -f "$list_file" ] && grep -Fqx "$repo_spec" "$list_file" 2>/dev/null; then
        echo "  ok $name (already configured)"
        continue
      fi

      echo "  add $name repo..."
      sudo install -d -m 755 "$(dirname "$keyring")" "$(dirname "$list_file")"

      if [ -n "$key_url" ]; then
        if echo "$key_url" | grep -q '\.gpg$'; then
          sudo wget -qO "$keyring" "$key_url"
        else
          wget -qO- "$key_url" | gpg --dearmor --yes | sudo tee "$keyring" >/dev/null
        fi
        sudo chmod go+r "$keyring"
      fi

      printf '%s\n' "$repo_spec" | sudo tee "$list_file" >/dev/null
      CHANGED=1
      ;;

    *)
      echo "  unknown repo type: $type for $name" >&2
      ;;
  esac
done < <(
  while IFS= read -r raw; do
    line="${raw#${raw%%[![:space:]]*}}"
    line="${line%${line##*[![:space:]]}}"
    [ -z "$line" ] && continue
    case "$line" in
      \#*) continue ;;
    esac
    parse_repo_line "$line"
  done < "$MAN"
)

if [ "$CHANGED" -eq 1 ]; then
  echo "Repos changed, updating package lists..."
  APT="apt-get"
  command -v nala &>/dev/null && APT="nala"
  sudo $APT update
fi
#+END_SRC

* Release-backed .deb sync
#+BEGIN_SRC bash :tangle debian/.local/bin/nala-release-sync :shebang "#!/usr/bin/env bash"
set -euo pipefail

MAN="${NALA_RELEASES_SSV:-$HOME/.nala/manifest/releases.ssv}"
CACHE_BASE="${XDG_CACHE_HOME:-$HOME/.cache}/nala/releases"

[ -f "$MAN" ] || exit 0

OS_ID="unknown"
if [ -f /etc/os-release ]; then
  # shellcheck disable=SC1091
  . /etc/os-release
  OS_ID="$ID"
fi
IS_DEVUAN=0
[ -f /etc/devuan_version ] && IS_DEVUAN=1
ARCH="$(dpkg --print-architecture)"

scope_matches() {
  local scope="$1"
  case "$scope" in
    shared|"") return 0 ;;
    devuan-only) [ "$IS_DEVUAN" -eq 1 ] ;;
    ubuntu-only) [ "$OS_ID" = "ubuntu" ] ;;
    *) return 0 ;;
  esac
}

arch_matches() {
  local want="$1"
  case "$want" in
    any|all|"") return 0 ;;
    "$ARCH") return 0 ;;
    *) return 1 ;;
  esac
}

detect_apt() {
  if command -v nala &>/dev/null; then
    printf '%s\n' nala
  else
    printf '%s\n' apt-get
  fi
}

gh_latest_json() {
  local repo="$1" url="https://api.github.com/repos/${repo}/releases/latest"
  if [ -n "${GH_TOKEN:-}" ]; then
    curl -fsSL -H "Authorization: Bearer $GH_TOKEN" "$url"
  else
    curl -fsSL "$url"
  fi
}

normalize_ver() {
  sed -E 's/^v//' <<<"$1"
}

installed_version() {
  local pkg="$1"
  dpkg-query -W -f='${Version}\n' "$pkg" 2>/dev/null || true
}

cache_dir() {
  local pkg="$1"
  printf '%s/%s\n' "$CACHE_BASE" "$pkg"
}

write_cache_meta() {
  local dir="$1" json="$2" tag="$3" name="$4" url="$5" digest="$6"
  mkdir -p "$dir"
  printf '%s\n' "$json" > "$dir/latest.json"
  printf '%s\n' "$tag" > "$dir/tag"
  printf '%s\n' "$name" > "$dir/asset.name"
  printf '%s\n' "$url" > "$dir/asset.url"
  printf '%s\n' "$digest" > "$dir/asset.digest"
}

install_local_deb() {
  local apt_bin="$1" bundle="$2"
  local pkg_name
  pkg_name="$(dpkg-deb --field "$bundle" Package 2>/dev/null || true)"
  echo "  -> installing $(basename "$bundle") with $apt_bin"
  sudo DEBIAN_FRONTEND=noninteractive "$apt_bin" install -y "$bundle" || true
  # Nala sometimes returns 0 but silently skips — verify the package is installed
  if [ -n "$pkg_name" ] && dpkg -l "$pkg_name" 2>/dev/null | grep -q "^ii"; then
    return 0
  fi
  echo "  -> $apt_bin did not install $pkg_name, retrying with dpkg + apt-get -f"
  sudo DEBIAN_FRONTEND=noninteractive dpkg -i "$bundle" || true
  sudo DEBIAN_FRONTEND=noninteractive apt-get install -f -y
}

sync_one() {
  local apt_bin="$1" repo="$2" pkg="$3" asset_glob="$4" want_arch="$5" scope="$6"
  local json latest_tag latest_ver asset_name asset_url asset_digest dir bundle tmp old_name cached_digest cached_tag installed
  local bundle_pkg bundle_ver effective_pkg compare_ver
  local asset_entry_name asset_entry_url asset_entry_digest

  if ! scope_matches "$scope"; then
    echo "  skip $pkg (scope=$scope, this=$OS_ID)"
    return 0
  fi

  if ! arch_matches "$want_arch"; then
    echo "  skip $pkg (arch=$want_arch, this=$ARCH)"
    return 0
  fi

  json="$(gh_latest_json "$repo")"
  latest_tag="$(jq -r '.tag_name' <<<"$json")"
  latest_ver="$(normalize_ver "$latest_tag")"

  while IFS=$'\t' read -r asset_entry_name asset_entry_url asset_entry_digest; do
    case "$asset_entry_name" in
      $asset_glob)
        asset_name="$asset_entry_name"
        asset_url="$asset_entry_url"
        asset_digest="$asset_entry_digest"
        break
        ;;
    esac
  done < <(jq -r '.assets[] | [.name, .browser_download_url, (.digest // "")] | @tsv' <<<"$json")

  [ -n "${asset_url:-}" ] || {
    echo "  ! no release asset matching $asset_glob for $repo" >&2
    return 1
  }

  dir="$(cache_dir "$pkg")"
  mkdir -p "$dir"
  bundle="$dir/$asset_name"
  old_name="$( [ -f "$dir/asset.name" ] && cat "$dir/asset.name" || true )"
  cached_digest="$( [ -f "$dir/asset.digest" ] && cat "$dir/asset.digest" || true )"
  cached_tag="$( [ -f "$dir/tag" ] && cat "$dir/tag" || true )"

  if [ -n "$old_name" ] && [ "$old_name" != "$asset_name" ] && [ -f "$dir/$old_name" ]; then
    rm -f "$dir/$old_name"
  fi

  if [ ! -s "$bundle" ] \
    || { [ -n "$asset_digest" ] && [ "$cached_digest" != "$asset_digest" ]; } \
    || { [ -z "$asset_digest" ] && [ "$cached_tag" != "$latest_tag" ]; }; then
    tmp="$bundle.part"
    curl -fL "$asset_url" -o "$tmp"
    mv -f "$tmp" "$bundle"
  fi

  write_cache_meta "$dir" "$json" "$latest_tag" "$asset_name" "$asset_url" "$asset_digest"

  bundle_pkg="$(dpkg-deb -f "$bundle" Package 2>/dev/null || true)"
  bundle_ver="$(dpkg-deb -f "$bundle" Version 2>/dev/null || true)"
  effective_pkg="$pkg"
  [ -n "$bundle_ver" ] && compare_ver="$bundle_ver" || compare_ver="$latest_ver"

  if [ -n "$bundle_pkg" ] && [ "$bundle_pkg" != "$pkg" ]; then
    echo "  ! manifest package $pkg differs from deb metadata $bundle_pkg; using deb metadata"
    effective_pkg="$bundle_pkg"
  fi

  installed="$(installed_version "$effective_pkg")"
  echo "  release: pkg=$effective_pkg arch=$want_arch latest=$compare_ver installed=${installed:-none}"
  if [ -n "$installed" ] && dpkg --compare-versions "$installed" ge "$compare_ver"; then
    echo "  -> already current, skipping install for $effective_pkg"
    return 0
  fi

  install_local_deb "$apt_bin" "$bundle"
}

APT_BIN="$(detect_apt)"

echo "== Syncing release-backed .deb packages =="
while IFS='|' read -r repo pkg asset_glob want_arch scope _extra; do
  [ -n "$repo" ] || continue
  sync_one "$APT_BIN" "$repo" "$pkg" "$asset_glob" "$want_arch" "$scope" || true
done < <(
  awk -v FS='[ \t]+' '
    function dq(s){ return (s == "\"\"" ? "" : s) }
    /^[ \t]*#/ || NF==0 { next }
    {
      printf "%s|%s|%s|%s|%s|%s\n",
        dq($1), dq($2), dq($3), dq($4), dq($5), dq($6)
    }
  ' "$MAN"
)
#+END_SRC

* Apply script
#+BEGIN_SRC bash :tangle debian/.local/bin/nala-apply :shebang "#!/usr/bin/env bash"
set -euo pipefail

MAN="${NALA_SSV:-$HOME/.nala/manifest/packages.ssv}"
ENFORCE="${ENFORCE:-1}"
UNINSTALL="${UNINSTALL:-0}"
REL_SYNC_HELPER="${HOME}/.local/bin/nala-release-sync"
[ -x "$REL_SYNC_HELPER" ] || REL_SYNC_HELPER="${HOME}/DotCortex/debian/.local/bin/nala-release-sync"

if [ ! -f "$MAN" ]; then
  echo "No manifest at $MAN" >&2
  exit 1
fi

# --- Detect OS ---
OS_ID="unknown"
if [ -f /etc/os-release ]; then
  # shellcheck disable=SC1091
  . /etc/os-release
  OS_ID="$ID"
fi
IS_DEVUAN=0
[ -f /etc/devuan_version ] && IS_DEVUAN=1

scope_matches() {
  local scope="$1"
  case "$scope" in
    shared) return 0 ;;
    devuan-only) [ "$IS_DEVUAN" -eq 1 ] ;;
    ubuntu-only) [ "$OS_ID" = "ubuntu" ] ;;
    *) return 0 ;;
  esac
}

# --- Bootstrap nala if missing ---
APT="apt-get"
if command -v nala &>/dev/null; then
  APT="nala"
else
  echo "nala not found, installing..."
  sudo apt-get update
  sudo apt-get install -y nala || {
    echo "nala install failed, falling back to apt-get"
    APT="apt-get"
  }
  command -v nala &>/dev/null && APT="nala"
fi
echo "Using: $APT"

# --- Parse manifest, filter by scope ---
INSTALL_LIST=()

while IFS='|' read -r pkg _ver _cat scope _extra; do
  [ -z "$pkg" ] && continue

  if ! scope_matches "$scope"; then
    continue
  fi

  # Check if already installed
  if dpkg -s "$pkg" &>/dev/null 2>&1; then
    continue
  fi

  # Check if package exists in repos
  if ! apt-cache show "$pkg" &>/dev/null 2>&1; then
    echo "Error: $pkg not found" >&2
    continue
  fi

  INSTALL_LIST+=("$pkg")
done < <(
  awk -v FS='[ \t]+' '
    function dq(s){ return (s == "\"\"" ? "" : s) }
    /^[ \t]*#/ || NF==0 { next }
    {
      printf "%s|%s|%s|%s|%s\n",
        dq($1), dq($2), dq($3), dq($4), dq($5)
    }
  ' "$MAN"
)

# --- Install missing ---
if [ ${#INSTALL_LIST[@]} -eq 0 ]; then
  echo "All manifest packages are installed."
else
  echo "Installing ${#INSTALL_LIST[@]} packages..."
  printf "  %s\n" "${INSTALL_LIST[@]}"
  sudo DEBIAN_FRONTEND=noninteractive $APT install -y "${INSTALL_LIST[@]}" || {
    if [ "$APT" = "nala" ]; then
      echo "nala failed (broken resolver), falling back to apt-get..."
      sudo DEBIAN_FRONTEND=noninteractive apt-get install -y "${INSTALL_LIST[@]}"
    else
      exit 1
    fi
  }
fi

if [ -x "$REL_SYNC_HELPER" ]; then
  "$REL_SYNC_HELPER"
fi

echo "Done."

  echo "$to_install" | sed 's/^/  + /'

  echo "[to install] (none)"
fi

echo
echo "Manifest: $(wc -l < "$wanted_tmp") packages for this OS"

* Release-backed .deb diff
#+BEGIN_SRC bash :tangle debian/.local/bin/nala-release-diff :shebang "#!/usr/bin/env bash"
set -euo pipefail

MAN="${NALA_RELEASES_SSV:-$HOME/.nala/manifest/releases.ssv}"

[ -f "$MAN" ] || exit 0

OS_ID="unknown"
if [ -f /etc/os-release ]; then
  # shellcheck disable=SC1091
  . /etc/os-release
  OS_ID="$ID"
fi
IS_DEVUAN=0
[ -f /etc/devuan_version ] && IS_DEVUAN=1
ARCH="$(dpkg --print-architecture)"

scope_matches() {
  local scope="$1"
  case "$scope" in
    shared|"") return 0 ;;
    devuan-only) [ "$IS_DEVUAN" -eq 1 ] ;;
    ubuntu-only) [ "$OS_ID" = "ubuntu" ] ;;
    *) return 0 ;;
  esac
}

arch_matches() {
  local want="$1"
  case "$want" in
    any|all|"") return 0 ;;
    "$ARCH") return 0 ;;
    *) return 1 ;;
  esac
}

gh_latest_json() {
  local repo="$1" url="https://api.github.com/repos/${repo}/releases/latest"
  if [ -n "${GH_TOKEN:-}" ]; then
    curl -fsSL -H "Authorization: Bearer $GH_TOKEN" "$url"
  else
    curl -fsSL "$url"
  fi
}

normalize_ver() {
  sed -E 's/^v//' <<<"$1"
}

installed_version() {
  local pkg="$1"
  dpkg-query -W -f='${Version}\n' "$pkg" 2>/dev/null || true
}

echo "[release-backed debs]"
while IFS='|' read -r repo pkg asset_glob want_arch scope _extra; do
  [ -n "$repo" ] || continue

  if ! scope_matches "$scope"; then
    echo "  . $pkg skipped (scope=$scope, this=$OS_ID)"
    continue
  fi

  if ! arch_matches "$want_arch"; then
    echo "  . $pkg skipped (arch=$want_arch, this=$ARCH)"
    continue
  fi

  json="$(gh_latest_json "$repo")"
  latest_tag="$(jq -r '.tag_name' <<<"$json")"
  latest_ver="$(normalize_ver "$latest_tag")"
  resolved_asset_name=""
  while IFS=$'\t' read -r asset_entry_name _asset_entry_url _asset_entry_digest; do
    case "$asset_entry_name" in
      $asset_glob)
        resolved_asset_name="$asset_entry_name"
        break
        ;;
    esac
  done < <(jq -r '.assets[] | [.name, .browser_download_url, (.digest // "")] | @tsv' <<<"$json")

  if [ -z "$resolved_asset_name" ]; then
    echo "  ? $pkg (no release asset matching $asset_glob in $repo)"
    continue
  fi

  have="$(installed_version "$pkg")"
  if [ -z "$have" ]; then
    echo "  + $pkg (missing -> $latest_ver, asset $resolved_asset_name)"
    continue
  fi

  if dpkg --compare-versions "$have" ge "$latest_ver"; then
    echo "  = $pkg ($have, asset $resolved_asset_name)"
  else
    echo "  ~ $pkg ($have -> $latest_ver, asset $resolved_asset_name)"
  fi
done < <(
  awk -v FS='[ \t]+' '
    function dq(s){ return (s == "\"\"" ? "" : s) }
    /^[ \t]*#/ || NF==0 { next }
    {
      printf "%s|%s|%s|%s|%s|%s\n",
        dq($1), dq($2), dq($3), dq($4), dq($5), dq($6)
    }
  ' "$MAN"
)
#+END_SRC
