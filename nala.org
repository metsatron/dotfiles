* Repos setup script
#+BEGIN_SRC bash :tangle debian/.local/bin/nala-repos-setup :shebang "#!/usr/bin/env bash"
set -euo pipefail

MAN="${NALA_REPOS_SSV:-$HOME/.nala/manifest/repos.ssv}"

if [ ! -f "$MAN" ]; then
  echo "No repos manifest at $MAN" >&2
  exit 0
fi

# Detect OS
OS_ID="unknown"
if [ -f /etc/os-release ]; then
  # shellcheck disable=SC1091
  . /etc/os-release
  OS_ID="$ID"
fi
IS_DEVUAN=0
[ -f /etc/devuan_version ] && IS_DEVUAN=1

scope_matches() {
  local scope="$1"
  case "$scope" in
    all) return 0 ;;
    devuan) [ "$IS_DEVUAN" -eq 1 ] ;;
    ubuntu) [ "$OS_ID" = "ubuntu" ] ;;
    *) return 0 ;;
  esac
}

CHANGED=0

parse_repo_line() {
  local line="$1"
  local -a parts=()

  mapfile -t parts < <(printf '%s\n' "$line" | xargs -n1 printf '%s\n')
  while [ "${#parts[@]}" -lt 6 ]; do
    parts+=("")
  done
  printf '%s|%s|%s|%s|%s|%s\n' \
    "${parts[0]}" "${parts[1]}" "${parts[2]}" \
    "${parts[3]}" "${parts[4]}" "${parts[5]}"
}

while IFS='|' read -r name type key_url repo_spec os_match _extra; do
  [ -z "$name" ] && continue

  if ! scope_matches "$os_match"; then
    echo "  skip $name (os=$os_match, this=$OS_ID)"
    continue
  fi

  case "$type" in
    key+sources)
      # key_url is GPG key, repo_spec is URL to .sources file
      keyring="/usr/share/keyrings/${name}-keyring.gpg"
      sources_file="/etc/apt/sources.list.d/${name}.sources"

      if [ -f "$sources_file" ]; then
        echo "  ok $name (already configured)"
        continue
      fi

      echo "  add $name repo..."

      # Download and install key
      if [ -n "$key_url" ]; then
        if echo "$key_url" | grep -q '\.gpg$'; then
          sudo wget -qO "$keyring" "$key_url"
        else
          wget -qO- "$key_url" | gpg --dearmor | sudo tee "$keyring" >/dev/null
        fi
      fi

      # Download and install sources file
      sudo wget -qO "$sources_file" "$repo_spec"
      CHANGED=1
      echo "  done $name"
      ;;

    ppa)
      # Ubuntu PPA — repo_spec is the PPA string
      if [ "$OS_ID" != "ubuntu" ]; then
        echo "  skip $name (PPA, not Ubuntu)"
        continue
      fi
      if grep -rq "$repo_spec" /etc/apt/sources.list.d/ 2>/dev/null; then
        echo "  ok $name (already configured)"
        continue
      fi
      echo "  add $name PPA..."
      sudo add-apt-repository -y "ppa:$repo_spec"
      CHANGED=1
      ;;

    key+list)
      keyring="${_extra:-}"
      [ -n "$keyring" ] || keyring="/usr/share/keyrings/${name}-keyring.gpg"
      list_file="/etc/apt/sources.list.d/${name}.list"

      if [ -f "$list_file" ] && grep -Fqx "$repo_spec" "$list_file" 2>/dev/null; then
        echo "  ok $name (already configured)"
        continue
      fi

      echo "  add $name repo..."
      sudo mkdir -p "$(dirname "$keyring")"

      if [ -n "$key_url" ]; then
        if echo "$key_url" | grep -q '\.gpg$'; then
          sudo wget -qO "$keyring" "$key_url"
        else
          wget -qO- "$key_url" | gpg --dearmor --yes | sudo tee "$keyring" >/dev/null
        fi
      fi

      printf '%s\n' "$repo_spec" | sudo tee "$list_file" >/dev/null
      CHANGED=1
      ;;

    *)
      echo "  unknown repo type: $type for $name" >&2
      ;;
  esac
done < <(
  while IFS= read -r raw; do
    line="${raw#${raw%%[![:space:]]*}}"
    line="${line%${line##*[![:space:]]}}"
    [ -z "$line" ] && continue
    case "$line" in
      \#*) continue ;;
    esac
    parse_repo_line "$line"
  done < "$MAN"
)

if [ "$CHANGED" -eq 1 ]; then
  echo "Repos changed, updating package lists..."
  APT="apt-get"
  command -v nala &>/dev/null && APT="nala"
  sudo $APT update -y
fi
#+END_SRC

* Apply script
#+BEGIN_SRC bash :tangle debian/.local/bin/nala-apply :shebang "#!/usr/bin/env bash"
set -euo pipefail

MAN="${NALA_SSV:-$HOME/.nala/manifest/packages.ssv}"
ENFORCE="${ENFORCE:-1}"
UNINSTALL="${UNINSTALL:-0}"

if [ ! -f "$MAN" ]; then
  echo "No manifest at $MAN" >&2
  exit 1
fi

# --- Detect OS ---
OS_ID="unknown"
if [ -f /etc/os-release ]; then
  # shellcheck disable=SC1091
  . /etc/os-release
  OS_ID="$ID"
fi
IS_DEVUAN=0
[ -f /etc/devuan_version ] && IS_DEVUAN=1

scope_matches() {
  local scope="$1"
  case "$scope" in
    shared) return 0 ;;
    devuan-only) [ "$IS_DEVUAN" -eq 1 ] ;;
    ubuntu-only) [ "$OS_ID" = "ubuntu" ] ;;
    *) return 0 ;;
  esac
}

# --- Bootstrap nala if missing ---
APT="apt-get"
if command -v nala &>/dev/null; then
  APT="nala"
else
  echo "nala not found, installing..."
  sudo apt-get update -y
  sudo apt-get install -y nala || {
    echo "nala install failed, falling back to apt-get"
    APT="apt-get"
  }
  command -v nala &>/dev/null && APT="nala"
fi
echo "Using: $APT"

# --- Parse manifest, filter by scope ---
INSTALL_LIST=()

while IFS='|' read -r pkg _ver _cat scope _extra; do
  [ -z "$pkg" ] && continue

  if ! scope_matches "$scope"; then
    continue
  fi

  # Check if already installed
  if dpkg -s "$pkg" &>/dev/null 2>&1; then
    continue
  fi

  # Check if package exists in repos
  if ! apt-cache show "$pkg" &>/dev/null 2>&1; then
    echo "Error: $pkg not found" >&2
    continue
  fi

  INSTALL_LIST+=("$pkg")
done < <(
  awk -v FS='[ \t]+' '
    function dq(s){ return (s == "\"\"" ? "" : s) }
    /^[ \t]*#/ || NF==0 { next }
    {
      printf "%s|%s|%s|%s|%s\n",
        dq($1), dq($2), dq($3), dq($4), dq($5)
    }
  ' "$MAN"
)

# --- Install missing ---
if [ ${#INSTALL_LIST[@]} -eq 0 ]; then
  echo "All manifest packages are installed."
else
  echo "Installing ${#INSTALL_LIST[@]} packages..."
  printf "  %s\n" "${INSTALL_LIST[@]}"
  sudo $APT install -y "${INSTALL_LIST[@]}" || {
    if [ "$APT" = "nala" ]; then
      echo "nala failed (broken resolver), falling back to apt-get..."
      sudo apt-get install -y "${INSTALL_LIST[@]}"
    else
      exit 1
    fi
  }
fi

echo "Done."
#+END_SRC

* Diff script
#+BEGIN_SRC bash :tangle debian/.local/bin/nala-diff :shebang "#!/usr/bin/env bash"
set -euo pipefail

MAN="${NALA_SSV:-$HOME/.nala/manifest/packages.ssv}"

if [ ! -f "$MAN" ]; then
  echo "No manifest at $MAN" >&2
  exit 1
fi

# --- Detect OS ---
OS_ID="unknown"
if [ -f /etc/os-release ]; then
  # shellcheck disable=SC1091
  . /etc/os-release
  OS_ID="$ID"
fi
IS_DEVUAN=0
[ -f /etc/devuan_version ] && IS_DEVUAN=1

# --- Extract wanted packages (filtered by scope) ---
wanted_tmp="$(mktemp)"
trap 'rm -f "$wanted_tmp"' EXIT

awk -v FS='[ \t]+' -v is_devuan="$IS_DEVUAN" -v os_id="$OS_ID" '
  function dq(s){ return (s == "\"\"" ? "" : s) }
  /^[ \t]*#/ || NF==0 { next }
  {
    pkg = dq($1)
    scope = dq($4)
    if (scope == "shared") print pkg
    else if (scope == "devuan-only" && is_devuan == 1) print pkg
    else if (scope == "ubuntu-only" && os_id == "ubuntu") print pkg
  }
' "$MAN" | sort -u > "$wanted_tmp"

# --- Get installed manual packages ---
have_tmp="$(mktemp)"
trap 'rm -f "$wanted_tmp" "$have_tmp"' EXIT

dpkg-query -W -f='${Package}\n' | sort -u > "$have_tmp"

# --- Diff ---
to_install="$(comm -23 "$wanted_tmp" "$have_tmp")"
# Show manually installed that aren't in manifest (informational only)

echo "=== nala:diff ==="
echo

if [ -n "$to_install" ]; then
  count=$(echo "$to_install" | wc -l)
  echo "[to install] ($count packages)"
  echo "$to_install" | sed 's/^/  + /'
else
  echo "[to install] (none)"
fi

echo
echo "Manifest: $(wc -l < "$wanted_tmp") packages for this OS"
echo "Installed: $(wc -l < "$have_tmp") total dpkg packages"
#+END_SRC

* Capture script
#+BEGIN_SRC bash :tangle debian/.local/bin/nala-capture :shebang "#!/usr/bin/env bash"
set -euo pipefail

MAN="${NALA_SSV:-$HOME/.nala/manifest/packages.ssv}"

if [ ! -f "$MAN" ]; then
  echo "No manifest at $MAN — creating with header" >&2
  mkdir -p "$(dirname "$MAN")"
  echo '# PKG VERSION CATEGORY SCOPE EXTRA' > "$MAN"
fi

# Get packages already in manifest
existing_tmp="$(mktemp)"
trap 'rm -f "$existing_tmp"' EXIT
awk -v FS='[ \t]+' '!/^[ \t]*#/ && NF { print $1 }' "$MAN" | sort -u > "$existing_tmp"

# Get manually installed packages
manual_tmp="$(mktemp)"
trap 'rm -f "$existing_tmp" "$manual_tmp"' EXIT
apt-mark showmanual | sort -u > "$manual_tmp"

# Find new (manually installed but not in manifest)
new_pkgs="$(comm -23 "$manual_tmp" "$existing_tmp")"

if [ -z "$new_pkgs" ]; then
  echo "No new manually-installed packages to capture."
  exit 0
fi

count=$(echo "$new_pkgs" | wc -l)
echo "Found $count new manually-installed packages:"
echo "$new_pkgs" | sed 's/^/  /'
echo

echo "Appending to $MAN with category=uncategorized scope=shared"
echo "" >> "$MAN"
echo "# --- Captured $(date +%Y-%m-%d) ---" >> "$MAN"
while IFS= read -r pkg; do
  printf '%s "" uncategorized shared ""\n' "$pkg" >> "$MAN"
done <<< "$new_pkgs"

echo "Done. Review and recategorize in nala.org."
#+END_SRC
