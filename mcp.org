* MCP Loom
Declarative home for MCP launchers and client config that should survive
fresh-machine bootstrap and normal DotCortex stow flows.

Secrets do not live in this repo. Runtime wrappers source =~/.env= and
fail fast when required keys are missing.

** OpenCode - Brave Search
The OpenCode global config lives at =~/.opencode.json=. Manage it here,
tangle to =all/.opencode.json=, then bring it home with
=make safe-stow=.

#+BEGIN_SRC bash :tangle all/.local/bin/opencode-mcp-brave-search :shebang "#!/usr/bin/env bash"
# ~/DotCortex/all/.local/bin/opencode-mcp-brave-search
set -euo pipefail

if [ -r "$HOME/.env" ]; then
  # shellcheck source=/dev/null
  . "$HOME/.env"
fi

: "${BRAVE_API_KEY:?BRAVE_API_KEY must be set in \$HOME/.env}"

if ! command -v npx >/dev/null 2>&1; then
  echo "opencode-mcp-brave-search: npx not found in PATH." >&2
  exit 1
fi

export BRAVE_API_KEY

exec npx -y @modelcontextprotocol/server-brave-search
#+END_SRC

** Claude - Telegram Bot API
Minimal Bot API MCP server. Reads =TELEGRAM_BOT_TOKEN= and =TELEGRAM_CHAT_ID=
from the environment. Set those per workspace in local MCP client config.

#+BEGIN_SRC python :tangle all/.local/bin/claude-telegram-mcp :shebang "#!/usr/bin/env python3" :comments no
"""Telegram Bot API MCP server — send/receive messages via configured bot."""
import os
import requests
from mcp.server.fastmcp import FastMCP

mcp = FastMCP("telegram")
_TOKEN = os.environ["TELEGRAM_BOT_TOKEN"]
_CHAT = os.environ.get("TELEGRAM_CHAT_ID", "")
_API = f"https://api.telegram.org/bot{_TOKEN}"


@mcp.tool()
def send_message(text: str, chat_id: str = "") -> str:
    """Send a text message to the configured Telegram chat. Supports Markdown."""
    target = chat_id or _CHAT
    r = requests.post(f"{_API}/sendMessage", json={
        "chat_id": target, "text": text, "parse_mode": "Markdown"
    }, timeout=10)
    return "ok" if r.ok else f"error: {r.text}"


@mcp.tool()
def get_recent_messages(limit: int = 10) -> str:
    """Get recent messages sent to this bot (getUpdates)."""
    r = requests.get(f"{_API}/getUpdates", params={"limit": limit}, timeout=10)
    if not r.ok:
        return f"error: {r.text}"
    updates = r.json().get("result", [])
    if not updates:
        return "No recent messages."
    lines = []
    for u in updates:
        msg = u.get("message", {})
        user = msg.get("from", {}).get("first_name", "?")
        text = msg.get("text", "[no text]")
        lines.append(f"{user}: {text}")
    return "\n".join(lines)


if __name__ == "__main__":
    mcp.run(transport="stdio")
#+END_SRC
