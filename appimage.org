* Purpose
Manage AppImages under =~/Applications=: daemon integration, cross-init
scheduling, conf-driven release updates (hybrid ZSync + per-repo polling),
and cleaning of broken desktop entries.

** Update strategy (hybrid, two-pass)
- *Pass 1 — ZSync:* appimageupdatetool -r for AppImages with embedded update info
- *Pass 2 — Conf:* apprelease update-all for conf-managed AppImages (GitHub, GitLab,
  Codeberg, Forgejo, direct URL)
- *Pass 3 — Cleanup:* remove .zs-old stale versions, scrub broken .desktop entries

** Conf-managed apps
Each conf file in =all/.config/apprelease.d/= declares:
: NAME, FORGE, REPO, TYPE=appimage, APPIMG_NAME, ASSET_RE, [FORGE_BASE_URL], [DIRECT_URL], [DIRECT_VERSION_CMD]

* Directory conventions
- AppImages live in: =~/Applications=
- User binaries/scripts: =~/.local/bin=
- User services: =~/.config/systemd/user= (systemd) or =~/.config/appimaged/= (sysv-init)
- User desktop cache: =~/.local/share/applications=
- Conf dir: =~/.config/apprelease.d/= (stowed from =all/.config/apprelease.d/=)
- Cache: =~/.cache/apprelease/=

* Requirements (packages/tools)
- =desktop-file-utils= (for =desktop-file-validate=)
- =glib2.0-bin= (for =gio=) — optional; falls back to rm if missing
- =jq= (for JSON parsing in apprelease)
- For daemon integration: systemd user session (X230) or =~/.config/appimaged/= daemon (T480s)

* apprelease — Multi-forge release fetcher for AppImages
#+BEGIN_SRC bash :tangle all/.local/bin/apprelease :shebang "#!/usr/bin/env bash"
set -Eeuo pipefail

CONF_DIR="${XDG_CONFIG_HOME:-$HOME/.config}/apprelease.d"
REPO_CONF_DIR="$HOME/DotCortex/all/.config/apprelease.d"
DL_DIR="${XDG_CACHE_HOME:-$HOME/.cache}/apprelease"
APPIMG_DIR="$HOME/Applications"

mkdir -p "$DL_DIR" "$APPIMG_DIR"

require() { command -v "$1" >/dev/null 2>&1 || { echo "Missing: $1" >&2; exit 2; }; }

normalize_ver() { sed -E 's/^v//' <<<"$1"; }

cache_dir() { printf '%s/%s\n' "$DL_DIR" "$1"; }

write_cache_meta() {
    local dir="$1" tag="$2" name="$3" url="$4" digest="$5"
    mkdir -p "$dir"
    printf '%s\n' "$tag"    > "$dir/tag"
    printf '%s\n' "$name"   > "$dir/asset.name"
    printf '%s\n' "$url"    > "$dir/asset.url"
    printf '%s\n' "${digest:-}" > "$dir/asset.digest"
    touch "$dir/etag"
}

etag_get() {
    local url="$1" etag_file="$2"
    local etag=""
    [[ -f "$etag_file" ]] && etag="$(cat "$etag_file")"
    if [[ -n "$etag" ]]; then
        curl -sL -H "If-None-Match: \"$etag\"" -w "%{http_code}" -o /dev/null "$url"
    else
        echo "000"
    fi
}

etag_save() {
    local url="$1" etag_file="$2"
    local etag=""
    etag="$(curl -sI -H "If-None-Match: \"$(cat "$etag_file" 2>/dev/null || echo "")\"" "$url" 2>/dev/null | awk -F: '/^etag:/ {gsub(/[ "]/,"",$2); print $2}')"
    if [[ -n "$etag" ]]; then
        printf '%s' "$etag" > "$etag_file"
    fi
}

json_asset_url() {
    jq -r --arg re "$2" '.assets[].browser_download_url | select(test($re))' <<<"$1" | head -n1
}

json_asset_name() {
    jq -r --arg re "$2" '.assets[] | select(.browser_download_url | test($re)) | .name' <<<"$1" | head -n1
}

json_asset_digest() {
    jq -r --arg re "$2" '.assets[] | select(.browser_download_url | test($re)) | .digest // ""' <<<"$1" | head -n1
}

normalize_release_json() {
    local json="$1"
    if jq -e 'type == "array"' <<<"$json" >/dev/null 2>&1; then
        jq '.[0]' <<<"$json"
    else
        echo "$json"
    fi
}

forge_release_json() {
    local forge="$1" repo="$2" base_url="${3:-}" token="${4:-}"
    case "$forge" in
        github)
            local url="https://api.github.com/repos/$repo/releases/latest"
            if [[ -n "$token" ]]; then
                curl -fsSL -H "Authorization: Bearer $token" "$url"
            else
                curl -fsSL "$url"
            fi
            ;;
        gitlab)
            local project_id="${repo%%/*}"
            local slug="${repo#*/}"
            local encoded_proj
            encoded_proj="$(python3 -c "import urllib.parse; print(urllib.parse.quote('$project_id/$slug'))" 2>/dev/null || echo "$project_id/$slug")"
            curl -fsSL "https://gitlab.com/api/v4/projects/$encoded_proj/releases?per_page=1"
            ;;
        codeberg)
            curl -fsSL "https://codeberg.org/api/v1/repos/$repo/releases?limit=1"
            ;;
        forgejo)
            [[ -z "$base_url" ]] && { echo "ERROR: FORGE_BASE_URL required for forgejo" >&2; return 1; }
            curl -fsSL "${base_url}/api/v1/repos/${repo}/releases?limit=1${token:+&token=$token}"
            ;;
        *)
            echo "ERROR: unknown FORGE: $forge" >&2; return 1
            ;;
    esac
}

forge_tags_json() {
    local forge="$1" repo="$2" base_url="${3:-}" token="${4:-}"
    case "$forge" in
        github)
            curl -fsSL "https://api.github.com/repos/$repo/tags?per_page=1"
            ;;
        gitlab)
            local encoded_proj
            encoded_proj="$(python3 -c "import urllib.parse; print(urllib.parse.quote('$repo'))" 2>/dev/null || echo "$repo")"
            curl -fsSL "https://gitlab.com/api/v4/projects/$encoded_proj/repository/tags?per_page=1"
            ;;
        codeberg)
            curl -fsSL "https://codeberg.org/api/v1/repos/$repo/tags?limit=1"
            ;;
        forgejo)
            [[ -z "$base_url" ]] && { echo "ERROR: FORGE_BASE_URL required for forgejo" >&2; return 1; }
            curl -fsSL "${base_url}/api/v1/repos/${repo}/tags?limit=1${token:+&token=$token}"
            ;;
        *)
            echo "ERROR: unknown FORGE: $forge" >&2; return 1
            ;;
    esac
}

github_asset_url()  { jq -r --arg re "$2" '.assets[] | select(.name | test($re; "i")) | .browser_download_url' <<<"$1" | head -n1; }
github_asset_name() { jq -r --arg re "$2" '.assets[] | select(.name | test($re; "i")) | .name' <<<"$1" | head -n1; }

gitlab_asset_url()    { jq -r --arg re "$2" '.assets[] | select(.link.url | test($re; "i")) | .link.url' <<<"$1" | head -n1; }
gitlab_asset_name()   { jq -r --arg re "$2" '.assets[] | select(.link.url | test($re; "i")) | .name' <<<"$1" | head -n1; }

codeberg_latest_tag() { jq -r 'if .tag_name then .tag_name else .name end' <<<"$1" | head -n1; }
codeberg_asset_url()  { jq -r --arg re "$2" '.assets[] | select(.browser_download_url | test($re; "i")) | .browser_download_url' <<<"$1" | head -n1; }
codeberg_asset_name() { jq -r --arg re "$2" '.assets[] | select(.browser_download_url | test($re; "i")) | .name' <<<"$1" | head -n1; }

forgejo_latest_tag() { jq -r 'if .tag_name then .tag_name else .name end' <<<"$1" | head -n1; }

appimage_current_ver() {
    local name="$1"
    ls "$APPIMG_DIR"/"$name"-*.AppImage 2>/dev/null | \
        while IFS= read -r f; do
            [[ -L "$f" ]] && continue
            basename "$f"
        done | \
        sed -nE "s#^${name}-([0-9][^-]*).*.AppImage#\1#p" | \
        sort -V | tail -n1 || true
}

find_conf() {
    local app="$1"
    if [[ -r "$CONF_DIR/$app.conf" ]]; then
        printf '%s\n' "$CONF_DIR/$app.conf"; return 0
    fi
    if [[ -r "$REPO_CONF_DIR/$app.conf" ]]; then
        printf '%s\n' "$REPO_CONF_DIR/$app.conf"; return 0
    fi
    return 1
}

usage() {
    cat <<EOF
Usage: apprelease <verb> [app]
  list                list all conf-managed apps
  update <app>        update a single conf-managed app
  update-all          update all conf-managed apps
  cache <app>         show cache state for an app
  health <app>        check forge reachability (no download)
Verbs affect only apps with conf files in:
  $CONF_DIR/  (user overrides)
  $REPO_CONF_DIR/  (DotCortex defaults)
EOF
}

list_apps() {
    local dir file
    for dir in "$CONF_DIR" "$REPO_CONF_DIR"; do
        [[ -d "$dir" ]] || continue
        for file in "$dir"/*.conf; do
            [[ -e "$file" ]] || continue
            basename "$file" .conf
        done
    done | sort -u
}

show_cache() {
    local app="$1" dir
    dir="$(cache_dir "$app")"
    if [[ ! -d "$dir" ]]; then
        echo "No cache for $app"; return 1
    fi
    echo "cache: $dir"
    [[ -f "$dir/tag" ]]         && printf '  tag:     %s\n' "$(cat "$dir/tag")"
    [[ -f "$dir/asset.name" ]]  && printf '  asset:   %s\n' "$(cat "$dir/asset.name")"
    [[ -f "$dir/asset.url" ]]   && printf '  url:     %s\n' "$(cat "$dir/asset.url")"
    [[ -f "$dir/etag" ]]        && printf '  etag:    %s\n' "$(cat "$dir/etag")"
}

health_check() {
    local app="$1" conf forge repo base_url
    conf="$(find_conf "$app")" || { echo "No conf for $app"; return 1; }
    source "$conf"
    : "${FORGE?}" "${NAME?}"
    if [[ "$FORGE" == "dead" ]]; then
        echo "[dead] $NAME is marked dead"; return 1
    fi
    if [[ "$FORGE" == "direct" ]]; then
        if curl -sfI --max-time 10 "$DIRECT_URL" >/dev/null 2>&1; then
            echo "[ok] $NAME direct URL reachable"; return 0
        else
            echo "[fail] $NAME direct URL unreachable"; return 1
        fi
    fi
    local json
    json="$(forge_release_json "$FORGE" "$REPO" "${FORGE_BASE_URL:-}" "${FORGE_TOKEN:-}" 2>&1)" || {
        echo "[fail] $NAME forge API unreachable: $json"; return 1
    }
    json="$(normalize_release_json "$json")"
    if jq -e '.message' <<<"$json" >/dev/null 2>&1; then
        local msg; msg="$(jq -r '.message' <<<"$json")"
        echo "[fail] $NAME API error: $msg"; return 1
    fi
    echo "[ok] $NAME forge API reachable"; return 0
}

update_one() {
    local app="$1" conf json latest_tag latest_ver asset_url asset_name
    local cached_asset have installed_ver dest
    conf="$(find_conf "$app")" || { echo "[skip] $app: no conf"; return 0; }
    source "$conf"
    : "${FORGE?}" "${NAME?}" "${APPIMG_NAME?}" "${ASSET_RE?}"

    if [[ "$FORGE" == "dead" ]]; then
        echo "[dead] $NAME is marked dead — skipping"
        return 0
    fi

    local dir; dir="$(cache_dir "$app")"

    if [[ "$FORGE" == "direct" ]]; then
        require curl
    echo "[conf] $NAME: direct URL mode"
    installed_ver="$(appimage_current_ver "$APPIMG_NAME")"
    echo "[conf] $NAME installed: ${installed_ver:-none}"
    mkdir -p "$dir"
    local tmp asset_final
        tmp="$dir/cursor-latest.AppImage"
        if ! curl -fL "$DIRECT_URL" -o "$tmp" 2>/dev/null; then
            echo "[fail] $NAME: download failed"; return 1
        fi
        asset_name="$(basename "$DIRECT_URL")"
        local fetched_ver
        fetched_ver="$(echo "$asset_name" | grep -oE '[0-9]+\.[0-9]+\.[0-9]+' | head -n1 || echo "unknown")"
        if [[ -n "$installed_ver" && "$installed_ver" == "$fetched_ver" ]]; then
            echo "[skip] $NAME is up-to-date ($installed_ver)"
            rm -f "$tmp"
            return 0
        fi
        asset_final="$APPIMG_DIR/$APPIMG_NAME-$fetched_ver.AppImage"
        mv -f "$tmp" "$asset_final"
        chmod +x "$asset_final"
        ln -sf "$asset_final" "$APPIMG_DIR/$APPIMG_NAME-latest.AppImage"
        echo "[conf] $NAME updated: $fetched_ver → $asset_final"
        return 0
    fi

    require jq curl
    json="$(forge_release_json "$FORGE" "$REPO" "${FORGE_BASE_URL:-}" "${FORGE_TOKEN:-}" 2>&1)" || {
        echo "[fail] $NAME: forge API unreachable"; return 1
    }
    json="$(normalize_release_json "$json")"

    if jq -e '.message' <<<"$json" >/dev/null 2>&1; then
        local msg; msg="$(jq -r '.message' <<<"$json")"
        echo "[fail] $NAME: $msg"; return 1
    fi

    case "$FORGE" in
        github)
            latest_tag="$(github_latest_tag "$json")"
            asset_url="$(github_asset_url "$json" "$ASSET_RE")"
            asset_name="$(github_asset_name "$json" "$ASSET_RE")"
            ;;
        gitlab)
            latest_tag="$(gitlab_latest_tag "$json")"
            asset_url="$(gitlab_asset_url "$json" "$ASSET_RE")"
            asset_name="$(gitlab_asset_name "$json" "$ASSET_RE")"
            ;;
        codeberg)
            latest_tag="$(codeberg_latest_tag "$json")"
            asset_url="$(codeberg_asset_url "$json" "$ASSET_RE")"
            asset_name="$(codeberg_asset_name "$json" "$ASSET_RE")"
            ;;
        forgejo)
            latest_tag="$(forgejo_latest_tag "$json")"
            asset_url="$(forgejo_asset_url "$json" "$ASSET_RE")"
            asset_name="$(forgejo_asset_name "$json" "$ASSET_RE")"
            ;;
        *)
            echo "[fail] $NAME: unknown FORGE=$FORGE"; return 1
            ;;
    esac

    [[ -z "$latest_tag" ]] && { echo "[fail] $NAME: could not get latest tag"; return 1; }
    [[ -z "$asset_url" ]]  && { echo "[fail] $NAME: no asset matched $ASSET_RE"; return 1; }
    [[ -z "$asset_name" ]] && { echo "[warn] $NAME: matched URL but no asset name"; asset_name="unknown"; }

    latest_ver="$(normalize_ver "$latest_tag")"
    installed_ver="$(appimage_current_ver "$APPIMG_NAME")"

    echo "[conf] $NAME installed: ${installed_ver:-none}  latest: $latest_ver"

    if [[ -n "$installed_ver" && "$(printf '%s\n%s\n' "$installed_ver" "$latest_ver" | sort -V | tail -n1)" == "$installed_ver" ]]; then
        echo "[skip] $NAME is up-to-date ($installed_ver >= $latest_ver)"
        write_cache_meta "$dir" "$latest_tag" "$asset_name" "$asset_url" ""
        return 0
    fi

    mkdir -p "$dir"
    local tmp cached_asset
    tmp="$dir/$asset_name.part"
    cached_asset="$dir/$asset_name"

    echo "[conf] $NAME: downloading $asset_url"
    if ! curl -fL "$asset_url" -o "$tmp" 2>&1; then
        echo "[fail] $NAME: download failed"; rm -f "$tmp"; return 1
    fi
    mv -f "$tmp" "$cached_asset"
    chmod +x "$cached_asset"

    dest="$APPIMG_DIR/$APPIMG_NAME-$latest_ver.AppImage"
    cp -f "$cached_asset" "$dest"
    chmod +x "$dest"
    ln -sf "$dest" "$APPIMG_DIR/$APPIMG_NAME-latest.AppImage"

    # Remove old versions of the same app
    for old in "$APPIMG_DIR"/"$APPIMG_NAME"-*.AppImage; do
        [[ -L "$old" ]] && continue
        [[ "$old" == "$dest" ]] && continue
        echo "[conf] $NAME: removing old version $(basename "$old")"
        rm -f "$old"
    done

    write_cache_meta "$dir" "$latest_tag" "$asset_name" "$asset_url" ""
    echo "[conf] $NAME updated: $latest_ver → $dest"
}

case "${1:-}" in
    list)   list_apps ;;
    cache)  show_cache "${2:?app}" ;;
    health) health_check "${2:?app}" ;;
    update) shift; update_one "${1:?app}" ;;
    update-all)
        ok=0
        for a in $(list_apps); do
            update_one "$a" || ok=$?
        done
        exit $ok
        ;;
    *) usage; exit 1 ;;
esac

            echo "Run: ~/.config/appimaged/appimaged-launch  # to start now"
        fi
        ;;
    *)
        echo "!! Unknown init system: $INIT — cannot configure daemon"
        echo "  Manual start: $BIN -v -q &"
        ;;
esac

* Bulk update of all AppImages (hybrid two-pass)

Pass 1: ZSync via appimageupdatetool
Pass 2: Conf via apprelease
Pass 3: Cleanup + desktop scrub

#+BEGIN_SRC bash :tangle all/.local/bin/appimage-update-all :shebang "#!/usr/bin/env bash"
set -Eeuo pipefail

APPS_DIR="${APPS_DIR:-$HOME/Applications}"
LOG="${LOG:-$HOME/appimage-update.log}"
TOOL="$APPS_DIR/appimageupdatetool-x86_64.AppImage"
SELF_UPDATE="${SELF_UPDATE:-1}"
CONF_DIR="$HOME/.config/apprelease.d"
REPO_CONF_DIR="$HOME/DotCortex/all/.config/apprelease.d"
DEAD_OVER="$HOME/DotCortex/all/.appimage/inventory/dead-overrides.ssv"

ts() { date +"%Y-%m-%d %H:%M:%S"; }

is_symlink() { [[ -L "$1" ]]; }

is_conf_managed() {
    local base="$1" name app dir f appimg_name
    name="${base%.AppImage}"
    for dir in "$CONF_DIR" "$REPO_CONF_DIR"; do
        [[ -d "$dir" ]] || continue
        for f in "$dir"/*.conf; do
            [[ -f "$f" ]] || continue
            appimg_name="$(awk -F= '/^APPIMG_NAME=/{gsub(/"/,"",$2); print $2; exit}' "$f")"
            if [[ -n "$appimg_name" && "$name" == "${appimg_name}"* ]]; then return 0; fi
        done
    done
    return 1
}

is_dead() {
    local base="$1"
    [[ -f "$DEAD_OVER" ]] || return 1
    grep -qiF "$base" "$DEAD_OVER"
}

echo "== appimage-update-all @ $(ts) ==" | tee -a "$LOG"
echo "PASS 1: ZSync" | tee -a "$LOG"

if [[ "$SELF_UPDATE" = "1" && -x "$TOOL" ]]; then
    echo "-- self-updating appimageupdatetool..." | tee -a "$LOG"
    if "$TOOL" --self-update >>"$LOG" 2>&1; then
        echo "ok: self-update" | tee -a "$LOG"
    else
        echo "warn: self-update failed (continuing)" | tee -a "$LOG"
    fi
fi

shopt -s nullglob
{
    for path in "$APPS_DIR"/*.AppImage; do
        base="$(basename -- "$path")"

        if is_symlink "$path"; then
            continue
        fi

        case "$base" in
            appimageupdatetool-*.AppImage|AppImageUpdate-*.AppImage)
                echo "skip: $base (self-update tool)" | tee -a "$LOG"; continue ;;
        esac

        if is_dead "$base"; then
            echo "skip: $base (dead)" | tee -a "$LOG"; continue
        fi

        if is_conf_managed "$base"; then
            echo "skip: $base (conf-managed, Pass 2)" | tee -a "$LOG"; continue
        fi

        echo "==> $base" | tee -a "$LOG"

        if [[ ! -x "$TOOL" ]]; then
            echo "fail: no appimageupdatetool at $TOOL" | tee -a "$LOG"
            continue
        fi

        set +e
        "$TOOL" -r "$path" >>"$LOG" 2>&1
        rc=$?
        set -e

        if [[ $rc -eq 0 ]]; then
            echo "ok: $base" | tee -a "$LOG"
        else
            echo "fail: $base (exit $rc). continuing." | tee -a "$LOG"
        fi
    done
} 2>>"$LOG"
shopt -u nullglob

echo "" | tee -a "$LOG"
echo "PASS 2: Conf-managed via apprelease" | tee -a "$LOG"
APRELEASE="$HOME/.local/bin/apprelease"
if [[ -x "$APRELEASE" ]]; then
    if "$APRELEASE" update-all 2>&1 | tee -a "$LOG"; then
        echo "ok: conf pass" | tee -a "$LOG"
    else
        echo "warn: conf pass completed with errors" | tee -a "$LOG"
    fi
else
    echo "skip: apprelease not found" | tee -a "$LOG"
fi

# Refresh the appimaged wrapper (idempotent)
if [[ -d /run/systemd/system ]]; then
    if systemctl --user is-enabled --quiet appimaged.service 2>/dev/null; then
        ln -sf "$APPS_DIR"/appimaged-*-x86_64.AppImage "$HOME/.local/bin/appimaged" 2>/dev/null || true
        chmod +x "$HOME/.local/bin/appimaged" 2>/dev/null || true
        echo "-- appimaged wrapper refreshed" | tee -a "$LOG"
    fi
fi

echo "" | tee -a "$LOG"
echo "PASS 3: Cleanup" | tee -a "$LOG"

# Remove stale .zs-old files
zs_old_count=0
for f in "$APPS_DIR"/*.zs-old; do
    rm -f "$f" && ((zs_old_count++)) || true
done
echo "cleanup: removed $zs_old_count .zs-old files" | tee -a "$LOG"

# Make sure everything we touch is executable
chmod -f +x "$APPS_DIR"/*.AppImage 2>/dev/null || true
chmod -f +x "$HOME/.local/share/applications/"*.desktop 2>/dev/null || true

# Scrub broken .desktop entries (integrated)
APPS="$APPS_DIR" ~/.local/bin/appimage-scrub-desktops 2>&1 | tee -a "$LOG"

- trashes entries whose =Exec= binary name isn't on PATH

* Inventory tool
Reads overrides from the tangled SSV. Classifies apps as:
- ZSync-capable (HAS_UPDATE_INFO=yes, embedded update info)
- Conf-managed (listed in apprelease.d/ conf files)
- Dead (in dead-overrides.ssv)
- Unmanaged (no update info, no conf)

#+BEGIN_SRC bash :tangle all/.local/bin/appimage-inventory :shebang "#!/usr/bin/env bash"
set -Eeuo pipefail
APPS_DIR="${APPS_DIR:-$HOME/Applications}"
INV_DIR="${INV_DIR:-$HOME/DotCortex/all/.appimage/inventory}"
TOOL="$APPS_DIR/appimageupdatetool-x86_64.AppImage"
CONF_DIR="$HOME/.config/apprelease.d"
REPO_CONF_DIR="$HOME/DotCortex/all/.config/apprelease.d"

mkdir -p "$INV_DIR"

all_ssv="$INV_DIR/all.ssv"
sup_ssv="$INV_DIR/supports.ssv"
man_ssv="$INV_DIR/manual.ssv"
dead_ssv="$INV_DIR/dead.ssv"
dead_over="$INV_DIR/dead-overrides.ssv"

printf "FILE NAME HAS_UPDATE_INFO UPDATE_KIND NOTE MANAGED_BY\n" >"$all_ssv"

STRICT="${STRICT:-1}"
RELAX_ON_403="${RELAX_ON_403:-0}"

quiet_rc() {
    ( set +e; "$@"; exit $? ) >/dev/null 2>&1
    echo $?
}

# Build list of conf-managed app names (for MANAGED_BY column)
conf_apps=""
for dir in "$CONF_DIR" "$REPO_CONF_DIR"; do
    for f in "$dir"/*.conf 2>/dev/null; do
        [[ -f "$f" ]] || continue
        app="$(basename "$f" .conf)"
        conf_apps="$conf_apps $app"
    done
done

is_conf_managed() {
    local name="$1"
    for ca in $conf_apps; do
        if [[ "$ca" == "$name" ]]; then return 0; fi
    done
    return 1
}

# Build set of dead overrides
declare -A dead_set
if [[ -f "$dead_over" ]]; then
    while IFS= read -r line; do
        [[ -z "$line" || "$line" =~ ^# ]] && continue
        line="$(echo "$line" | sed 's/[[:space:]]//g')"
        [[ -n "$line" ]] && dead_set["$line"]=1
    done < "$dead_over"
fi

is_dead() {
    local base="$1"
    [[ "${dead_set[$base]}" == "1" ]] && return 0
    return 1
}

shopt -s nullglob
for f in "$APPS_DIR"/*.AppImage; do
    base="$(basename -- "$f")"
    name="${base%.AppImage}"
    has="no"; kind=""; note=""; managed="unmanaged"

    if is_dead "$base"; then
        managed="dead"
        printf "%s %s %s %s %s %s\n" "$base" "$name" "$has" "$kind" "dead" "$managed" >>"$all_ssv"
        continue
    fi

    if is_conf_managed "$name"; then
        # Conf-managed: check if it's actually installed and report
        if [[ -x "$f" ]]; then
            managed="conf:$name"
            has="no"; kind="conf"; note="apprelease-managed"
        else
            managed="conf:$name"
            has="no"; kind="conf"; note="not-executable"
        fi
        printf "%s %s %s %s %s %s\n" "$base" "$name" "$has" "$kind" "$note" "$managed" >>"$all_ssv"
        continue
    fi

    if [[ -x "$TOOL" ]]; then
        desc_out="$("$TOOL" -d "$f" 2>&1 || true)"

        if grep -q '^[[:space:]]*Update information type:' <<<"$desc_out"; then
            has="yes"
            kind="$(sed -n 's/^[[:space:]]*Update information type:[[:space:]]*//p' <<<"$desc_out" | head -n1)"

            if [[ "$STRICT" = "1" ]]; then
                tries=0; chk_rc=255; j_out=""
                while (( tries < 3 )); do
                    set +e
                    j_out="$("$TOOL" -j "$f" 2>&1)"
                    chk_rc=$?
                    set -e
                    [[ $chk_rc -eq 0 || $chk_rc -eq 1 ]] && break
                    if grep -q 'HTTP status 403' <<<"$j_out"; then
                        sleep 2
                    else
                        break
                    fi
                    ((tries++))
                done

                if [[ $chk_rc -eq 0 || $chk_rc -eq 1 ]]; then
                    note="ok"; managed="zsync"
                else
                    if grep -q 'HTTP status 403' <<<"$j_out"; then
                        note="github-403"
                        managed="zsync"
                        [[ "${RELAX_ON_403:-0}" != "1" ]] && has="no"
                    elif grep -qi 'None of the artifacts matched' <<<"$j_out"; then
                        note="invalid-pattern"; has="no"; managed="unmanaged"
                    else
                        note="broken-update-info-$chk_rc"; has="no"; managed="unmanaged"
                    fi
                fi
            else
                note="ok"; managed="zsync"
            fi
        else
            if grep -qi 'could not find update information' <<<"$desc_out"; then
                note="no-update-info"
            elif grep -qi 'None of the artifacts matched' <<<"$desc_out"; then
                note="invalid-pattern"
            else
                note="tool-error"
            fi
        fi
    else
        note="no-tool"
    fi

    printf "%s %s %s %s %s %s\n" "$base" "$name" "$has" "$kind" "$note" "$managed" >>"$all_ssv"
done
shopt -u nullglob

# Write filtered views
awk 'NR==1 {print; next} $6=="zsync" || $6 ~ /^conf:/ {print}' "$all_ssv" >"$sup_ssv"
awk 'NR==1 {print; next} $6=="unmanaged" {print}' "$all_ssv" >"$man_ssv"
awk 'NR==1 {print; next} $6=="dead" {print}' "$all_ssv" >"$dead_ssv"

ORG_FILE="${ORG_FILE:-$HOME/DotCortex/appimage.org}"
if [[ -f "$ORG_FILE" ]]; then
    appimage-org-inject "$ORG_FILE" appimage-inventory-all "$all_ssv" || true
    appimage-org-inject "$ORG_FILE" appimage-inventory-supports "$sup_ssv" || true
    appimage-org-inject "$ORG_FILE" appimage-inventory-manual "$man_ssv" || true
    echo "updated blocks in $ORG_FILE:"
    echo " - appimage-inventory-all"
    echo " - appimage-inventory-supports"
    echo " - appimage-inventory-manual"
else
    echo "note: $ORG_FILE not found; skipped org injection."
fi

sup_n=$(($(wc -l <"$sup_ssv")-1))
man_n=$(($(wc -l <"$man_ssv")-1))
dead_n=$(($(wc -l <"$dead_ssv")-1))
echo "== inventory =="
echo "supports (zsync+conf): $sup_n"
echo "manual/unmanaged: $man_n"
echo "dead: $dead_n"
echo "wrote:"
echo " $all_ssv"
echo " $sup_ssv"
echo " $man_ssv"
echo " $dead_ssv"
#+END_SRC

* AppImage Org-Mode Injector
Replace the contents of a named org src block.
#+BEGIN_SRC bash :tangle all/.local/bin/appimage-org-inject :shebang "#!/usr/bin/env bash"
set -Eeuo pipefail
org="${1:?org file}"; name="${2:?block name}"; payload="${3:?payload file}"
tmp="$(mktemp)"

awk -v name="$name" -v payload="$payload" '
BEGIN { in_target=0; in_body=0 }
{
    if (!in_target) {
        if ($0 ~ "^#\\+name:[[:space:]]*" name "[[:space:]]*$") { print; in_target=1; next }
        else { print; next }
    } else if (in_target && !in_body) {
        if ($0 ~ "^#\\+BEGIN_SRC[[:space:]]") { print; in_body=1; next }
        else { print; next }
    } else if (in_target && in_body) {
        if ($0 ~ "^#\\+END_SRC[[:space:]]*$") {
            while ((getline line < payload) > 0) print line
            close(payload)
            print
            in_target=0; in_body=0; next
        } else { next }
    }
}
END {
    if (in_target || in_body) {
        while ((getline line < payload) > 0) print line
        close(payload)
        print "#+END_SRC"
    }
}
' "$org" > "$tmp"

mv "$tmp" "$org"
echo "Injected into $org :: $name"
#+END_SRC

** Conf file format

DIRECT_VERSION_CMD="..."  # for FORGE=direct
#+END_EXAMPLE

** Forgejo / Gitea self-hosted
Set FORGE=forgejo and FORGE_BASE_URL pointing to your Forgejo/Gitea instance.
The API endpoint used is: ${FORGE_BASE_URL}/api/v1/repos/${REPO}/releases?limit=1

** Rate limit strategy
- GitHub: 60 req/hour unauthenticated. apprelease stores ETag in =~/.cache/apprelease/<app>/etag=
  and sends If-None-Match header. On 304 Not Modified, no count against limit.
- GitLab: 500 req/min — generous, ETag caching also used.
- Codeberg: depends on instance settings, generally permissive.
- Use GH_TOKEN in conf file to lift GitHub limit to 5000 req/hour.

** Emulator succession (post-DMCA)
- yuzu (Nintendo, Feb 2024) → Eden (Forgejo: git.eden-emu.dev/eden-emu/eden) — DMCA'd GitHub Mar 2026
- yuzu forks (Suyu, Sudachi, etc.) → all DMCA'd
- citra (2024) → Azahar (github.com/azahar-emu/azahar) + AzaharPlus
- Lime3DS → archived, absorbed into Azahar
- Ryujinx → mirror at github.com/enbyte/ryujinx-mirror-cubed (unstable)
