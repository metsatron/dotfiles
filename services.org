* Services Loom
This loom tracks small service wrappers and helper scripts that live in
the =.local/bin= overlays and are deployed via stow.

* Early OOM
Scripts to apply a conservative, balanced OOM policy.
The X230 runs systemd; the T480s runs Devuan/sysvinit.
Both expose the same =MODE= knob (=balanced= | =proactive=).

** earlyoom-balanced — X230 (systemd drop-in)
Writes a systemd drop-in at =/etc/systemd/system/earlyoom.service.d/override.conf=
and restarts the service.  Avoids =/etc/default/= quoting issues by setting
=ExecStart= directly.

#+BEGIN_SRC bash :tangle x230/.local/bin/earlyoom-balanced :shebang "#!/usr/bin/env bash"
set -euo pipefail

MODE="${MODE:-balanced}"   # balanced or proactive

# Kill this stuff first if we MUST kill something:
#  - non-critical XFCE daemons
#  - media / background apps
#  - tools that autosave (Obsidian, Telegram)
PREFER='^(xfce4-screensaver|xfce4-clipman|xfdesktop|xfce4-notifyd|kdeconnectd|fluidsynth|jellyfin|[Oo]bsidian|Telegram|WebExtensions|qmmp)$'

# Protect these at all costs:
#  - browsers (no autosave guarantees)
#  - Codium / Code (no autosave guarantees in your current setup)
#  - file managers + GVFS plumbing
#  - file transfer / backup tools (rsync, etc.)
#  - xfce4-session so we don't drop the whole session
AVOID='^(firedragon|floorp|zen|ungoogled-chromium|firefox|codium|code|thunar|gvfs.*|rsync|scp|sftp|rclone|syncthing|borg|restic|xfce4-session|barrier|barriers|signal-desktop|Betterbird|betterbird|thunderbird)$'

if [[ "$MODE" = "balanced" ]]; then
  M_TERM=5;  M_KILL=3;  S_TERM=15; S_KILL=10
elif [[ "$MODE" = "proactive" ]]; then
  M_TERM=7;  M_KILL=5;  S_TERM=20; S_KILL=10
else
  echo "Unknown MODE='$MODE' (use balanced or proactive)" >&2
  exit 1
fi

DROPIN_DIR="/etc/systemd/system/earlyoom.service.d"
DROPIN="${DROPIN_DIR}/override.conf"

sudo install -d -m 0755 "$DROPIN_DIR"

# Write drop-in atomically
TMP="$(mktemp)"
cat >"$TMP" <<EOF
[Service]
EnvironmentFile=
ExecStart=
ExecStart=/usr/bin/earlyoom -r 3600 -m ${M_TERM},${M_KILL} -s ${S_TERM},${S_KILL} \
  --prefer ${PREFER} \
  --avoid  ${AVOID}
OOMScoreAdjust=-100
Nice=-20
EOF
sudo install -m 0644 "$TMP" "$DROPIN"
rm -f "$TMP"

sudo systemctl daemon-reload
sudo systemctl restart earlyoom

echo "Applied Early OOM policy: mem ${M_TERM}%/${M_KILL}%, swap ${S_TERM}%/${S_KILL}%"
echo "Prefer: ${PREFER}"
echo "Avoid:  ${AVOID}"

# Show the running command and thresholds
if pidof earlyoom >/dev/null; then
  tr '\0' ' ' </proc/$(pidof earlyoom)/cmdline; echo
  journalctl -u earlyoom -b --no-pager | tail -n 20
else
  echo "earlyoom is not running" >&2
  exit 2
fi
#+END_SRC

** hermes-gateway — T480s (Devuan / sysvinit)
The Hermes Gateway is a user-level process. We manage it via a wrapper script and @reboot cron.

#+BEGIN_SRC bash :tangle t480s/.local/bin/hermes-gateway :shebang "#!/usr/bin/env bash"
set -euo pipefail

PID_FILE="$HOME/.hermes/gateway.pid"
LOG_FILE="$HOME/.hermes/gateway.log"
CMD="$HOME/.local/bin/hermes gateway run"

# Helper to extract PID from JSON PID file
get_pid() {
 local pid_file="$1"
 if command -v jq >/dev/null 2>&1; then
 jq -r .pid "$pid_file" 2>/dev/null || cat "$pid_file"
 else
 # Fallback: grep for "\"pid\":" in JSON
 grep -o '"pid":[0-9]*' "$pid_file" 2>/dev/null | cut -d: -f2 || cat "$pid_file"
 fi
}

start_gateway() {
 if [ -f "$PID_FILE" ] && kill -0 $(get_pid "$PID_FILE") 2>/dev/null; then
 echo "Gateway is already running (PID $(get_pid "$PID_FILE"))"
 return 0
 fi
 echo "Starting Hermes Gateway..."
 nohup $CMD >"$LOG_FILE" 2>&1 &
 echo $! >"$PID_FILE"
 echo "Gateway started with PID $!"
}

stop_gateway() {
 if [ -f "$PID_FILE" ]; then
 PID=$(get_pid "$PID_FILE")
 echo "Stopping Gateway (PID $PID)..."
 kill $PID || true
 rm -f "$PID_FILE"
 echo "Gateway stopped."
 else
 echo "Gateway is not running."
 fi
}

case "${1:-}" in
  start)    start_gateway ;;
  stop)     stop_gateway ;;
  restart)  stop_gateway; start_gateway ;;
 status) 
 if [ -f "$PID_FILE" ] && kill -0 $(get_pid "$PID_FILE") 2>/dev/null; then
 echo "Gateway is running (PID $(get_pid "$PID_FILE"))"
 else
 echo "Gateway is stopped."
 fi
 ;;
 *)
 echo "Usage: ${0##*/} {start|stop|restart|status}"
 exit 1
 ;;
esac
#+END_SRC

** swaptop – top by VmSwap (btop-ish)
#+BEGIN_SRC bash :tangle linux/.local/bin/swaptop :shebang "#!/usr/bin/env bash"
set -euo pipefail

interval="${1:-5}"

# === Colors – keep in sync with swapfetch ===

# Nerd Font style: same palette as swapfetch
C_HDR=$'\e[1;36m'     # cyan bold (headers)
C_LABEL=$'\e[1;33m'   # yellow bold (labels)
C_VAL=$'\e[1;37m'     # white bold  (values)
C_DIM=$'\e[2;37m'     # dim white   (background / secondary)
C_OK=$'\e[1;32m'      # green
C_WARM=$'\e[1;33m'    # yellow
C_HOT=$'\e[1;31m'     # red
C_RESET=$'\e[0m'
C_BOLD=$'\033[1m'

# Bar background: “12th” color – aligned with swapfetch’s style
C_BAR_BG=$'\e[38;5;11m'

# Logo/header color
: "${C_LOGO:=$C_HDR}"

gauge_color_for() {
  local pct=$1
  if   (( pct >= 90 )); then printf '%s' "$C_HOT"
  elif (( pct >= 70 )); then printf '%s' "$C_WARM"
  else                       printf '%s' "$C_OK"
  fi
}

gauge_bar() {
  local pct=$1
  local width=${2:-24}
  local col="${3:-$C_OK}"      # filled color
  local bg="${4:-$C_BAR_BG}"   # unfilled “background”

  ((pct < 0))   && pct=0
  ((pct > 100)) && pct=100

  local filled=$(( (pct * width + 50) / 100 ))
  ((filled > width)) && filled=$width

  local filled_str="" empty_str=""
  local i
  for ((i=1; i<=filled; i++)); do
    filled_str+="█"
  done
  for ((; i<=width; i++)); do
    empty_str+="█"
  done

  printf '%b' "${col}${filled_str}${bg}${empty_str}${C_RESET}"
}

# Use tput if available so we can move the cursor without nuking scrollback
HAS_TPUT=0
if command -v tput >/dev/null 2>&1; then
  HAS_TPUT=1
  tput civis 2>/dev/null || true
  trap 'tput cnorm 2>/dev/null || true; exit' INT TERM
fi

# How many process rows we printed last time
LAST_ROWS=0

print_row() {
  local idx="$1"
  local line pid kb pct col bar mib cmdline

  line=${_rows[idx]}
  set -- $line
  pid=$1
  kb=$2

  [[ -z "$pid" || -z "$kb" ]] && return 0

  # percent of total swap
  pct=$(( (kb * 100 + local_total_swap_kb/2) / local_total_swap_kb ))
  ((pct < 0))   && pct=0
  ((pct > 100)) && pct=100

  col="$(gauge_color_for "$pct")"
  bar="$(gauge_bar "$pct" 24 "$col" "$C_BAR_BG")"

  # human readable MiB
  mib=$(awk -v k="$kb" 'BEGIN {printf "%.1f", k/1024}')

  # command line (shortened single-line)
  cmdline=$(ps -p "$pid" -o comm=,args= 2>/dev/null || true)
  cmdline=${cmdline# }
  cmdline=${cmdline//$'\n'/ }
  cmdline=${cmdline:0:60}

  printf "%7s %7s MiB %3d%% %s %s\n" \
    "$pid" "$mib" "$pct" "$bar" "$cmdline"
}

while :; do
  # Terminal height (only if tput exists)
  if (( HAS_TPUT )); then
    term_lines=$(tput lines 2>/dev/null || echo 24)
  else
    term_lines=24
  fi

  # Total swap in KiB
  local_total_swap_kb=$(awk '/^SwapTotal:/ {print $2}' /proc/meminfo)
  (( local_total_swap_kb == 0 )) && local_total_swap_kb=1

  # Collect processes sorted by VmSwap DESC
  mapfile -t _rows < <(
    awk '/^VmSwap:/ {print FILENAME, $2}' /proc/[0-9]*/status 2>/dev/null \
      | sed 's#/proc/\([0-9]*\)/status#\1#' \
      | sort -k2,2nr
  )

  rows_len=${#_rows[@]}
  if (( rows_len == 0 )); then
    current_rows=0
  else
    # 1 title + 1 header + 1 separator + 1 breathing line = 4
    max_rows=$((term_lines - 4))
    ((max_rows < 1)) && max_rows=1
    current_rows=$rows_len
    ((current_rows > max_rows)) && current_rows=$max_rows
  fi

  if (( HAS_TPUT )); then
    # --- Header lines (each one: move, clear line, print) ---

    tput cup 0 0 2>/dev/null
    printf '\033[K'
    printf '%b%s%b  (interval %ss)\n' \
      "$C_HDR" "swaptop – top VmSwap users" "$C_RESET" "$interval"

    tput cup 1 0 2>/dev/null
    printf '\033[K'
    printf '%b%-7s %-8s %-8s %-5s %-s%b\n' \
      "$C_LABEL" "PID" "SWAP" "PCT" "BAR" "COMMAND" "$C_RESET"

    tput cup 2 0 2>/dev/null
    printf '\033[K'
    printf '%s\n' "--------------------------------------------------------------------------"

    if (( current_rows == 0 )); then
      # No rows: print a single status line
      tput cup 3 0 2>/dev/null
      printf '\033[K'
      printf "   (no processes with VmSwap)\n"
    else
      # Print each row at a fixed line, clearing only that line
      for ((idx=0; idx<current_rows; idx++)); do
        line_no=$((3 + idx))
        tput cup "$line_no" 0 2>/dev/null
        printf '\033[K'
        print_row "$idx"
      done
    fi

    # If we had more rows last time, blank the leftovers
    if (( LAST_ROWS > current_rows )); then
      for ((idx=current_rows; idx<LAST_ROWS; idx++)); do
        line_no=$((3 + idx))
        tput cup "$line_no" 0 2>/dev/null
        printf '\033[K'
      done
    fi

  else
    # Fallback: dumb terminal – just print a fresh block each tick
    printf '%b%s%b  (interval %ss)\n' \
      "$C_HDR" "swaptop – top VmSwap users" "$C_RESET" "$interval"
    printf '%b%-7s %-8s %-8s %-5s %-s%b\n' \
      "$C_LABEL" "PID" "SWAP" "PCT" "BAR" "COMMAND" "$C_RESET"
    printf '%s\n' "--------------------------------------------------------------------------"

    if (( current_rows == 0 )); then
      printf "   (no processes with VmSwap)\n"
    else
      for ((idx=0; idx<current_rows; idx++)); do
        print_row "$idx"
      done
    fi
  fi

  LAST_ROWS=$current_rows
  sleep "$interval"
done
#+END_SRC

** xfce4-refresh – recycle XFCE desktop daemons via host-wrap
#+BEGIN_SRC bash :tangle x230/.local/bin/xfce4-refresh :shebang "#!/usr/bin/env bash"
set -euo pipefail

log() {
  printf '[xfce4-refresh] %s\n' "$*"
}

kill_if_running() {
  local name="$1"
  if pgrep -x "$name" >/dev/null 2>&1; then
    log "Killing $name…"
    pkill -x "$name" || true
  fi
}

start_with_host_wrap_if_missing() {
  local name="$1"
  if ! pgrep -x "$name" >/dev/null 2>&1; then
    if command -v host-wrap >/dev/null 2>&1; then
      log "Starting $name via host-wrap…"
      nohup host-wrap "$name" >/dev/null 2>&1 &
    else
      log "host-wrap not found; starting $name directly…"
      nohup "$name" >/dev/null 2>&1 &
    fi
  fi
}

log "Refreshing XFCE desktop daemons"

# 1) Kill screensaver and leave it dead - handle truncated comm name
if pgrep -x xfce4-screensav >/dev/null 2>&1 || pgrep -f /usr/bin/xfce4-screensaver >/dev/null 2>&1; then
  log "Killing xfce4-screensaver..."
  pkill -x xfce4-screensav || pkill -f /usr/bin/xfce4-screensaver || true
fi

# 2) Clipman: kill + restart (via host-wrap)
kill_if_running xfce4-clipman
start_with_host_wrap_if_missing xfce4-clipman

# 3) Desktop: kill + restart (via host-wrap)
kill_if_running xfdesktop
start_with_host_wrap_if_missing xfdesktop

# 4) Notifications: kill + restart (via host-wrap)
if command -v xfce4-notifyd >/dev/null 2>&1; then
  kill_if_running xfce4-notifyd
  start_with_host_wrap_if_missing xfce4-notifyd
fi

# 5) xfrun4 - kill if it is hanging around - do not restart
kill_if_running xfrun4

# 6) Panel + settings: delegate to canonical healer
if command -v xfce4-heal >/dev/null 2>&1; then
  log "Delegating panel + settings restart to xfce4-heal…"
  xfce4-heal
elif command -v xfce4-panel >/dev/null 2>&1; then
  # Fallback path if xfce4-heal is missing for some reason
  if pgrep -x xfce4-panel >/dev/null 2>&1; then
    log "xfce4-heal missing; trying xfce4-panel -r fallback…"
    if ! xfce4-panel -r >/dev/null 2>&1; then
      log "xfce4-panel -r failed, trying kill + restart via host-wrap/direct…"
      kill_if_running xfce4-panel
      start_with_host_wrap_if_missing xfce4-panel
    fi
  else
    log "xfce4-heal missing; xfce4-panel not running, starting fresh via host-wrap…"
    start_with_host_wrap_if_missing xfce4-panel
  fi
fi

log "Done."
#+END_SRC

** xfce4-refresh-guardian – early panel refresh on high swap + panel revive
#+BEGIN_SRC bash :tangle x230/.local/bin/xfce4-refresh-guardian :shebang "#!/usr/bin/env bash"
#!/usr/bin/env bash
# xfce4-refresh-guardian – early refresh trigger based on swap
# and “is the panel actually visible?”

set -euo pipefail

SWAP_LIMIT="${SWAP_LIMIT:-80}"

DO_RESTART=0
if [[ "${1:-}" == "--restart" || "${1:-}" == "-r" ]]; then
  DO_RESTART=1
fi

# --- helper: is the xfce4-panel actually alive (proc + window)? ---
panel_is_alive=0

if pgrep -x xfce4-panel >/dev/null 2>&1; then
  # We have a process – now check for a real panel window if wmctrl exists.
  if command -v wmctrl >/dev/null 2>&1; then
    if wmctrl -lx 2>/dev/null | grep -qi "xfce4-panel.Xfce4-panel"; then
      panel_is_alive=1
    else
      printf '[xfce4-refresh-guardian] xfce4-panel process found, but no panel window.\n'
    fi
  else
    # No wmctrl? fall back to trusting the process
    panel_is_alive=1
  fi
else
  printf '[xfce4-refresh-guardian] No xfce4-panel process.\n'
fi

# If the panel is missing (no process OR no window), healing is top priority.
if (( panel_is_alive == 0 )); then
  printf '[xfce4-refresh-guardian] xfce4-panel is NOT alive, attempting heal via xfce4-heal…\n'
  if command -v xfce4-heal >/dev/null 2>&1; then
    xfce4-heal
    # Give it a moment, then re-check both process + window.
    sleep 1

    panel_is_alive_after=0
    if pgrep -x xfce4-panel >/dev/null 2>&1; then
      if command -v wmctrl >/dev/null 2>&1; then
        if wmctrl -lx 2>/dev/null | grep -qi "xfce4-panel.Xfce4-panel"; then
          panel_is_alive_after=1
        fi
      else
        panel_is_alive_after=1
      fi
    fi

    if (( panel_is_alive_after == 1 )); then
      printf '[xfce4-refresh-guardian] Panel is now running (process + window detected).\n'
    else
      printf '[xfce4-refresh-guardian] Heal attempted, but panel still not detected.\n'
    fi
  else
    printf '[xfce4-refresh-guardian] xfce4-heal not found; cannot restore panel.\n'
  fi

  # Panel health is a higher priority than swap tuning – stop here.
  exit 0
fi

# --- at this point: panel_is_alive == 1, proceed with swap threshold logic ---

# Read swap from /proc/meminfo (KiB)
swap_total_kb=$(awk '/^SwapTotal:/ {print $2}' /proc/meminfo)
swap_free_kb=$(awk '/^SwapFree:/  {print $2}' /proc/meminfo)

if (( swap_total_kb == 0 )); then
  SWAP_PCT=0
else
  swap_used_kb=$((swap_total_kb - swap_free_kb))
  SWAP_PCT=$(( (swap_used_kb * 100 + swap_total_kb/2) / swap_total_kb ))
fi

printf '[xfce4-refresh-guardian] THRESHOLD: SWAP=%d%% (limit %d%%)\n' \
  "$SWAP_PCT" "$SWAP_LIMIT"

if (( SWAP_PCT >= SWAP_LIMIT )); then
  if (( DO_RESTART )); then
    if command -v xfce4-refresh >/dev/null 2>&1; then
      printf '[xfce4-refresh-guardian] Triggering xfce4-refresh (early recycle)…\n'
      xfce4-refresh
    else
      printf '[xfce4-refresh-guardian] xfce4-refresh not found, nothing to do.\n'
    fi
  else
    printf '[xfce4-refresh-guardian] Would run xfce4-refresh (dry run).\n'
  fi
else
  printf '[xfce4-refresh-guardian] Below threshold, nothing to do.\n'
fi
#+END_SRC

** xfce-show-desktop – toggle show-desktop and ensure xfdesktop
#+BEGIN_SRC bash :tangle x230/.local/bin/xfce4-show-desktop :shebang "#!/usr/bin/env bash"
set -euo pipefail

log() {
  printf '[xfce-show-desktop] %s\n' "$*"
}

ensure_xfdesktop() {
  if ! pgrep -x xfdesktop >/dev/null 2>&1; then
    log "xfdesktop not running, starting…"
    if command -v host-wrap >/dev/null 2>&1; then
      nohup host-wrap xfdesktop >/dev/null 2>&1 &
    else
      nohup xfdesktop >/dev/null 2>&1 &
    fi
  fi
}

toggle_show_desktop() {
  if ! command -v wmctrl >/dev/null 2>&1; then
    log "wmctrl not found, cannot toggle show desktop"
    exit 1
  fi

  # _NET_SHOWING_DESKTOP: 0 = normal, 1 = show desktop
  local state
  state=$(xprop -root _NET_SHOWING_DESKTOP 2>/dev/null | awk '{print $3}' || echo 0)

  if [[ "$state" = "0" ]]; then
    log "Enabling show desktop"
    wmctrl -k on
  else
    log "Disabling show desktop"
    wmctrl -k off
  fi
}

ensure_xfdesktop
toggle_show_desktop

** swap-heal – safe swap purge + desktop/Flatpak heal
#+BEGIN_SRC bash :tangle x230/.local/bin/swap-heal :shebang "#!/usr/bin/env bash"
#!/usr/bin/env bash
set -euo pipefail

log() { printf '[swap-heal] %s\n' "$*"; }

# 1) Inspect RAM + swap
mem_available_kb=$(awk '/^MemAvailable:/ {print $2}' /proc/meminfo)
swap_total_kb=$(awk '/^SwapTotal:/     {print $2}' /proc/meminfo)
swap_free_kb=$(awk '/^SwapFree:/      {print $2}' /proc/meminfo)
swap_used_kb=$((swap_total_kb - swap_free_kb))

log "MemAvailable: ${mem_available_kb} KiB"
log "Swap used:    ${swap_used_kb} KiB"

if (( swap_total_kb == 0 )); then
  log "No swap configured, nothing to do."
  exit 0
fi

# Simple safety rule: need at least as much free RAM as swap in use
needed_kb=${swap_used_kb}

if (( mem_available_kb < needed_kb )) && [[ "${SWAP_HEAL_FORCE:-0}" != 1 ]]; then
  log "Not enough free RAM to safely clear swap (need ${needed_kb} KiB)."
  log "Set SWAP_HEAL_FORCE=1 if you really want to override this (risk of OOM)."
  exit 1
fi

if [[ "${SWAP_HEAL_FORCE:-0}" == 1 ]]; then
  log "SWAP_HEAL_FORCE=1 set, ignoring RAM safety check. This may OOM the system."
else
  log "Enough RAM available, proceeding with swap purge."
fi

# 2) Drop caches
log "Dropping caches (vm.drop_caches=3)…"
if ! sudo sh -c 'echo 3 > /proc/sys/vm/drop_caches'; then
  log "WARNING: could not write vm.drop_caches; continuing anyway."
fi

# 3) swapoff + zramswap restart + swapon
log "swapoff -a…"
sudo swapoff -a

log "Restarting zramswap.service if present…"
if ! sudo systemctl restart zramswap.service 2>/dev/null; then
  log "zramswap.service restart failed or unit not present; continuing."
fi

log "swapon -a…"
sudo swapon -a

# Make sure zram actually ends up as swap if it exists
if [ -e /dev/zram0 ]; then
  disksize=$(cat /sys/block/zram0/disksize 2>/dev/null || echo 0)
  if [ "$disksize" -gt 0 ] && ! grep -q '^/dev/zram0' /proc/swaps; then
    log "Enabling zram swap on /dev/zram0…"
    if ! sudo swapon /dev/zram0 2>/dev/null; then
      log "WARNING: swapon /dev/zram0 failed; zram still not active."
    fi
  else
    log "zram0 already active or no disksize set, skipping swapon."
  fi
else
  log "/dev/zram0 not present; nothing to enable."
fi

# 4) Heal XFCE desktop
if command -v xfce4-heal >/dev/null 2>&1; then
  log "Running xfce4-heal…"
  xfce4-heal
else
  log "xfce4-heal not found; skipping."
fi

# 5) Heal Flatpak bridge + fontcache
if command -v flatpak-desktop-bridge >/dev/null 2>&1; then
  log "Refreshing Flatpak desktop bridge…"
  flatpak-desktop-bridge
fi

if command -v flatpak-fontcache-refresh >/dev/null 2>&1; then
  log "Refreshing Flatpak font cache…"
  flatpak-fontcache-refresh
fi

log "swap-heal ritual complete."
#+END_SRC
