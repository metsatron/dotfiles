# ffetch: fastfetch with WezTerm overlay when inside WezTerm
ffetch() {
  # Optional debug - only fires if you export FFETCH_DEBUG=1
  if [[ -n ${FFETCH_DEBUG-} ]]; then
    printf '[ffetch] TERM="%s" WEZTERM_EXECUTABLE="%s" WEZTERM_VERSION="%s" WEZTERM_PANE="%s" TERM_PROGRAM="%s"\n' \
      "${TERM-}" "${WEZTERM_EXECUTABLE-}" "${WEZTERM_VERSION-}" "${WEZTERM_PANE-}" "${TERM_PROGRAM-}" >&2
  fi

  if [[ ${TERM-} == wezterm ]] \
     || [[ -n ${WEZTERM_PANE-}${TERM_PROGRAM-} ]]; then
    command fastfetch --config "$HOME/.config/fastfetch/wezterm.jsonc" "$@"
  else
    command fastfetch "$@"
  fi
}

#+BEGIN_SRC bash :tangle all/.local/bin/ssh-agent-ensure :shebang "#!/usr/bin/env bash" :mkdirp yes
#!/usr/bin/env bash
set -euo pipefail

AGENT_ENV="${XDG_CACHE_HOME:-$HOME/.cache}/ssh-agent.env"

agent_live() {
  [ -n "${SSH_AUTH_SOCK:-}" ] || return 1
  ssh-add -l >/dev/null 2>&1 && return 0
  [ "$?" -eq 1 ]
}

load_agent_env() {
  [ -r "$AGENT_ENV" ] || return 1
  # shellcheck disable=SC1090
  . "$AGENT_ENV"
  export SSH_AUTH_SOCK SSH_AGENT_PID
  agent_live
}

start_agent() {
  mkdir -p "$(dirname "$AGENT_ENV")"
  ssh-agent -s | sed '/^echo /d' >"$AGENT_ENV"
  chmod 600 "$AGENT_ENV"
  # shellcheck disable=SC1090
  . "$AGENT_ENV"
  export SSH_AUTH_SOCK SSH_AGENT_PID
}

if ! agent_live && ! load_agent_env; then
  start_agent
fi

if [ "${1:-}" = "--print" ]; then
  printf 'export SSH_AUTH_SOCK=%q\n' "$SSH_AUTH_SOCK"
  printf 'export SSH_AGENT_PID=%q\n' "$SSH_AGENT_PID"
fi

** SSHFS remote home launcher
#+BEGIN_SRC bash :tangle linux/.local/bin/sshfs-open-home :shebang "#!/usr/bin/env bash" :mkdirp yes
set -euo pipefail

HOST="${1:-}"
MOUNT_NAME="${2:-${1:-}}"

if [ -z "$HOST" ] || [ -z "$MOUNT_NAME" ]; then
  printf 'Usage: %s HOST [MOUNT_NAME]\n' "${0##*/}" >&2
  exit 2
fi

MNT="$HOME/mnt/$MOUNT_NAME"
mkdir -p "$MNT"

if mountpoint -q "$MNT"; then
  if timeout 2 ls "$MNT" >/dev/null 2>&1; then
    exec thunar "$MNT"
  fi
  (fusermount3 -uz "$MNT" || fusermount -uz "$MNT" || true) >/dev/null 2>&1
fi

if [ -n "${DISPLAY:-}${WAYLAND_DISPLAY:-}" ]; then
  for askpass in \
    "${SSH_ASKPASS:-}" \
    /usr/libexec/seahorse/ssh-askpass \
    /usr/libexec/gcr4-ssh-askpass \
    /usr/libexec/gcr-ssh-askpass \
    /usr/libexec/ssh-askpass/x11-ssh-askpass
  do
    [ -n "$askpass" ] || continue
    [ -x "$askpass" ] || continue
    export SSH_ASKPASS="$askpass"
    export SSH_ASKPASS_REQUIRE="${SSH_ASKPASS_REQUIRE:-prefer}"
    break
  done
fi

if sshfs "${HOST}:" "$MNT" \
  -o reconnect,ServerAliveInterval=15,ServerAliveCountMax=3,ConnectTimeout=8,ConnectionAttempts=1,Compression=no,idmap=user,dir_cache=yes,dcache_timeout=30,dcache_stat_timeout=30,dcache_link_timeout=30,dcache_dir_timeout=30,auto_cache,attr_timeout=30,entry_timeout=30,negative_timeout=5,max_conns=4 \
  </dev/null
then
  exec thunar "$MNT"
fi

if command -v notify-send >/dev/null 2>&1; then
  notify-send "Mount ${HOST} failed" "Try ssh ${HOST} in a terminal to confirm access."
fi

exit 1

# ~/.local/bin/claude is canonical. If it is absent, fall back to the newest
# native Claude binary under ~/.config/Claude or Antigravity.
if ! command -v claude >/dev/null 2>&1; then
  claude_native_dir=

  for claude_base in \
    "$HOME/.config/Claude/claude-code" \
    "$HOME/.config/Claude/claude-code-vm"
  do
    [ -d "$claude_base" ] || continue
    claude_native_dir="$(find "$claude_base" -mindepth 1 -maxdepth 1 -type d 2>/dev/null | sort -V | tail -n 1)"
    if [ -n "$claude_native_dir" ] && [ -x "$claude_native_dir/claude" ]; then
      export PATH="$claude_native_dir:$PATH"
      break
    fi
  done

  if [ -z "$claude_native_dir" ] && [ -d "$HOME/.antigravity/extensions" ]; then
    _claude_dir="$(find "$HOME/.antigravity/extensions" -path '*/resources/native-binary' -type d 2>/dev/null | sort -V | tail -n 1)"
    if [ -n "${_claude_dir:-}" ] && [ -x "$_claude_dir/claude" ]; then
      export PATH="$_claude_dir:$PATH"
    fi
    unset _claude_dir
  fi

  unset claude_base claude_native_dir
fi

  local name="$1" val new elem

# ~/.zsh_00_guix_fpath — Guix zsh function dirs into fpath EARLY
# compinit, add-zsh-hook, is-at-least are autoload functions that live
# in zsh's function dirs.  When zsh is installed via Guix, these live
# inside the Guix store, not under /usr/share/zsh.  The compiled-in

typeset -gU fpath

  _guix_zsh_store="$(readlink -f "$(command -v zsh)" 2>/dev/null)"
fi
if [[ -n "$_guix_zsh_store" && "$_guix_zsh_store" == /gnu/store/* ]]; then
  _store_prefix="${_guix_zsh_store%/bin/zsh}"
  _zsh_fn_base="$_store_prefix/share/zsh"
  # Add the versioned functions dirs (e.g. share/zsh/5.9/functions/*)
  for d in "$_zsh_fn_base"/*/functions(N/); do

      for sub2 in "$sub"/*(N/); do

The =zz_fzf= prefix ensures this loads AFTER all other =.zsh_*= files
(=completion=, =exports=, =plugins=, =prompt=), since =z= sorts last.
=zoxide init zsh= lives here — not in =.zsh_exports= — so it runs AFTER
=zsh-syntax-highlighting= and =zsh-autosuggestions=, which otherwise clobber
=chpwd_functions= and trigger the zoxide doctor warning.
#+BEGIN_SRC sh :tangle all/.zsh_zz_fzf :tangle-mode (identity #o644)
#!/bin/zsh
# ~/.zsh_zz_fzf -- fzf keybindings and zoxide bridge (loads after all plugins)
[[ $- != *i* ]] && return 0
[[ -n ${DOTCORTEX_ZSH_FZF_INIT-} ]] && return 0
typeset -g DOTCORTEX_ZSH_FZF_INIT=1

# --- fzf defaults ---
export FZF_DEFAULT_OPTS="
  --height=40%
  --layout=reverse
  --border=rounded
  --info=inline
  --prompt='> '
  --pointer='>'
  --marker='+'
  --color=bg+:#454545,bg:#242424,fg:#d2d4d4,fg+:#ffffff
  --color=hl:#268bd2,hl+:#15b8cf,info:#b58900,marker:#4caf50
  --color=prompt:#cb4b16,spinner:#a681d6,pointer:#268bd2,header:#6a6a6a
  --color=border:#454545,gutter:#242424
  --bind=ctrl-u:half-page-up,ctrl-d:half-page-down
  --bind=ctrl-y:preview-up,ctrl-e:preview-down
"

# Use fd if available for file search (faster, respects .gitignore)
if (( $+commands[fd] )); then
  export FZF_DEFAULT_COMMAND='fd --type f --hidden --follow --exclude .git'
  export FZF_CTRL_T_COMMAND="$FZF_DEFAULT_COMMAND"
  export FZF_ALT_C_COMMAND='fd --type d --hidden --follow --exclude .git'
fi

# CTRL-T preview with bat or cat
export FZF_CTRL_T_OPTS="
  --preview '([[ -d {} ]] && ls -la --color=always {} || bat --style=plain --color=always -n {} 2>/dev/null || cat {}) 2>/dev/null | head -200'
  --preview-window=right:50%:wrap
"

# ALT-C preview with ls
export FZF_ALT_C_OPTS="
  --preview 'ls -la --color=always {} 2>/dev/null | head -50'
  --preview-window=right:50%:wrap
"

# CTRL-R history search opts
export FZF_CTRL_R_OPTS="
  --preview 'echo {}'
  --preview-window=down:3:wrap
  --bind 'ctrl-y:execute-silent(echo -n {2..} | xclip -selection clipboard)+abort'
  --header 'CTRL-Y to copy'
"

# --- Source fzf keybindings (NOT completion.zsh — it rebinds ^I and breaks zoxide tab) ---
_fzf_sourced=0
for fzf_base in \
  "${GUIX_PROFILE:-}/share/fzf" \
  "$HOME/.guix-extra-profiles/core/core/share/fzf" \
  "/usr/share/doc/fzf/examples" \
  "/usr/share/fzf" \
  "/usr/local/share/fzf" \
  "$HOME/.fzf"
do
  if [[ -r "$fzf_base/key-bindings.zsh" ]]; then
    source "$fzf_base/key-bindings.zsh"
    _fzf_sourced=1
    break
  elif [[ -r "$fzf_base/shell/key-bindings.zsh" ]]; then
    source "$fzf_base/shell/key-bindings.zsh"
    _fzf_sourced=1
    break
  fi
done
unset _fzf_sourced fzf_base

# --- zoxide init (must be last — after all plugins that touch chpwd_functions) ---
(( $+commands[zoxide] )) && eval "$(zoxide init zsh)"

# --- zoxide + fzf integration ---
# Interactive directory picker: zi (or cdi) uses zoxide's database with fzf
if (( $+commands[zoxide] )) && (( $+commands[fzf] )); then
  # zi is already provided by zoxide init --cmd z as 'zi'
  # Add a custom cdi alias with preview
  cdi() {
    local dir
    dir="$(zoxide query -l 2>/dev/null | fzf \
      --height=40% \
      --layout=reverse \
      --border \
      --preview 'ls -la --color=always {} 2>/dev/null | head -30' \
      --preview-window=right:40%:wrap \
      --prompt='cd> ')" && cd "$dir"
  }
fi
#+END_SRC

#+BEGIN_SRC sh :tangle all/.bash_fzf :tangle-mode (identity #o644)
#!/bin/bash
# ~/.bash_fzf -- fzf keybindings and completions for bash
[[ $- != *i* ]] && return 0

# --- fzf defaults (same as zsh) ---
export FZF_DEFAULT_OPTS="
  --height=40%
  --layout=reverse
  --border=rounded
  --info=inline
  --prompt='> '
  --pointer='>'
  --marker='+'
  --color=bg+:#454545,bg:#242424,fg:#d2d4d4,fg+:#ffffff
  --color=hl:#268bd2,hl+:#15b8cf,info:#b58900,marker:#4caf50
  --color=prompt:#cb4b16,spinner:#a681d6,pointer:#268bd2,header:#6a6a6a
  --color=border:#454545,gutter:#242424
  --bind=ctrl-u:half-page-up,ctrl-d:half-page-down
  --bind=ctrl-y:preview-up,ctrl-e:preview-down
"

if command -v fd >/dev/null 2>&1; then
  export FZF_DEFAULT_COMMAND='fd --type f --hidden --follow --exclude .git'
  export FZF_CTRL_T_COMMAND="$FZF_DEFAULT_COMMAND"
  export FZF_ALT_C_COMMAND='fd --type d --hidden --follow --exclude .git'
fi

export FZF_CTRL_T_OPTS="
  --preview '([[ -d {} ]] && ls -la --color=always {} || bat --style=plain --color=always -n {} 2>/dev/null || cat {}) 2>/dev/null | head -200'
  --preview-window=right:50%:wrap
"

export FZF_ALT_C_OPTS="
  --preview 'ls -la --color=always {} 2>/dev/null | head -50'
  --preview-window=right:50%:wrap
"

# Source fzf keybindings and completion
for fzf_base in \
  "/usr/share/doc/fzf/examples" \
  "/usr/share/fzf" \
  "/usr/local/share/fzf" \
  "$HOME/.fzf"
do
  [ -r "$fzf_base/key-bindings.bash" ] && source "$fzf_base/key-bindings.bash"
  [ -r "$fzf_base/completion.bash" ] && source "$fzf_base/completion.bash"
  [ -r "$fzf_base/shell/key-bindings.bash" ] && source "$fzf_base/shell/key-bindings.bash"
  [ -r "$fzf_base/shell/completion.bash" ] && source "$fzf_base/shell/completion.bash"
done
unset fzf_base

# Fallback
if ! bind -x 2>/dev/null | grep -q fzf 2>/dev/null; then
  if command -v fzf >/dev/null 2>&1; then
    eval "$(fzf --bash 2>/dev/null)" || true
  fi
fi
#+END_SRC
