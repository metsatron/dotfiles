#!/bin/bash
# ~/.bash_prompt
# --------------------------------------------------
# Mètsàtron's Bash Prompt
# --------------------------------------------------

if [ "$METSA_PROMPT_THEME" = "fancy" ]; then
  PS1='\[\e[35m\]\u\[\e[0m\]@\[\e[36m\]\h\[\e[0m\]:\[\e[32m\]\w\[\e[0m\]\$ '
else
  PS1='\u@\h:\w\$ '
fi

# Debian chroot detection
if [ -z "${debian_chroot:-}" ] && [ -r /etc/debian_chroot ]; then
    debian_chroot=$(cat /etc/debian_chroot)
fi

# Prompt color logic
case "$TERM" in
    xterm-color|*-256color) color_prompt=yes;;
esac

# Force color prompt (commented)
#force_color_prompt=yes

# Main PS1 logic
if [ -n "$force_color_prompt" ]; then
    if [ -x /usr/bin/tput ] && tput setaf 1 >&/dev/null; then
	color_prompt=yes
    else
	color_prompt=
    fi
fi

if [ "$color_prompt" = yes ]; then
    PS1='${debian_chroot:+($debian_chroot)}\[\033[01;32m\]\u@\h\[\033[00m\]:\[\033[01;34m\]\w\[\033[00m\]$( [ $? -ne 0 ] && echo "\[\033[01;31m\]✗\[\033[00m\]" )\$ '
else
    PS1='${debian_chroot:+($debian_chroot)}\u@\h:\w$( [ $? -ne 0 ] && echo "✗" )\$ '
fi
unset color_prompt force_color_prompt

# Xterm title setting
case "$TERM" in
xterm*|rxvt*)
    PS1="\[\e]0;${debian_chroot:+($debian_chroot)}\u@\h: \w\a\]$PS1"
    ;;
*)
    ;;
esac

# PROMPT_COMMAND to set Xterm titles
case ${TERM} in
    xterm-termite|xterm*|rxvt*|Eterm|aterm|kterm|gnome*)
        PROMPT_COMMAND=${PROMPT_COMMAND:+$PROMPT_COMMAND; }'printf "\033]0;%s@%s:%s\007" "${USER}" "${HOSTNAME%%.*}" "${PWD/#$HOME/~}"'
        ;;
    screen)
        PROMPT_COMMAND=${PROMPT_COMMAND:+$PROMPT_COMMAND; }'printf "\033_%s@%s:%s\033\\" "${USER}" "${HOSTNAME%%.*}" "${PWD/#$HOME/~}"'
        ;;
esac
