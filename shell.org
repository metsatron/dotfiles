# ffetch: fastfetch with WezTerm overlay when inside WezTerm
ffetch() {
  # Optional debug - only fires if you export FFETCH_DEBUG=1
  if [[ -n ${FFETCH_DEBUG-} ]]; then
    printf '[ffetch] TERM="%s" WEZTERM_EXECUTABLE="%s" WEZTERM_VERSION="%s" WEZTERM_PANE="%s" TERM_PROGRAM="%s"\n' \
      "${TERM-}" "${WEZTERM_EXECUTABLE-}" "${WEZTERM_VERSION-}" "${WEZTERM_PANE-}" "${TERM_PROGRAM-}" >&2
  fi

  if [[ ${TERM-} == wezterm ]] \
     || [[ -n ${WEZTERM_PANE-}${TERM_PROGRAM-} ]]; then
    command fastfetch --config "$HOME/.config/fastfetch/wezterm.jsonc" "$@"
  else
    command fastfetch "$@"
  fi
}

# Simple colon-list deduper for bash/zsh (no output)
dedupe_path_var() {
  local name="$1" val new elem
  eval "val=\${$name}"
  new=
  IFS=':'
  for elem in $val; do
    [ -z "$elem" ] && continue
    case ":$new:" in
      ,*:"$elem":*) ;;
      ,*) new="${new:+$new:}$elem" ;;
    esac
  done
  eval "export $name=\$new"
}

# Set the title before each prompt (idle state)
la_term_title_precmd() {
  emulate -L zsh
  local base="${PWD:t}"
  printf '\e]2;%s\a' "$base"
}

# Set the title when a command is *about to run*
la_term_title_preexec() {
  emulate -L zsh
  # $1 is the full command line, grab just first word
  local cmd="$1"
  local prog="${cmd%% *}"

  # if it looks like a flatpak-spawn wrapper, try to surface the inner thing
  if [[ "$prog" == flatpak-spawn ]]; then
    # crude but good enough: look for last word
    prog="${cmd##* }"
  fi

  # keep it short (nvim, htop, btop, etc)
  printf '\e]2;%s\a' "$prog"
}

add-zsh-hook precmd  la_term_title_precmd
add-zsh-hook preexec la_term_title_preexec

#+BEGIN_SRC sh :tangle all/.zsh_00_guix_fpath :tangle-mode (identity #o644)
#!/bin/zsh
# ~/.zsh_00_guix_fpath — Guix zsh function dirs into fpath EARLY
# compinit, add-zsh-hook, is-at-least are autoload functions that live
# in zsh's function dirs.  When zsh is installed via Guix, these live
# inside the Guix store, not under /usr/share/zsh.  The compiled-in
# fpath points to /usr/share/zsh/functions/* which is empty on a Guix
# system.  Fix: resolve the real zsh binary, find its store functions
# dir, and prepend it.
_guix_zsh_store="$(readlink -f "$(command -v zsh)" 2>/dev/null)"
if [[ -n "$_guix_zsh_store" && "$_guix_zsh_store" == /gnu/store/* ]]; then
  _store_prefix="${_guix_zsh_store%/bin/zsh}"
  _zsh_fn_base="$_store_prefix/share/zsh"
  # Add the versioned functions dirs (e.g. share/zsh/5.9/functions/*)
  for d in "$_zsh_fn_base"/*/functions(N/); do
    fpath=("$d" $fpath)
    # Also add subdirs (Completion/*, VCS_Info/*, Zle, Misc, etc.)
    for sub in "$d"/*(N/); do
      fpath=("$sub" $fpath)
      # Two-level subdirs (Completion/Unix, Completion/Zsh, etc.)
      for sub2 in "$sub"/*(N/); do
        fpath=("$sub2" $fpath)
      done
    done
  done
  unset _store_prefix _zsh_fn_base
fi
unset _guix_zsh_store

# Also add Guix profile site-functions for plugins/completions
for d in \
  "${GUIX_PROFILE:-}/share/zsh/site-functions" \
  "$HOME/.guix-extra-profiles/core/core/share/zsh/site-functions" \
  "$HOME/.guix-profile/share/zsh/site-functions"; do
  [[ -d "$d" ]] && fpath=("$d" $fpath)

#+BEGIN_SRC bash :tangle all/.config/ranger/scope.sh :shebang "#!/usr/bin/env bash" :tangle-mode (identity #o755) :mkdirp yes
set -o noclobber -o nounset -o pipefail

file="$1"
w="${2:-$COLUMNS}"
h="${3:-$LINES}"
x="${4:-0}"
y="${5:-0}"
tty="/dev/tty"

mime="$(file --mime-type -Lb "$file")"

case "$mime" in
  image/*)
    # 1) Kitty graphics with exact placement (works in WezTerm), write to the TTY
    if command -v kitty >/dev/null 2>&1 && [ -z "${TMUX:-}" ]; then
      printf '\e7' >"$tty"                                 # save cursor
      printf '\e[%d;%dH' "$((y+1))" "$((x+1))" >"$tty"     # move to preview pane origin
      kitty +kitten icat \
        --place "${w}x${h}@${x}x${y}" \
        --transfer-mode=memory \
        --stdin=no \
        "$file" >"$tty" 2>/dev/null || true
      printf '\e8' >"$tty"                                 # restore cursor
      exit 6
    fi

    # 2) WezTerm's imgcat (width in cells), write to the TTY
    if command -v wezterm-imgcat >/dev/null 2>&1; then
      printf '\e7' >"$tty"
      printf '\e[%d;%dH' "$((y+1))" "$((x+1))" >"$tty"
      WEZTERM_IMGCAT_WIDTH="${w}cell" wezterm-imgcat "$file" >"$tty" 2>/dev/null || true
      printf '\e8' >"$tty"
      exit 6
    fi

    # 3) High-quality text fallback (sixel), goes to stdout inside the pane
    if command -v chafa >/dev/null 2>&1; then
      chafa --format=sixel --size "${w}x${h}" "$file"
      exit 0
    fi

    # 4) Simple text fallback
    if command -v viu >/dev/null 2>&1; then
      viu -w "$w" "$file"
      exit 0
    fi

    echo "Image preview available but no graphics backend found."
    exit 0
    ;;
esac

exit 0
#+END_SRC
