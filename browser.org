* Firedragon overrides
#+BEGIN_SRC js :tangle all/.firedragon/firedragon.overrides.cfg
// keep this file's first line blank

// Home, Top Sites, bookmarks, and window-openers all behave sanely
unlockPref("browser.tabs.loadBookmarksInTabs");
clearPref("browser.tabs.loadBookmarksInTabs");
defaultPref("browser.tabs.loadBookmarksInTabs", false);     // open in same tab

// global tabs
unlockPref("browser.link.open_newwindow");
clearPref("browser.link.open_newwindow");
defaultPref("browser.link.open_newwindow", 1);

// force tabs for popups with features too
unlockPref("browser.link.open_newwindow.restriction");
clearPref("browser.link.open_newwindow.restriction");
defaultPref("browser.link.open_newwindow.restriction", 0);

// external links from other apps
unlockPref("browser.link.open_newwindow.override.external");
clearPref("browser.link.open_newwindow.override.external");
defaultPref("browser.link.open_newwindow.override.external", 1);

// fonts visible and editable
unlockPref("browser.display.use_document_fonts");
unlockPref("layout.css.font-visibility");
unlockPref("font.default.x-western");
unlockPref("font.name.sans-serif.x-western");
unlockPref("font.name.serif.x-western");
unlockPref("font.name.monospace.x-western");

// sane defaults, UI not greyed out
defaultPref("browser.display.use_document_fonts", 1);
defaultPref("layout.css.font-visibility", 3);
defaultPref("font.default.x-western", "sans-serif");
defaultPref("font.name.sans-serif.x-western", "San Francisco Display");
defaultPref("font.name.serif.x-western", "New York");
defaultPref("font.name.monospace.x-western", "SFMono Nerd Font");

// keep existing
lockPref("widget.use-xdg-desktop-portal.file-picker", 2);

#+END_SRC
