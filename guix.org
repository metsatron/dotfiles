* Intent
A minimal, sovereign Guix install on a Debian-family host, managed from this repo.
We keep privileged steps explicit, package state declarative, and paths inside overlays.
Nothing in here overwrites your shell files directly, it only adds small blocks that tangle into your existing exports files.

* Layout
- Guix channels, profiles, manifests under =all/.config/guix/=
- Helper scripts under =all/.local/bin/=
- Shell exports for PATH added in tiny, self-contained blocks that tangle into =all/.bash_exports= and =all/.zsh_exports=
- No files removed, only appended sections with clear sentinels

* One-time system bootstrap
Only run once on this machine. This mirrors the upstream manual for a foreign distro install.
#+BEGIN_SRC bash :tangle no
# 0) Prereqs
sudo apt update
sudo apt install -y build-essential git curl xz-utils bzip2

# 1) Download official installer, verify, run
cd /tmp
rm -f guix-install.sh guix-install.sh.sig

# Follow redirects and fail if not 200
curl -fL -o guix-install.sh 'https://git.savannah.gnu.org/cgit/guix.git/plain/etc/guix-install.sh'
# Quick sanity check
head -n 3 guix-install.sh
# Expect to see: #!/bin/sh on the first line
chmod +x guix-install.sh
# Optional: verify PGP signature if you added Guix keys
# Run the installer
sudo ./guix-install.sh

# 2) Authorize substitute servers, improves build time
# ci.guix.gnu.org should be present after installer, we add bordeaux as well if available
if [ -r /usr/share/guix/bordeaux.guix.gnu.org.pub ]; then
  sudo guix archive --authorize < /usr/share/guix/bordeaux.guix.gnu.org.pub
fi

# 3) Enable and start the daemon
sudo systemctl enable --now guix-daemon
systemctl status guix-daemon --no-pager
#+END_SRC

- After installing Guix, run a substitute bench once and apply it to the daemon:
#+BEGIN_SRC bash

** Guix (bench caches + pull that never re-benchmarks)
#+BEGIN_SRC makefile :tangle all/.mk/guix.mk :mkdirp yes
# all/.mk/guix.mk
ifndef GUIX_MK_INCLUDED
GUIX_MK_INCLUDED := 1

.RECIPEPREFIX := |
SHELL := /bin/bash
.ONESHELL:

GUIX ?= guix
GUIX_HOME := $(HOME)/.config/guix
EXTRA := $(HOME)/.guix-extra-profiles
BENCH_DIR := $(GUIX_HOME)/.bench
PULL_CACHE := $(BENCH_DIR)/pull-url
FALLBACK_PULL_URLS ?= https://git.cbaines.net/guix/guix https://codeberg.org/guix/guix.git https://git.savannah.gnu.org/git/guix.git

.PHONY: guix-dirs guix-pull-bench guix-pull guix-core guix-dev guix-gc guix-nonguix

guix-dirs:
| @chmod +x $(HOME)/.local/bin/guix-mkdirs 2>/dev/null || true
| @$(HOME)/.local/bin/guix-mkdirs
| @mkdir -p $(BENCH_DIR)
# 1) BENCH → writes cache, prints URL (LOUD)
guix-pull-bench: guix-dirs
| set -Eeuo pipefail; \
| BENCH="$(HOME)/DotCortex/all/.local/bin/guix-bench-pull"; \
| [ -x "$$BENCH" ] || { echo "guix-bench-pull not found/executable" >&2; exit 1; }; \
| URL="$$( "$$BENCH" )" || URL=""; \
| if [ -z "$$URL" ]; then echo "No URL found by bench" >&2; exit 1; fi; \
| echo "$$URL" >"$(PULL_CACHE)"; \
| echo "cached: $(PULL_CACHE) → $$URL"; \
| echo "$$URL"

# 2) PULL → uses cached URL if present; NEVER re-bench
guix-pull: guix-dirs
| set -Eeuo pipefail; \
| URL=""; [ -r "$(PULL_CACHE)" ] && URL="$$(tr -d '\r\n' <"$(PULL_CACHE)")"; \
| tried=0; \
| success=0; \
| choose_and_pull() { \
|   u="$$1"; \
|   [ -z "$$u" ] && return 1; \
|   tried=1; \
|   echo "guix pull --url=$$u (cached/fallback)"; \
|   if $(GUIX) pull --url="$$u"; then \
|     success=1; \
|     printf '%s\n' "$$u" >"$(PULL_CACHE)";  # <-- promote last success to cache
|     return 0; \
|   else \
|     return 1; \
|   fi; \
| }; \
| # 1) try cached (if present)
| if [ -n "$$URL" ]; then \
|   choose_and_pull "$$URL" || echo "cached URL failed → trying fallbacks…"; \
| fi; \
| # 2) try fallbacks (skip if equal to cached)
| if [ "$$success" -eq 0 ]; then \
|   for fb in $(FALLBACK_PULL_URLS); do \
|     [ "$$fb" = "$$URL" ] && continue; \
|     if choose_and_pull "$$fb"; then break; fi; \
|   done; \
| fi; \
| # 3) final fallback: plain `guix pull` (no mirror override)
| if [ "$$success" -eq 0 ]; then \
|   if [ "$$tried" -eq 1 ]; then echo "all mirrors failed → falling back to plain 'guix pull'"; else echo "guix pull (no cache present)"; fi; \
|   $(GUIX) pull; \
| fi

guix-core: guix-dirs
| mkdir -p $(EXTRA)/core
| $(GUIX) package -m $(GUIX_HOME)/manifests/core.scm -p $(EXTRA)/core/core

guix-dev: guix-core
| mkdir -p $(EXTRA)/dev
| $(GUIX) package -m $(GUIX_HOME)/manifests/dev.scm -p $(EXTRA)/dev/dev

guix-nonguix: guix-dirs
| mkdir -p $(EXTRA)/nonguix
| $(GUIX) package -m $(GUIX_HOME)/manifests/nonguix.scm -p $(EXTRA)/nonguix/nonguix

guix-gc:
| $(GUIX) gc
endif
#+END_SRC

* Session wiring for Guix profiles
#+BEGIN_SRC bash :tangle linux/.xsessionrc.d/20-guix-core-profile.sh :shebang "#!/usr/bin/env bash"
# Sourced by Xsession under /bin/sh; never enable nounset here.
GUIX_PROFILE="$HOME/.guix-extra-profiles/core/core"
export GUIX_PROFILE
if [ -r "$GUIX_PROFILE/etc/profile" ]; then
  . "$GUIX_PROFILE/etc/profile"
fi
# The Guix profile sets GIO_EXTRA_MODULES to its own gio/modules dir, which
# contains GIO modules built against Guix GLib (currently 2.83+). System
# binaries (flatpak-session-helper, xdg-desktop-portal) link against the
# system GLib (2.72 on Ubuntu 22.04) and lack newer symbols like
# g_once_init_enter_pointer — causing them to crash on startup.
#
# /etc/X11/Xsession.d/95dbus_update-activation-env runs after this script and
# calls `dbus-update-activation-environment --systemd --all`, which bakes the
# entire current environment — including GIO_EXTRA_MODULES — into the systemd
# user session. Removing it from the systemd session here, before step 95
# runs, prevents it from ever reaching system service processes.
#
# Guix-built apps find their own GIO modules via their compiled-in rpath
# (/gnu/store/.../lib/gio/modules), so this does not break Guix apps.
if systemctl --user is-active --quiet default.target 2>/dev/null; then
  systemctl --user unset-environment GIO_EXTRA_MODULES 2>/dev/null || true
fi
unset GIO_EXTRA_MODULES
#+END_SRC

** Apply cached Guix pull mirror (no bench)
#+BEGIN_SRC bash :tangle all/.local/bin/guix-apply-pull-url :shebang "#!/usr/bin/env bash" :tangle-mode (identity #o755) :mkdirp yes
# all/.local/bin/guix-apply-pull-url
#!/usr/bin/env bash
set -Eeuo pipefail
BENCH_CACHE="${XDG_CONFIG_HOME:-$HOME/.config}/guix/.bench/pull-url"

RAW="${1:-}"
if [[ -z "$RAW" && -r "$BENCH_CACHE" ]]; then RAW="$(head -n1 "$BENCH_CACHE" | tr -d '\r')"; fi
[[ -n "${RAW:-}" ]] || { echo "guix-apply-pull-url: no URL" >&2; exit 1; }

URL="$(sed -nE 's/.*\b(https?:\/\/[^[:space:]]+).*/\1/p' <<<"$RAW")"
[[ -n "$URL" ]] || { echo "guix-apply-pull-url: unusable line: $RAW" >&2; exit 1; }

echo "Using channel mirror: $URL"
exec guix pull --url="$URL"
#+END_SRC

** Bench fastest Guix channel mirror
#+BEGIN_SRC bash :tangle all/.local/bin/guix-bench-pull :shebang "#!/usr/bin/env bash" :tangle-mode (identity #o755) :mkdirp yes
#!/usr/bin/env bash
# all/.local/bin/guix-bench-pull
set -Eeuo pipefail

BENCH_DIR="${XDG_CONFIG_HOME:-$HOME/.config}/guix/.bench"
CACHE="${BENCH_DIR}/pull-url"
BL="${BENCH_DIR}/pull-blacklist"
SSV="${XDG_CONFIG_HOME:-$HOME/.config}/guix/pull-candidates.ssv"

mkdir -p "$BENCH_DIR"; touch "$BL"

urls_from_ssv() {
  [[ -r "$SSV" ]] || return 1
  awk '
    BEGIN{IGNORECASE=1}
    /^[[:space:]]*#/ {next}           # comments
    NF==0 {next}                      # blank lines
    NR==1 && $1 ~ /^url$/ {next}      # optional header row
    {print $1}                        # first column is the URL
  ' "$SSV"
}

declare -a CANDS=()
if [[ -n "${GUIX_PULL_CANDIDATES:-}" ]]; then
  # space-separated URLs override SSV
  read -r -a CANDS <<< "$GUIX_PULL_CANDIDATES"
else
  mapfile -t CANDS < <(urls_from_ssv || true)
  # final fallback if SSV missing/empty
  if [[ ${#CANDS[@]} -eq 0 ]]; then
    CANDS=( "https://codeberg.org/guix/guix.git"
            "https://git.cbaines.net/guix/guix"
            "https://git.savannah.gnu.org/git/guix.git"
            "https://mirror.git.savannah.gnu.org/r/guix.git" )
  fi
fi

have_git=0; command -v git >/dev/null 2>&1 && have_git=1

probe_git() { # url -> seconds
  local u="$1"
  local t
  t="$( TIMEFORMAT=%R; { time git ls-remote --heads --tags --refs --timeout=12 "$u" >/dev/null 2>&1; } 2>&1 )" || return 1
  printf '%s\n' "$t"
}
probe_tcp() { # host:443 reachable?
  timeout 4 bash -lc "exec 3<>/dev/tcp/$1/443" 2>/dev/null
}

rows=()
for u in "${CANDS[@]}"; do
  grep -qxF "$u" "$BL" && continue
  host="$(sed -n 's#https\?://\([^/]*\).*#\1#p' <<<"$u")"
  probe_tcp "$host" || continue
  t=""
  if [[ $have_git -eq 1 ]]; then
    t="$(probe_git "$u" || true)"
  fi
  [[ -z "$t" ]] && continue
  rows+=("$t $u")
done

# choose the fastest
best="$(printf '%s\n' "${rows[@]}" | sort -n | awk 'NR==1{print $2}')"

if [[ -z "${best:-}" ]]; then
  echo "no-viable-mirror" >&2
  exit 2
fi

printf '%s\n' "$best"
#+END_SRC

** Substitutes bench script (loud, hardcoded list, env-override)
#+BEGIN_SRC bash :tangle all/.local/bin/guix-bench-substitutes :shebang "#!/usr/bin/env bash" :tangle-mode (identity #o755) :mkdirp yes
#!/usr/bin/env bash
set -Eeuo pipefail

read -r -a CANDS <<< "${GUIX_SUB_CANDIDATES:-https://bordeaux.guix.gnu.org https://ci.guix.gnu.org https://substitutes.nonguix.org}"

BENCH_DIR="${XDG_CONFIG_HOME:-$HOME/.config}/guix/.bench"
ORDER_FILE="${BENCH_DIR}/substitute-order"
BL="${BENCH_DIR}/substitute-blacklist"

mkdir -p "$BENCH_DIR"; touch "$BL"

PKGS=(bash coreutils grep sed gzip nss-certs glibc neovim)

rows=()
for u in "${CANDS[@]}"; do
  grep -qxF "$u" "$BL" && { echo "SKIP $u (blacklisted)"; continue; }
  if ! out=$(guix weather -m <(printf '(specifications->manifest (quote (%s)))\n' "${PKGS[*]}") --substitute-urls="$u" 2>&1); then
    echo "SKIP $u (weather error → blacklisted)"
    echo "$u" >>"$BL"
    continue
  fi
  avail=$(printf '%s\n' "$out" | awk '/substitutes available/{print $1}' | tr -d '%')
  sec=$(printf '%s\n' "$out"   | awk '/seconds per request/{print $1}')
  items=$(printf '%s\n' "$out" | awk '/substitutes available/{print $(NF-3)}')
  [[ -z "$avail" || -z "$items" ]] && continue
  printf '%03d %04d %0.3f %s\n' "${avail%.*}" "${items}" "${sec:-999}" "$u"
done | sort -r -n -k1,1 -k2,2 -k3,3g | {
  mapfile -t rows
  if [[ ${#rows[@]} -eq 0 ]]; then
    echo "No usable substitute servers." >&2
    exit 1
  fi
  printf '%s' "${rows[@]}" | awk '{print $4}' | paste -sd' ' -
}
#+END_SRC

** Guix substitutes (bench caches order; apply reads cache only)
#+BEGIN_SRC makefile :tangle all/.mk/guix-substitutes.mk :mkdirp yes
ifndef GUIX_SUB_MK_INCLUDED
GUIX_SUB_MK_INCLUDED := 1
.RECIPEPREFIX := |
SHELL := /bin/bash
.ONESHELL:

GUIX_HOME := $(HOME)/.config/guix
BENCH_DIR := $(GUIX_HOME)/.bench
ORDER_CACHE := $(BENCH_DIR)/substitute-order

# Full candidate set (bench decides)
SUB_CANDS = "https://bordeaux.guix.gnu.org https://ci.guix.gnu.org https://substitutes.nonguix.org https://hydra-guix-129.guix.gnu.org https://mirrors.sjtug.sjtu.edu.cn/guix https://mirror.yandex.ru/mirrors/guix/ https://bordeaux-singapore-mirror.cbaines.net https://bordeaux-us-east-mirror.cbaines.net"

.PHONY: guix-sub-bench guix-sub-apply

guix-sub-bench:
| set -Eeuo pipefail; \
| mkdir -p "$(BENCH_DIR)"; \
| ORDER="$$( GUIX_SUB_CANDIDATES=$(SUB_CANDS) ~/.local/bin/guix-bench-substitutes )" || { echo "bench failed"; exit 1; }; \
| if ! grep -Eq 'https?://[^ ]+' <<<"$$ORDER"; then echo "bench produced no URLs"; exit 1; fi; \
| printf '%s\n' "$$ORDER" >"$(ORDER_CACHE)"; \
| echo "cached: $(ORDER_CACHE) → $$ORDER"

guix-sub-apply:
| set -Eeuo pipefail; \
| [ -s "$(ORDER_CACHE)" ] || { echo "No $(ORDER_CACHE). Run 'guix-sub-bench' first." >&2; exit 1; }; \
| ORDER="$$(tr -d '\r\n' <"$(ORDER_CACHE)")"; \
| grep -Eq 'https?://[^ ]+' <<<"$$ORDER" || { echo "Invalid ORDER in cache"; exit 1; }; \
| \
| # Start from bench result, then append full candidate list, then dedup. \
| CANDS=$(SUB_CANDS); \

** Guix Env Bypassing Host Wrapper
#+BEGIN_SRC bash :tangle linux/.local/bin/host-wrap :shebang "#!/usr/bin/env bash" :mkdirp yes
#!/usr/bin/env bash
# Guix Env Bypassing Host Wrapper

set -euo pipefail

# Preserve a sane locale
LANG="${LANG:-C.UTF-8}"
LC_ALL="${LC_ALL:-}"

# Always disable CSD unless explicitly overridden
GTK_CSD="${GTK_CSD:-0}"

# If nocsd is available and LD_PRELOAD is not already set, use it
if [ -z "${LD_PRELOAD-}" ] && [ -e /usr/lib/x86_64-linux-gnu/libgtk3-nocsd.so.0 ]; then
  LD_PRELOAD="/usr/lib/x86_64-linux-gnu/libgtk3-nocsd.so.0"
fi

# Strip Guix profile bins out of PATH for this wrapper only
strip_guix_from_path() {
  local elem out=""
  IFS=':' read -r -a parts <<< "${PATH:-}"
  for elem in "${parts[@]}"; do
    [ -z "$elem" ] && continue
    if [[ "$elem" == "$HOME/.config/guix/current/bin" ]] \
       || [[ "$elem" == "$HOME/.guix-profile/bin" ]] \
       || [[ "$elem" == "$HOME/.guix-extra-profiles/"*"/bin" ]]; then
      continue
    fi
    if [ -z "$out" ]; then
      out="$elem"
    else
      out="$out:$elem"
    fi
  done
  printf '%s\n' "$out"
}

CLEAN_PATH="$(strip_guix_from_path)"

# Now exec the real command in a mostly clean env,
# but keep the bits that matter for X, DBus, global menu, nocsd, Qt and Flatpak metadata.
exec env -i \
  HOME="$HOME" \
  PATH="$CLEAN_PATH" \
  LANG="$LANG" \
  LC_ALL="$LC_ALL" \
  DISPLAY="${DISPLAY:-}" \
  XAUTHORITY="${XAUTHORITY:-}" \
  DBUS_SESSION_BUS_ADDRESS="${DBUS_SESSION_BUS_ADDRESS:-}" \
  XDG_RUNTIME_DIR="${XDG_RUNTIME_DIR:-}" \
  XDG_CURRENT_DESKTOP="${XDG_CURRENT_DESKTOP:-}" \
  XDG_SESSION_TYPE="${XDG_SESSION_TYPE:-}" \
  XDG_CONFIG_HOME="${XDG_CONFIG_HOME:-$HOME/.config}" \
  XDG_DATA_DIRS="${XDG_DATA_DIRS:-}" \
  XDG_DATA_HOME="${XDG_DATA_HOME:-$HOME/.local/share}" \
  GTK_MODULES="${GTK_MODULES:-}" \
  GTK_CSD="$GTK_CSD" \
  LD_PRELOAD="${LD_PRELOAD:-}" \
  QT_QPA_PLATFORMTHEME="${QT_QPA_PLATFORMTHEME:-}" \
  UBUNTU_MENUPROXY="${UBUNTU_MENUPROXY:-1}" \
  "$@"
#+END_SRC

* Notes
- Guix Home is intentionally not enabled here, since you use literate dotfiles as the source of truth. We keep Guix focused on packages and profiles.
- If you ever add =nonguix=, you must explicitly authorize its substitute key. Keep that ritual documented alongside the channels pin.
- Guix can install many tools that overlap with apt, snaps, or flatpaks. Prefer one source per tool to reduce path ambiguity.

Repro snippets:
