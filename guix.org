* Intent
A minimal, sovereign Guix install on a Debian-family host, managed from this repo.
We keep privileged steps explicit, package state declarative, and paths inside overlays.
Nothing in here overwrites your shell files directly, it only adds small blocks that tangle into your existing exports files.

* Layout
- Guix channels, profiles, manifests under =all/.config/guix/=
- Helper scripts under =all/.local/bin/=
- Shell exports for PATH added in tiny, self-contained blocks that tangle into =all/.bash_exports= and =all/.zsh_exports=
- No files removed, only appended sections with clear sentinels
  

* One-time system bootstrap
Only run once on this machine. This mirrors the upstream manual for a foreign distro install.

#+BEGIN_SRC bash :tangle no
# 0) Prereqs
sudo apt update
sudo apt install -y build-essential git curl xz-utils bzip2

# 1) Download official installer, verify, run
cd /tmp
rm -f guix-install.sh guix-install.sh.sig

# Follow redirects and fail if not 200
curl -fL -o guix-install.sh 'https://git.savannah.gnu.org/cgit/guix.git/plain/etc/guix-install.sh'
# Quick sanity check
head -n 3 guix-install.sh
# Expect to see: #!/bin/sh on the first line
chmod +x guix-install.sh
# Optional: verify PGP signature if you added Guix keys
# Run the installer
sudo ./guix-install.sh

# 2) Authorize substitute servers, improves build time
# ci.guix.gnu.org should be present after installer, we add bordeaux as well if available
if [ -r /usr/share/guix/bordeaux.guix.gnu.org.pub ]; then
  sudo guix archive --authorize < /usr/share/guix/bordeaux.guix.gnu.org.pub
fi

# 3) Enable and start the daemon
sudo systemctl enable --now guix-daemon
systemctl status guix-daemon --no-pager
#+END_SRC

If the installer complains about guixbuild users, re-run it or create them with =addgroup guixbuild= and =adduser --system= 10 workers linked to that group, which the installer usually handles.

** Guix (bench caches + pull that never re-benchmarks)
#+BEGIN_SRC makefile :tangle all/.mk/guix.mk :mkdirp yes
# all/.mk/guix.mk
ifndef GUIX_MK_INCLUDED
GUIX_MK_INCLUDED := 1

.RECIPEPREFIX := |
SHELL := /bin/bash
.ONESHELL:

GUIX ?= guix
GUIX_HOME := $(HOME)/.config/guix
EXTRA := $(HOME)/.guix-extra-profiles
BENCH_DIR := $(GUIX_HOME)/.bench
PULL_CACHE := $(BENCH_DIR)/pull-url
FALLBACK_PULL_URLS ?= https://git.cbaines.net/guix/guix https://codeberg.org/guix/guix.git https://git.savannah.gnu.org/git/guix.git

.PHONY: guix-dirs guix-pull-bench guix-pull guix-core guix-dev guix-gc

guix-dirs:
| @chmod +x $(HOME)/.local/bin/guix-mkdirs 2>/dev/null || true
| @$(HOME)/.local/bin/guix-mkdirs
| @mkdir -p $(BENCH_DIR)
# 1) BENCH → writes cache, prints URL (LOUD)
guix-pull-bench: guix-dirs
| set -Eeuo pipefail; \
| BENCH="$(HOME)/.dotfiles/all/.local/bin/guix-bench-pull"; \
| [ -x "$$BENCH" ] || { echo "guix-bench-pull not found/executable" >&2; exit 1; }; \
| URL="$$( "$$BENCH" )" || URL=""; \
| if [ -z "$$URL" ]; then echo "No URL found by bench" >&2; exit 1; fi; \
| echo "$$URL" >"$(PULL_CACHE)"; \
| echo "cached: $(PULL_CACHE) → $$URL"; \
| echo "$$URL"

# 2) PULL → uses cached URL if present; NEVER re-bench
guix-pull: guix-dirs
| set -Eeuo pipefail; \
| URL=""; [ -r "$(PULL_CACHE)" ] && URL="$$(tr -d '\r\n' <"$(PULL_CACHE)")"; \
| tried=0; \
| success=0; \
| choose_and_pull() { \
|   u="$$1"; \
|   [ -z "$$u" ] && return 1; \
|   tried=1; \
|   echo "guix pull --url=$$u (cached/fallback)"; \
|   if $(GUIX) pull --url="$$u"; then \
|     success=1; \
|     printf '%s\n' "$$u" >"$(PULL_CACHE)";  # <-- promote last success to cache
|     return 0; \
|   else \
|     return 1; \
|   fi; \
| }; \
| # 1) try cached (if present)
| if [ -n "$$URL" ]; then \
|   choose_and_pull "$$URL" || echo "cached URL failed → trying fallbacks…"; \
| fi; \
| # 2) try fallbacks (skip if equal to cached)
| if [ "$$success" -eq 0 ]; then \
|   for fb in $(FALLBACK_PULL_URLS); do \
|     [ "$$fb" = "$$URL" ] && continue; \
|     if choose_and_pull "$$fb"; then break; fi; \
|   done; \
| fi; \
| # 3) final fallback: plain `guix pull` (no mirror override)
| if [ "$$success" -eq 0 ]; then \
|   if [ "$$tried" -eq 1 ]; then echo "all mirrors failed → falling back to plain 'guix pull'"; else echo "guix pull (no cache present)"; fi; \
|   $(GUIX) pull; \
| fi

guix-core: guix-dirs
| mkdir -p $(EXTRA)/core
| $(GUIX) package -m $(GUIX_HOME)/manifests/core.scm -p $(EXTRA)/core/core

guix-dev: guix-core
| mkdir -p $(EXTRA)/dev
| $(GUIX) package -m $(GUIX_HOME)/manifests/dev.scm -p $(EXTRA)/dev/dev

guix-gc:
| $(GUIX) gc
endif
#+END_SRC

** Bench fastest Guix channel mirror
#+BEGIN_SRC bash :tangle all/.local/bin/guix-bench-pull :shebang "#!/usr/bin/env bash" :tangle-mode (identity #o755) :mkdirp yes
#!/usr/bin/env bash
set -Eeuo pipefail

# Benchmark Guix channel mirrors for `guix pull`.
# - Per-URL result lines (to STDERR)
# - Best URL (to STDOUT only)
# Env:
#   GUIX_PULL_CANDIDATES="url1 url2 …" to override list
#   GUIX_PULL_TIMEOUT=15  (git ls-remote timeout seconds)
#   GUIX_PULL_VERBOSE=1   (default 1; controls STDERR chatter)
# Flags:
#   -v/--verbose     force verbose
#   --force-curl     skip git timing; use curl HEAD latency

VERBOSE="${GUIX_PULL_VERBOSE:-1}"
TIMEOUT="${GUIX_PULL_TIMEOUT:-15}"
FORCE_CURL=0

for a in "$@"; do
  case "$a" in
    -v|--verbose) VERBOSE=1 ;;
    --force-curl) FORCE_CURL=1 ;;
    *) ;;
  esac
done

# Default candidates (include Codeberg)
CANDIDATES=(
  "https://git.savannah.gnu.org/git/guix.git"
  "https://mirror.git.savannah.gnu.org/r/guix.git"
  "https://git.cbaines.net/guix/guix"
  "https://codeberg.org/guix/guix.git"
)
[ -n "${GUIX_PULL_CANDIDATES:-}" ] && read -r -a CANDIDATES <<<"${GUIX_PULL_CANDIDATES}"

have_git=0
if [ "$FORCE_CURL" -eq 0 ] && command -v git >/dev/null 2>&1; then
  have_git=1
fi

ptime_git() { # -> float seconds or 99
  local url="$1" t
  t="$( TIMEFORMAT=%R; { time git ls-remote --heads --tags --refs --timeout="$TIMEOUT" "$url" >/dev/null 2>&1; } 2>&1 )" || t=99
  printf '%s' "$t"
}

ptime_curl() { # -> float seconds or 99
  curl -sS -o /dev/null -m "$TIMEOUT" -w '%{time_total}' -I "$1" || printf '%s' 99
}

mode_desc() { [ "$have_git" -eq 1 ] && echo "git" || echo "curl"; }

[ "$VERBOSE" -eq 1 ] && {
  >&2 echo "guix-bench-pull: probing ${#CANDIDATES[@]} candidate(s) using $(mode_desc)… (timeout=${TIMEOUT}s)"
}

best=""; best_t="999999"
any_ok=0

# header
[ "$VERBOSE" -eq 1 ] && >&2 printf '%-52s  %8s  %s\n' "URL" "TIME(s)" "MODE"

for url in "${CANDIDATES[@]}"; do
  # try git timing first unless forced curl / no git
  t="99"; mode="$(mode_desc)"
  if [ "$have_git" -eq 1 ]; then
    t="$(ptime_git "$url")"
    if [ "$t" = "99" ]; then
      # fallback to curl latency for this URL only
      t="$(ptime_curl "$url")"
      mode="curl"
    fi
  else
    t="$(ptime_curl "$url")"
    mode="curl"
  fi

  [ "$t" != "99" ] && any_ok=1

  # pretty row
  [ "$VERBOSE" -eq 1 ] && >&2 printf '%-52s  %8s  %s\n' "$url" "$t" "$mode"

  # numeric compare (float-safe)
  awk -v a="$t" -v b="$best_t" 'BEGIN{exit (a+0<b+0)?0:1}' \
  && { best="$url"; best_t="$t"; }
done

if [ -n "$best" ] && [ "$any_ok" -eq 1 ]; then
  [ "$VERBOSE" -eq 1 ] && >&2 echo "best=${best} time=${best_t}"
  echo "$best"
  exit 0
fi

# If absolutely everything failed (all 99), still return a sane default.
fallback="https://git.savannah.gnu.org/git/guix.git"
[ "$VERBOSE" -eq 1 ] && >&2 echo "guix-bench-pull: all probes failed; falling back to ${fallback}"
echo "$fallback"
exit 0
#+END_SRC

** Substitutes bench script (loud, hardcoded list, env-override)
#+BEGIN_SRC bash :tangle all/.local/bin/guix-bench-substitutes :shebang "#!/usr/bin/env bash" :tangle-mode (identity #o755) :mkdirp yes
#!/usr/bin/env bash
set -Eeuo pipefail

VERBOSE="${GUIX_SUB_VERBOSE:-1}"

# Hardcoded default set (literate, reproducible)
CANDIDATES=( "https://bordeaux.guix.gnu.org" "https://ci.guix.gnu.org" "https://substitutes.nonguix.org" )
[ -n "${GUIX_SUB_CANDIDATES:-}" ] && read -r -a CANDIDATES <<< "${GUIX_SUB_CANDIDATES}"

DEFAULT_PKGS=(bash coreutils grep sed gzip nss-certs glibc neovim)
[ -n "${GUIX_SUB_BENCH_PKGS:-}" ] && read -r -a DEFAULT_PKGS <<< "${GUIX_SUB_BENCH_PKGS}"

TIMEOUT=25
if guix weather --help 2>/dev/null | grep -q -- '--timeout'; then
  TIMEOUT_FLAG=(--timeout="$TIMEOUT")
else
  TIMEOUT_FLAG=()
fi

[ "$VERBOSE" -eq 1 ] && {
  >&2 echo "guix-bench-substitutes: probing ${#CANDIDATES[@]} server(s) with guix weather…"
  >&2 echo "PKGS: ${DEFAULT_PKGS[*]}"
  >&2 echo
}

tmp="$(mktemp)"; err="$(mktemp)"; sum="$(mktemp)"
trap 'rm -f "$tmp" "$err" "$sum"' EXIT

ok_any=0
for url in "${CANDIDATES[@]}"; do
  [ "$VERBOSE" -eq 1 ] && >&2 echo "=== $url ==="
  if guix weather "${TIMEOUT_FLAG[@]}" --substitute-urls="$url" "${DEFAULT_PKGS[@]}" >"$tmp" 2>"$err"; then
    ok_any=1
    avail="$(grep -Eo '([0-9]+\.[0-9]+|[0-9]+)% substitutes available' "$tmp" | awk '{print $1}' | head -n1 || echo 0)"
    items="$(grep -Eo '\([0-9]+ out of [0-9]+\)' "$tmp" | sed -E 's/^\(([0-9]+) out of [0-9]+\)$/\1/' | head -n1 || echo 0)"
    speed="$(grep -Ei 'estimated download speed' "$tmp" | grep -Eo '[0-9.]+\s*(KiB/s|MiB/s|GiB/s)' | head -n1 || echo unknown)"
    [ "$VERBOSE" -eq 1 ] && {
      >&2 echo "availability=${avail}%  items=${items}  speed=${speed}"
      >&2 echo
      sed -n '1,80p' "$tmp" >&2
      >&2 echo
    }
    printf "SUMMARY|%s|%s|%s\n" "$url" "$avail" "$items" >>"$sum"
  else
    [ "$VERBOSE" -eq 1 ] && { echo "ERROR probing $url" >&2; sed -n '1,40p' "$err" >&2; echo >&2; }
    printf "SUMMARY|%s|0|0\n" "$url" >>"$sum"
  fi
done

if (( ok_any )); then
  [ "$VERBOSE" -eq 1 ] && >&2 echo "Ranking by availability then items…"
  mapfile -t order < <(awk -F'|' '/^SUMMARY\|/ {printf "%s %s %s\n",$2,$3,$4}' "$sum" \
    | sort -k2,2nr -k3,3nr | awk '{print $1}')
  [ "$VERBOSE" -eq 1 ] && >&2 echo "Chosen order: ${order[*]}"
  printf "%s\n" "${order[*]}"
  exit 0
fi

[ "$VERBOSE" -eq 1 ] && >&2 echo "weather failed; falling back to curl HEAD latency…"
declare -A latency ok
for url in "${CANDIDATES[@]}"; do
  read -r code total <<<"$(curl -sS -o /dev/null -w "%{http_code} %{time_total}" -m 10 -I "$url" || echo "000 99")"
  latency["$url"]="$total"; ok["$url"]="$code"
  [ "$VERBOSE" -eq 1 ] && >&2 echo "curl: $url  code=$code  time=${total}s"
done
mapfile -t order < <(for u in "${!latency[@]}"; do printf "%s %s %s\n" "${latency[$u]}" "${ok[$u]}" "$u"; done \
  | awk '$2!="000"{print}' | sort -n | awk '{print $3}')
[ "${#order[@]}" -gt 0 ] && { [ "$VERBOSE" -eq 1 ] && >&2 echo "Chosen order (latency): ${order[*]}"; printf "%s\n" "${order[@]}"; } || printf "%s\n" "${CANDIDATES[@]}"
#+END_SRC

.PHONY: nonguix-sub-bench nonguix-sub-apply nonguix-authorize-key

nonguix-sub-bench:
| set -Eeuo pipefail; \
| mkdir -p "$(BENCH_DIR)"; \
| ORDER="$$( GUIX_SUB_CANDIDATES='https://substitutes.nonguix.org https://bordeaux.guix.gnu.org https://ci.guix.gnu.org' \
|            ~/.local/bin/guix-bench-substitutes | awk 'NF{last=$$0} END{print last}' )"; \
| if [ -z "$$ORDER" ]; then echo "No ORDER from bench" >&2; exit 1; fi; \
| echo "$$ORDER" >"$(NONGUIX_SUB_CACHE)"; \
| echo "cached: $(NONGUIX_SUB_CACHE) → $$ORDER"; \
| echo "$$ORDER"

nonguix-sub-apply:
| set -Eeuo pipefail; \
| [ -r "$(NONGUIX_SUB_CACHE)" ] || { echo "No $(NONGUIX_SUB_CACHE). Run 'make -f all/.mk/nonguix-substitutes.mk nonguix-sub-bench' first." >&2; exit 1; }; \
| ORDER="$$(tr -d '\r\n' <"$(NONGUIX_SUB_CACHE)")"; \
| if [ -z "$$ORDER" ]; then echo "Empty $(NONGUIX_SUB_CACHE)" >&2; exit 1; fi; \
| DAEMON_BIN="$$(systemctl show -p ExecStart guix-daemon 2>/dev/null | sed -E 's/^ExecStart=..([^ ]*guix-daemon).*/\1/')"; \
| [ -z "$$DAEMON_BIN" -o ! -x "$$DAEMON_BIN" ] && DAEMON_BIN="/var/guix/profiles/per-user/root/current-guix/bin/guix-daemon"; \
| sudo mkdir -p /etc/systemd/system/guix-daemon.service.d; \
| tmp="$$(mktemp)"; \
| printf '%s\n' \
|   "[Service]" \
|   "Environment=GUIX_LOCPATH=/var/guix/profiles/per-user/root/guix-profile/lib/locale" \
|   "ExecStart=" \
|   "ExecStart=$$DAEMON_BIN --build-users-group=guixbuild --substitute-urls=\"$$ORDER\"" \
|   > "$$tmp"; \
| sudo install -m 0644 "$$tmp" /etc/systemd/system/guix-daemon.service.d/override.conf; \
| rm -f "$$tmp"; \
| sudo systemctl daemon-reload; \
| sudo systemctl restart guix-daemon; \
| systemctl --no-pager status guix-daemon || true

nonguix-authorize-key:
| set -Eeuo pipefail; \
| if [ -r /usr/share/guix/nonguix.pub ]; then key=/usr/share/guix/nonguix.pub; \
| elif [ -r /etc/guix/nonguix.pub ]; then key=/etc/guix/nonguix.pub; \
| elif [ -r $(HOME)/.dotfiles/all/.config/guix/nonguix-signing-key.pub ]; then key=$(HOME)/.dotfiles/all/.config/guix/nonguix-signing-key.pub; \
| else echo "WARNING: nonguix key not found. Put it at all/.config/guix/nonguix-signing-key.pub and rerun." >&2; exit 0; fi; \
| echo "Authorizing nonguix key: $$key"; \
| sudo guix archive --authorize < "$$key"
endif
#+END_SRC

* Notes
- Guix Home is intentionally not enabled here, since you use literate dotfiles as the source of truth. We keep Guix focused on packages and profiles.
- If you ever add =nonguix=, you must explicitly authorize its substitute key. Keep that ritual documented alongside the channels pin.
- Guix can install many tools that overlap with apt, snaps, or flatpaks. Prefer one source per tool to reduce path ambiguity.

* Known issue: Inkscape via Guix (pinned note)
:PROPERTIES:
:CREATED:  [2025-09-24]
:END:

Symptoms (current channel snapshot):
- Runtime complains: libxml2 “no version information available”
- Crash from libxslt: undefined symbol ‘inputPush, version LIBXML2_2.4.30’

Repro snippets:
#+BEGIN_SRC bash
guix shell --pure inkscape -- inkscape
# or show what binary actually runs:
REAL="$(readlink -f "$(which inkscape)"; true)"; REAL="${REAL%/inkscape}/.inkscape-real"; echo "$REAL"
LD_DEBUG=libs "$REAL" 2>&1 | grep -E 'libxml2|libxslt' | sed -n '1,80p'
#+END_SRC

Workarounds:
- Prefer distro/Flatpak Inkscape for now.
- Or use a known-good Guix generation *only* for Inkscape:
#+BEGIN_SRC bash
guix describe --generations
guix time-machine --generation=N -- shell --pure inkscape -- inkscape
#+END_SRC

We intentionally skip mixed LD_PRELOAD tricks (fragile across ICU/libxml2).
