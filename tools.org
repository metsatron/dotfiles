* Tools Loom (Linux)
Small helper tools that live in the =linux/.local/bin= overlay and are
deployed via stow. These are generic enough to follow the machine as the
hardware evolves.

** monitor-check – detect EDID-blind DisplayPort outputs
This helper inspects an =xrandr= output (default =DisplayPort-0=) and
tells you whether it looks “EDID-blind” (no monitor size, only 640x480
exposed). It does *not* try to be clever with automatic fixes; it’s a
diagnostic + ritual reminder so you don’t waste time guessing.

Usage:

- Check default DP:
  #+BEGIN_EXAMPLE
  monitor-check
  #+END_EXAMPLE
- Check a specific output:
  #+BEGIN_EXAMPLE
  monitor-check DisplayPort-1
  #+END_EXAMPLE

If it detects the “640x480 only, 0mm x 0mm” state, it will tell you to
power-cycle the monitor + reseat the cable, then re-run:

#+BEGIN_SRC bash :tangle linux/.local/bin/monitor-check :shebang "#!/usr/bin/env bash"
set -euo pipefail

OUT="${1:-DisplayPort-0}"

# Ensure xrandr is available
if ! command -v xrandr >/dev/null 2>&1; then
  echo "monitor-check: xrandr not found in PATH."
  exit 1
fi

# Check that the output exists
if ! xrandr | grep -q "^${OUT} "; then
  echo "monitor-check: output '${OUT}' not found in xrandr."
  echo "Available outputs:"
  xrandr | awk '/ connected| disconnected/ {print "  -", $1, $2}'
  exit 1
fi

# Grab the header line and mode block for this output
header_line="$(xrandr | awk -v o="${OUT}" '$1 == o {print; exit}')"
mode_block="$(xrandr | sed -n "/^${OUT} /,/^$/p" | sed '1d')"

echo "monitor-check: inspecting '${OUT}'"
echo "  xrandr header: ${header_line}"

# Heuristic 1: 0mm x 0mm -> no EDID info
edid_blind=0
if printf '%s\n' "${header_line}" | grep -q ' 0mm x 0mm'; then
  edid_blind=1
fi

# Heuristic 2: check if there are any modes larger than 640x480
has_large_mode=0

while IFS= read -r line; do
  # Expect mode lines like: "   1920x1080     60.00"
  if [[ "${line}" =~ ^[[:space:]]+([0-9]+)x([0-9]+) ]]; then
    w="${BASH_REMATCH[1]}"
    h="${BASH_REMATCH[2]}"

    # Treat anything strictly larger than 640x480 as a "real" mode
    if (( w > 640 || h > 480 )); then
      has_large_mode=1
      break
    fi
  fi
done <<< "${mode_block}"

# Also detect whether 640x480 is the only mode line at all
has_640_only=0
mode_count=0

while IFS= read -r line; do
  if [[ "${line}" =~ ^[[:space:]]+([0-9]+)x([0-9]+) ]]; then
    mode_count=$((mode_count + 1))
  fi
done <<< "${mode_block}"

if (( mode_count == 1 )) && printf '%s\n' "${mode_block}" | grep -q '640x480'; then
  has_640_only=1
fi

echo

if (( edid_blind == 1 && (has_large_mode == 0 || has_640_only == 1) )); then
  echo "⚠️  ${OUT} appears to be EDID-blind:"
  echo "    - Reported size: 0mm x 0mm"
  echo "    - Exposed modes: only 640x480 fallback"
  echo
  echo "This is the classic 'monitor woke up dumb' state."
  echo
  echo "Suggested ritual:"
  echo "  1) Power the monitor OFF."
  echo "  2) Reseat the DisplayPort cable (or dock cable)."
  echo "  3) Power the monitor ON."
  echo "  4) Then run:"
  echo "       xrandr --output ${OUT} --auto"
  echo
  echo "You can re-run 'monitor-check ${OUT}' afterwards to confirm higher modes."
  exit 0
fi

if (( edid_blind == 1 )); then
  echo "⚠️  ${OUT} reports 0mm x 0mm but some larger modes are present."
  echo "    EDID might be partial or stale. If you see glitches, try:"
  echo
  echo "      xrandr --output ${OUT} --off"
  echo "      xrandr --output ${OUT} --auto"
  echo
  echo "and/or perform the power-cycle ritual as above."
elif (( has_large_mode == 0 )); then
  echo "⚠️  ${OUT} has no modes larger than 640x480, but EDID is not obviously blind."
  echo "    Modes exposed:"
  echo "${mode_block}" | sed 's/^/      /'
  echo
  echo "You may still need to power-cycle the monitor and reseat the cable."
else
  echo "✅  ${OUT} reports proper EDID and larger modes are available."
  echo
  echo "Extracted modes for ${OUT}:"
  echo "${mode_block}" | sed 's/^/    /'
  echo
  echo "You can switch with, e.g.:"
  echo "  xrandr --output ${OUT} --mode <desired_mode>"
fi
#+END_SRC

** ai-usage - Linux usage wrapper for CodexBar, Fuelcheck, Antigravity, DeepInfra
This is a Linux-first wrapper that chooses the best local usage probe per
provider instead of forcing everything through =codexbar=. It prefers
=fuelcheck-cli= for OAuth/local providers, =antigravity-usage= for
Antigravity, =gemini-cli-usage= for Gemini CLI quotas, keeps =codexbar=
where it is actually useful on Linux, and adds a DeepInfra section using
local environment variables or =deepctl auth token= as a fallback.

Usage:

- Default report:
  #+BEGIN_EXAMPLE
  ai-usage
  #+END_EXAMPLE
- JSON output:
  #+BEGIN_EXAMPLE
  ai-usage --json
  #+END_EXAMPLE
- Restrict to a few providers:
  #+BEGIN_EXAMPLE
  ai-usage --providers codex,claude,deepinfra
  #+END_EXAMPLE

def load_dotenv_exports() -> dict[str, str]:
    path = os.path.expanduser("~/.env")
    exports: dict[str, str] = {}

    if not os.path.exists(path):
        return exports

    try:
        with open(path, "r", encoding="utf-8") as handle:
            for raw_line in handle:
                line = raw_line.strip()
                if not line or line.startswith("#"):
                    continue
                if line.startswith("export "):
                    line = line[7:].strip()
                if "=" not in line:
                    continue
                key, value = line.split("=", 1)
                key = key.strip()
                value = value.strip()
                if not key:
                    continue
                if len(value) >= 2 and value[0] == value[-1] and value[0] in "\"'":
                    value = value[1:-1]
                exports.setdefault(key, value)
    except OSError:
        return {}

    return exports


def nested_get(obj: Any, *path: str) -> Any:
    cur = obj
    for key in path:
        if not isinstance(cur, dict) or key not in cur:
            return None
        cur = cur[key]
    return cur


def compact_json(obj: Any, limit: int = 240) -> str:
    text = json.dumps(obj, ensure_ascii=True, sort_keys=True, separators=(",", ":"))
    if len(text) <= limit:
        return text
    return text[: limit - 3] + "..."


def last_meaningful_line(*texts: str) -> str:
    for text in texts:
        for line in text.splitlines():
            line = line.strip()
            if line:
                return line
    return ""


def parse_json_docs(text: str) -> list[Any]:
    text = text.strip()
    if not text:
        return []

    try:
        return [json.loads(text)]
    except json.JSONDecodeError:
        pass

    docs: list[Any] = []
    for line in text.splitlines():
        line = line.strip()
        if not line or line[0] not in "[{":
            continue
        try:
            docs.append(json.loads(line))
        except json.JSONDecodeError:
            continue
    return docs


def first_item(doc: Any) -> Any:
    if isinstance(doc, list):
        return doc[0] if doc else None
    return doc


def choose_provider_doc(docs: list[Any], provider: str | None) -> Any:
    if provider:
        for doc in docs:
            item = first_item(doc)
            if isinstance(item, dict) and item.get("provider") == provider:
                return item
    return first_item(docs[0]) if docs else None

def claude_cookie_env() -> dict[str, str]:
    if os.environ.get("CLAUDE_COOKIE") or load_dotenv_exports().get("CLAUDE_COOKIE"):
        return {}

    cookie_path = Path.home() / ".config" / "Claude" / "Cookies"
    if not cookie_path.exists():
        return {}

    try:
        conn = sqlite3.connect(f"file:{cookie_path}?mode=ro", uri=True)
    except sqlite3.Error:
        return {}

    try:
        now_chrome = int((time.time() + 11644473600) * 1000000)
        rows = conn.execute(
            """
            select name, value, expires_utc
              from cookies
             where host_key in ('claude.ai', '.claude.ai')
               and value != ''
             order by case name when 'sessionKey' then 0 else 1 end, name
            """
        ).fetchall()
    except sqlite3.Error:
        return {}
    finally:
        conn.close()

    parts: list[str] = []
    for name, value, expires_utc in rows:
        if not isinstance(name, str) or not isinstance(value, str):
            continue
        try:
            expires = int(expires_utc)
        except (TypeError, ValueError):
            expires = 0
        if expires and expires <= now_chrome:
            continue
        parts.append(f"{name}={value}")

    if not any(part.startswith("sessionKey=") for part in parts):
        return {}
    return {"CLAUDE_COOKIE": "; ".join(parts)}


def gemini_cli_oauth_env() -> dict[str, str]:
    home = Path.home()
    search_roots = [
        home / ".npm-global" / "lib" / "node_modules",
        home / ".npm-global" / "lib",
        home / ".nvm" / "versions",
    ]

    client_id = None
    client_secret = None
    pattern_id = re.compile(r"const OAUTH_CLIENT_ID = '([^']+)';")
    pattern_secret = re.compile(r"const OAUTH_CLIENT_SECRET = '([^']+)';")

    for root in search_roots:
        if not root.exists():
            continue
        for oauth2_path in root.rglob("oauth2.js"):
            path_text = str(oauth2_path)
            if "@google/gemini-cli-core" not in path_text or "code_assist" not in path_text:
                continue
            try:
                source = oauth2_path.read_text(encoding="utf-8")
            except OSError:
                continue
            id_match = pattern_id.search(source)
            secret_match = pattern_secret.search(source)
            if id_match and secret_match:
                client_id = id_match.group(1)
                client_secret = secret_match.group(1)
                break
        if client_id and client_secret:
            break

    if client_id and client_secret:
        return {
            "GEMINI_OAUTH_CLIENT_ID": client_id,
            "GEMINI_OAUTH_CLIENT_SECRET": client_secret,
        }

    return {}


def parse_iso_datetime(value: str | None) -> datetime | None:
    if not value or not isinstance(value, str):
        return None

    text = value.strip()
    if not text:
        return None

    if text.endswith("Z"):
        text = text[:-1] + "+00:00"

    try:
        dt = datetime.fromisoformat(text)
    except ValueError:
        return None

    if dt.tzinfo is None:
        dt = dt.replace(tzinfo=timezone.utc)
    return dt


def format_absolute_reset(value: str | None) -> str | None:
    dt = parse_iso_datetime(value)
    if not dt:
        return None
    local_dt = dt.astimezone()
    now = datetime.now().astimezone()
    if local_dt.date() == now.date():
        return local_dt.strftime("%-I:%M %p")
    return local_dt.strftime("%-d %b %Y at %-I:%M %p")


def format_countdown(value: str | None) -> str | None:
    dt = parse_iso_datetime(value)
    if not dt:
        return None
    delta = dt - datetime.now(timezone.utc)
    seconds = max(0, int(delta.total_seconds()))
    days, rem = divmod(seconds, 86400)
    hours, rem = divmod(rem, 3600)
    minutes = rem // 60
    if days:
        return f"{days}d {hours}h {minutes}m"
    if hours:
        return f"{hours}h {minutes}m"
    return f"{minutes}m"


def extract_reset_hint(raw_text: str | None, label: str) -> str | None:
    if not raw_text:
        return None

    lines = raw_text.splitlines()
    label_lc = label.lower()
    for line in lines:
        clean = " ".join(line.split())
        clean_lc = clean.lower()
        if label_lc == "5h":
            matches_label = clean_lc.startswith("5h") or "session" in clean_lc
        else:
            matches_label = clean_lc.startswith("1w") or "weekly" in clean_lc
        if not matches_label:
            continue

        match = re.search(r"resets?\s+(.*?)(?:\)|$)", clean, flags=re.IGNORECASE)
        if match:
            return match.group(1).strip()

    return None


def format_minutes_window(value: Any) -> str | None:
    if not isinstance(value, (int, float)):
        return None
    minutes = int(value)
    if minutes % (60 * 24 * 7) == 0 and minutes >= 60 * 24 * 7:
        weeks = minutes // (60 * 24 * 7)
        return f"{weeks}w"
    if minutes % (60 * 24) == 0 and minutes >= 60 * 24:
        days = minutes // (60 * 24)
        return f"{days}d"
    if minutes % 60 == 0 and minutes >= 60:
        hours = minutes // 60
        return f"{hours}h"
    return f"{minutes}m"


def detect_windows(item: dict[str, Any]) -> list[tuple[str, dict[str, Any]]]:
    usage = item.get("usage") if isinstance(item.get("usage"), dict) else item
    windows: list[tuple[str, dict[str, Any]]] = []
    primary = nested_get(usage, "primary")
    secondary = nested_get(usage, "secondary")
    if isinstance(primary, dict) and primary:
        windows.append(("Current", primary))
    if isinstance(secondary, dict) and secondary:
        windows.append(("Secondary", secondary))
    return windows


def format_percent(value: Any) -> str | None:
    if not isinstance(value, (int, float)):
        return None
    pct = max(0.0, min(100.0, float(value)))
    return f"{pct:.0f}%"


def format_bar(pct: float, width: int = 10, color: str | None = None) -> str:
    filled = round((pct / 100.0) * width)
    bar_color = color or (GR if pct >= 99.5 else (YL if pct >= 40 else RD))
    return f"{bar_color}{'█' * filled}{'░' * (width - filled)}{R}"


def format_reset_line(value: str | None) -> str:
    absolute = format_absolute_reset(value)
    countdown = format_countdown(value)
    if absolute and countdown:
        if absolute in countdown:
            return f"{FGY}{NF['hourglass']} {absolute}{R}"
        return f"{FGY}{NF['hourglass']} {absolute} ({countdown}){R}"
    if absolute:
        return f"{FGY}{NF['hourglass']} {absolute}{R}"
    if countdown:
        return f"{FGY}{NF['hourglass']} {countdown}{R}"
    return f"{FGY}{NF['hourglass']} —{R}"


def format_quota_line(label: str, quota: dict[str, Any]) -> str | None:
    pct = quota.get("usedPercent")
    if not isinstance(pct, (int, float)):
        pct = quota.get("remainingFraction")
        if isinstance(pct, (int, float)):
            pct = (1.0 - float(pct)) * 100.0
    if not isinstance(pct, (int, float)):
        return None

    used = format_percent(pct)
    window = format_minutes_window(quota.get("windowMinutes"))
    reset_desc = quota.get("resetDescription")
    resets_at = quota.get("resetsAt")
    if isinstance(reset_desc, str) and reset_desc:
        reset = reset_desc
    elif isinstance(resets_at, str) and resets_at:
        absolute = format_absolute_reset(resets_at)
        reset = absolute or resets_at
    else:
        reset = ""

    parts = [f"{label}: {used} used"]
    if window:
        parts.append(f"of {window} window")
    if reset:
        parts.append(f"resets {reset}")
    return "  " + ", ".join(parts)


def format_window_line(label: str, window_data: dict[str, Any]) -> str | None:
    if not isinstance(window_data, dict):
        return None

    used = window_data.get("usedPercent")
    if not isinstance(used, (int, float)):
        remaining = window_data.get("remainingFraction")
        if isinstance(remaining, (int, float)):
            used = (1.0 - float(remaining)) * 100.0
    if not isinstance(used, (int, float)):
        return None

    parts = [f"{label}: {format_percent(used)} used"]
    window = format_minutes_window(window_data.get("windowMinutes"))
    if window:
        parts.append(f"of {window} window")

    reset_desc = window_data.get("resetDescription")
    resets_at = window_data.get("resetsAt")
    if isinstance(reset_desc, str) and reset_desc:
        parts.append(f"resets {reset_desc}")
    elif isinstance(resets_at, str) and resets_at:
        absolute = format_absolute_reset(resets_at)
        countdown = format_countdown(resets_at)
        if absolute and countdown:
            parts.append(f"resets {absolute} ({countdown})")
        elif absolute:
            parts.append(f"resets {absolute}")
        elif countdown:
            parts.append(f"resets {countdown}")
    return "  " + ", ".join(parts)


def format_antigravity_models(snapshot: dict[str, Any], width: int | None = None) -> list[str]:
    models = snapshot.get("models")
    if not isinstance(models, list) or not models:
        return []

    table_width = width or terminal_width()
    label_width = max(18, min(24, table_width - 44))
    header = f"  {FGY}{'Model':<{label_width}} Remaining                Resets In{R}"
    lines = [header]

    for model in models:
        if not isinstance(model, dict):
            continue
        label = str(model.get("label") or model.get("modelId") or "Unknown")
        remaining = model.get("remainingPercentage")
        if isinstance(remaining, (int, float)):
            pct = float(remaining)
            if pct <= 1.0:
                pct *= 100.0
            remaining_text = f"{round(pct):>3d}%"
            bar = format_bar(pct, 20)
        else:
            remaining_text = "  ?%"
            bar = format_bar(0.0, 20)
        reset_text = format_countdown(model.get("resetTime")) or "?"
        reset_text = reset_text.replace(f"{FGY}{NF['hourglass']} ", "").replace(R, "")
        lines.append(f"  {FGY}{label[:label_width]:<{label_width}}{R} {bar} {remaining_text:>3}  {FGY}{NF['hourglass']} {reset_text}{R}")

    return lines


def format_gemini_buckets(account_quota: dict[str, Any], width: int | None = None) -> list[str]:
    buckets = account_quota.get("buckets")
    if not isinstance(buckets, list) or not buckets:
        return []

    table_width = width or terminal_width()
    label_width = max(18, min(24, table_width - 44))
    lines = [f"  {FGY}{'Model':<{label_width}} Remaining                Resets In{R}"]
    for bucket in buckets:
        if not isinstance(bucket, dict):
            continue
        label = str(bucket.get("model") or bucket.get("label") or "Unknown")
        if "pro" in label.lower():
            continue
        used_pct = bucket.get("used_pct")
        remaining_fraction = bucket.get("remaining_fraction")
        if remaining_fraction is None:
            remaining_fraction = bucket.get("remainingFraction")
        if used_pct is None:
            used_pct = bucket.get("usedPercent")
        remaining_pct = None
        if isinstance(remaining_fraction, (int, float)):
            remaining_pct = round((float(remaining_fraction) * 100))
        elif isinstance(used_pct, (int, float)):
            remaining_pct = round(max(0.0, 100.0 - float(used_pct)))

        if remaining_pct is None:
            remaining_pct = 0
        bar = format_bar(float(remaining_pct), 20)

        reset_text = format_countdown(bucket.get("reset_time")) or "?"
        reset_text = reset_text.replace(f"{FGY}{NF['hourglass']} ", "").replace(R, "")
        lines.append(f"  {FGY}{label[:label_width]:<{label_width}}{R} {bar} {remaining_pct:>3d}%  {FGY}{NF['hourglass']} {reset_text}{R}")

    return lines


def quota_remaining_fraction(quota: dict[str, Any]) -> float | None:
    remaining = quota.get("remainingFraction")
    if isinstance(remaining, (int, float)):
        return max(0.0, min(1.0, float(remaining)))

    used = quota.get("usedPercent")
    if isinstance(used, (int, float)):
        return max(0.0, min(1.0, 1.0 - (float(used) / 100.0)))

    return None


def quota_display_label(provider_key: str, index: int, quota: dict[str, Any]) -> str:
    label = quota.get("label")
    if isinstance(label, str) and label:
        return label

    window = format_minutes_window(quota.get("windowMinutes"))
    if window:
        if provider_key == "claude" and index == 0 and window == "5h":
            return "5h"
        return window

    if provider_key == "claude" and index == 0:
        return "5h"
    if index == 1:
        return "1w"
    return "Current" if index == 0 else "Weekly"


def format_compact_quota_line(label: str, quota: dict[str, Any]) -> str | None:
    remaining_fraction = quota_remaining_fraction(quota)
    if remaining_fraction is None:
        return None

    bar = format_bar(remaining_fraction * 100.0, 20)
    remaining_pct = round(remaining_fraction * 100)
    reset_at = quota.get("resetTime") or quota.get("resetsAt")
    reset = format_countdown(reset_at) if isinstance(reset_at, str) else None
    reset_text = f"  {FGY}{NF['hourglass']} {reset}{R}" if reset else ""
    return f"  {FGY}{label:<8}{R} {bar} {remaining_pct:>3d}% left{reset_text}"


def format_codex_status_line(label: str, pct_left: Any, raw: dict[str, Any]) -> str | None:
    if not isinstance(pct_left, (int, float)):
        return None

    remaining_pct = max(0.0, min(100.0, float(pct_left)))
    bar = format_bar(remaining_pct, 20)
    reset_text: str | None = None
    if label == "1w":
        weekly_reset = raw.get("weeklyResetDescription")
        if isinstance(weekly_reset, str) and weekly_reset:
            reset_text = weekly_reset
        else:
            weekly_reset_at = raw.get("weeklyResetsAt")
            if isinstance(weekly_reset_at, str):
                reset_text = format_countdown(weekly_reset_at)
    else:
        raw_text = raw.get("rawText")
        if isinstance(raw_text, str):
            reset_text = extract_reset_hint(raw_text, label)

    reset_suffix = f"  {FGY}{NF['hourglass']} {reset_text}{R}" if reset_text else ""
    return f"  {FGY}{label:<8}{R} {bar} {round(remaining_pct):>3d}% left{reset_suffix}"


def week_of_year_label(value: str | None) -> str | None:
    dt = parse_iso_datetime(value)
    if not dt:
        return None
    return str(dt.astimezone().isocalendar().week)


def format_week_left_line(label: str, reset_at: str | None) -> str | None:
    dt = parse_iso_datetime(reset_at)
    if not dt:
        return None

    delta = dt - datetime.now(timezone.utc)
    seconds = max(0.0, min(float(7 * 24 * 60 * 60), delta.total_seconds()))
    pct_left = round((seconds / (7 * 24 * 60 * 60)) * 100.0)
    bar = format_bar(float(pct_left), 20, color=IB)
    week_no = week_of_year_label(reset_at)
    if not week_no:
        return None
    return f"  {FGY}{label:<8}{R} {bar} {FGY}{pct_left:>3d}% left of week {week_no}{R}"


def print_fitted_text(text: str, *, indent: str, width: int, color: str = FGY, dim: bool = True) -> None:
    clean = " ".join(str(text).split())
    if not clean:
        return

    fit_width = max(12, width - len(indent) - 2)
    prefix = f"{indent}{color}{DIM if dim else ''}"
    suffix = R
    if len(clean) > fit_width:
        clean = clean[: max(0, fit_width - 1)] + "…"
    print(f"{prefix}{clean}{suffix}")


def summarize_usage(item: dict[str, Any]) -> str:
    if isinstance(item.get("error"), str):
        return str(item["error"])

    if isinstance(item.get("message"), str) and item.get("status") == "error":
        return str(item["message"])

    usage = item.get("usage") if isinstance(item.get("usage"), dict) else item
    primary = nested_get(usage, "primary") or {}
    secondary = nested_get(usage, "secondary") or {}
    identity = nested_get(usage, "identity") or nested_get(item, "identity") or {}

    parts: list[str] = []

    used = primary.get("usedPercent")
    if isinstance(used, (int, float)):
        parts.append(f"{used:g}% used")

    window_minutes = primary.get("windowMinutes")
    if isinstance(window_minutes, (int, float)):
        parts.append(f"{int(window_minutes)}m window")

    reset_description = primary.get("resetDescription")
    if isinstance(reset_description, str) and reset_description:
        parts.append(reset_description)
    else:
        resets_at = primary.get("resetsAt")
        if isinstance(resets_at, str) and resets_at:
            parts.append(f"resets {resets_at}")

    secondary_used = secondary.get("usedPercent")
    if isinstance(secondary_used, (int, float)):
        parts.append(f"secondary {secondary_used:g}%")

    for key in (
        "creditsRemaining",
        "remainingCredits",
        "remaining",
        "balance",
        "keyUsage",
    ):
        value = usage.get(key)
        if isinstance(value, (int, float)):
            parts.append(f"{key}={value:g}")
            break

    openrouter_usage = usage.get("openRouterUsage")
    if isinstance(openrouter_usage, dict):
        balance = openrouter_usage.get("balance")
        key_usage = openrouter_usage.get("keyUsage")
        if isinstance(balance, (int, float)):
            parts.append(f"balance=${balance:g}")
        if isinstance(key_usage, (int, float)):
            parts.append(f"keyUsage=${key_usage:g}")

    if isinstance(identity, dict):
        for key in ("plan", "email", "org", "loginMethod", "providerID"):
            value = identity.get(key)
            if isinstance(value, str) and value:
                parts.append(f"{key}={value}")
                break

    if parts:
        return "; ".join(parts)

    return compact_json(item)


def run_command(
    cmd: list[str],
    timeout_seconds: float,
    extra_env: dict[str, str] | None = None,
) -> subprocess.CompletedProcess[str]:
    env = os.environ.copy()
    env["HOME"] = os.path.expanduser("~")
    env.update({k: v for k, v in load_dotenv_exports().items() if k not in env})
    if extra_env:
        env.update(extra_env)
    return subprocess.run(
        cmd,
        capture_output=True,
        text=True,
        timeout=timeout_seconds,
        env=env,
    )


def codexbar_path() -> list[str] | None:
    home = os.path.expanduser("~")
    brew_env = os.path.join(home, "DotCortex", "all", ".local", "bin", "dotcortex-brew-env")
    if os.path.exists(brew_env):
        return [brew_env, "codexbar"]
    if shutil.which("codexbar"):
        return ["codexbar"]
    return None


def fetch_codexbar(spec: ProviderSpec, timeout_seconds: float) -> dict[str, Any]:
    base = codexbar_path()
    if not base:
        return {
            "provider": spec.key,
            "label": spec.label,
            "state": "missing",
            "source": "",
            "summary": "codexbar is not installed or not reachable.",
            "raw": None,
        }

    cmd = base + [
        "usage",
        "--provider",
        spec.codexbar_provider or spec.key,
        "--json-only",
        *spec.codexbar_args,
    ]

    try:
        proc = run_command(cmd, timeout_seconds)
    except subprocess.TimeoutExpired:
        return {
            "provider": spec.key,
            "label": spec.label,
            "state": "timeout",
            "source": requested_source(spec),
            "summary": f"Timed out after {timeout_seconds:g}s waiting for codexbar.",
            "raw": None,
        }

    docs = parse_json_docs(proc.stdout)
    item = choose_provider_doc(docs, spec.codexbar_provider or spec.key)

    if isinstance(item, dict) and isinstance(item.get("error"), dict):
        error = item["error"]
        return {
            "provider": spec.key,
            "label": spec.label,
            "state": "error",
            "source": item.get("source", ""),
            "summary": error.get("message", "Provider error"),
            "raw": item,
        }

    if isinstance(item, dict):
        return {
            "provider": spec.key,
            "label": spec.label,
            "state": "ok",
            "source": item.get("source", "") or requested_source(spec),
            "summary": summarize_usage(item),
            "raw": item,
        }

    summary = last_meaningful_line(proc.stdout, proc.stderr)
    return {
        "provider": spec.key,
        "label": spec.label,
        "state": "error" if proc.returncode else "ok",
        "source": requested_source(spec) if proc.returncode else "",
        "summary": summary or "No JSON payload returned by codexbar.",
        "raw": None,
    }


def fetch_fuelcheck(spec: ProviderSpec, timeout_seconds: float) -> dict[str, Any]:
    exe = shutil.which("fuelcheck-cli")
    if not exe:
        return {
            "provider": spec.key,
            "label": spec.label,
            "state": "missing",
            "source": "",
            "summary": "fuelcheck-cli is not installed or not reachable.",
            "raw": None,
        }

    provider = spec.fuelcheck_provider or spec.key
    source_args_list: list[tuple[str, ...]] = [spec.fuelcheck_args]
    if spec.key == "claude":
        source_args_list = [("--source", "web"), ("--source", "oauth")]

    last_error: dict[str, Any] | None = None
    for source_args in source_args_list:
        cmd = [exe, "usage", "--provider", provider, "--json-only", *source_args]
        extra_env = (
            claude_cookie_env()
            if spec.key == "claude" and len(source_args) > 1 and source_args[1] == "web"
            else None
        )

        try:
            proc = run_command(cmd, timeout_seconds, extra_env=extra_env)
        except subprocess.TimeoutExpired:
            return {
                "provider": spec.key,
                "label": spec.label,
                "state": "timeout",
                "source": source_args[1] if len(source_args) > 1 else requested_fuelcheck_source(spec),
                "summary": f"Timed out after {timeout_seconds:g}s waiting for fuelcheck-cli.",
                "raw": None,
            }

        docs = parse_json_docs(proc.stdout)
        item = choose_provider_doc(docs, provider)

        if isinstance(item, dict):
            error = item.get("error")
            if isinstance(error, dict):
                message = error.get("message") or compact_json(error)
                last_error = {
                    "provider": spec.key,
                    "label": spec.label,
                    "state": "error",
                    "source": item.get("source", "") or (source_args[1] if len(source_args) > 1 else requested_fuelcheck_source(spec)),
                    "summary": message,
                    "raw": item,
                }
                continue
            if isinstance(error, str):
                last_error = {
                    "provider": spec.key,
                    "label": spec.label,
                    "state": "error",
                    "source": item.get("source", "") or (source_args[1] if len(source_args) > 1 else requested_fuelcheck_source(spec)),
                    "summary": error,
                    "raw": item,
                }
                continue
            if item.get("status") == "error":
                last_error = {
                    "provider": spec.key,
                    "label": spec.label,
                    "state": "error",
                    "source": item.get("source", "") or (source_args[1] if len(source_args) > 1 else requested_fuelcheck_source(spec)),
                    "summary": str(item.get("message") or compact_json(item)),
                    "raw": item,
                }
                continue

            return {
                "provider": spec.key,
                "label": spec.label,
                "state": "ok",
                "source": item.get("source", "") or (source_args[1] if len(source_args) > 1 else requested_fuelcheck_source(spec)),
                "summary": summarize_usage(item),
                "raw": item,
            }

    if last_error is not None:
        return last_error

    summary = last_meaningful_line(proc.stdout, proc.stderr)
    return {
        "provider": spec.key,
        "label": spec.label,
        "state": "error" if proc.returncode else "ok",
        "source": source_args_list[-1][1] if source_args_list and len(source_args_list[-1]) > 1 else requested_fuelcheck_source(spec),
        "summary": summary or "No JSON payload returned by fuelcheck-cli.",
        "raw": None,
    }

def deepinfra_token(timeout_seconds: float) -> tuple[str | None, str]:
    for env_name in ("DEEPINFRA_API_KEY", "DEEPINFRA_TOKEN", "DEEPINFRA_API_TOKEN"):
        token = os.environ.get(env_name, "").strip()
        if token:
            return token, f"env:{env_name}"

    if not shutil.which("deepctl"):
        return None, "deepctl"

    try:
        proc = run_command(["deepctl", "auth", "token"], timeout_seconds)
    except subprocess.TimeoutExpired:
        return None, "deepctl"

    for line in reversed(proc.stdout.splitlines()):
        line = line.strip()
        if line:
            return line, "deepctl"
    return None, "deepctl"


def fetch_deepinfra(timeout_seconds: float) -> dict[str, Any]:
    token, token_source = deepinfra_token(timeout_seconds)
    if not token:
        return {
            "provider": "deepinfra",
            "label": "DeepInfra",
            "state": "missing",
            "source": token_source,
            "summary": "No DeepInfra token found in env or deepctl auth token.",
            "raw": None,
        }

    url = os.environ.get("DEEPINFRA_USAGE_URL", "https://api.deepinfra.com/v1/usage")
    request = urllib.request.Request(
        url,
        headers={
            "Authorization": f"Bearer {token}",
            "Accept": "application/json",
            "User-Agent": "DotCortex ai-usage",
        },
    )

    try:
        with urllib.request.urlopen(request, timeout=timeout_seconds) as response:
            body = response.read().decode("utf-8", errors="replace")
            data = json.loads(body)
    except urllib.error.HTTPError as error:
        body = error.read().decode("utf-8", errors="replace")
        try:
            data = json.loads(body)
            detail = data.get("detail") if isinstance(data, dict) else None
        except json.JSONDecodeError:
            detail = body.strip()
            data = None
        return {
            "provider": "deepinfra",
            "label": "DeepInfra",
            "state": "error",
            "source": "api",
            "summary": f"{url} returned HTTP {error.code}" + (f": {detail}" if detail else ""),
            "raw": data,
        }
    except urllib.error.URLError as error:
        return {
            "provider": "deepinfra",
            "label": "DeepInfra",
            "state": "error",
            "source": "api",
            "summary": f"Request failed: {error.reason}",
            "raw": None,
        }
    except json.JSONDecodeError:
        return {
            "provider": "deepinfra",
            "label": "DeepInfra",
            "state": "error",
            "source": "api",
            "summary": "Response was not valid JSON.",
            "raw": None,
        }

    if isinstance(data, dict) and data.get("detail"):
        return {
            "provider": "deepinfra",
            "label": "DeepInfra",
            "state": "error",
            "source": "api",
            "summary": str(data.get("detail")),
            "raw": data,
        }

    return {
        "provider": "deepinfra",
        "label": "DeepInfra",
        "state": "ok",
        "source": "api",
        "summary": compact_json(data),
        "raw": data,
    }


def fetch_provider(spec: ProviderSpec, timeout_seconds: float) -> dict[str, Any]:
    if spec.key == "gemini":
        result = fetch_gemini_cli_usage(spec, timeout_seconds)
        if result["state"] == "missing" and spec.fuelcheck_provider:
            return fetch_fuelcheck(spec, timeout_seconds)
        return result

    if spec.key == "claude":
        return fetch_fuelcheck(spec, timeout_seconds)

    if spec.kind == "deepinfra":
        return fetch_deepinfra(timeout_seconds)

    if spec.kind == "antigravity_usage":
        result = fetch_antigravity_usage(spec, timeout_seconds)
        if result["state"] == "missing" and spec.codexbar_provider:
            return fetch_codexbar(spec, timeout_seconds)
        return result

    if spec.kind == "fuelcheck":
        result = fetch_fuelcheck(spec, timeout_seconds)
        if result["state"] == "missing" and spec.codexbar_provider:
            return fetch_codexbar(spec, timeout_seconds)
        return result

    return fetch_codexbar(spec, timeout_seconds)


def fetch_provider_with_index(args: tuple[int, ProviderSpec, float]) -> tuple[int, dict[str, Any]]:
    index, spec, timeout_seconds = args
    return index, fetch_provider(spec, timeout_seconds)


def provider_lines(item: dict[str, Any]) -> list[str]:
    label = item["label"]
    state = item["state"]
    source = item.get("source", "")
    raw = item.get("raw")

    lines = [f"{label} [{source or '-'}] {state}"]
    windows: list[tuple[str, dict[str, Any]]] = []
    model_lines: list[str] = []

    if isinstance(raw, dict):
        windows = detect_windows(raw)
        for index, (window_label, window_data) in enumerate(windows):
            normalized_label = window_label
            if index == 1 and normalized_label == "Secondary":
                normalized_label = "Weekly"
            line = format_window_line(normalized_label, window_data)
            if line:
                lines.append(f"  {line}")

        identity = nested_get(raw, "usage", "identity") or raw.get("identity") or {}
        if isinstance(identity, dict):
            for key, title in (
                ("plan", "Plan"),
                ("loginMethod", "Login"),
                ("email", "Account"),
                ("providerID", "Provider"),
            ):
                value = identity.get(key)
                if isinstance(value, str) and value:
                    lines.append(f"  {title}: {value}")

        if item["provider"] == "antigravity":
            email = raw.get("email")
            method = raw.get("method")
            if isinstance(email, str) and email:
                lines.append(f"  Account: {email}")
            if isinstance(method, str) and method:
                lines.append(f"  Method: {method}")
            model_lines = format_antigravity_models(raw, width=width)
            if model_lines:
                lines.extend(model_lines)

    if len(lines) == 1:
        lines.append(f"  {item['summary']}")
    elif item["state"] == "ok" and item["summary"] and not windows and not model_lines:
        lines.append(f"  {item['summary']}")
    elif item["state"] != "ok" and item["summary"]:
        lines.append(f"  {item['summary']}")

    return lines

def main() -> int:
    args = parse_args()

    if args.providers == "all":
        selected = list(DEFAULT_PROVIDER_KEYS)
    else:
        selected = [chunk.strip().lower() for chunk in args.providers.split(",") if chunk.strip()]

    known = {spec.key for spec in PROVIDERS}
    unknown = [name for name in selected if name not in known]
    if unknown:
        eprint("Unknown providers:", ", ".join(unknown))
        return 2

    def collect_results() -> list[dict[str, Any]]:
        selected_specs = [spec for spec in PROVIDERS if spec.key in selected]
        tasks = [
            (index, spec, args.timeout)
            for index, spec in enumerate(selected_specs)
        ]
        results: list[dict[str, Any] | None] = [None for _ in tasks]
        if not tasks:
            return []

        done_count = 0
        done_lock = threading.Lock()

        def worker(index: int, spec: ProviderSpec, timeout_seconds: float) -> None:
            nonlocal done_count
            try:
                result = fetch_provider(spec, timeout_seconds)
            except Exception as exc:
                result = {
                    "provider": spec.key,
                    "label": spec.label,
                    "state": "error",
                    "source": "",
                    "summary": f"Unhandled error: {exc}",
                    "raw": None,
                }
            results[index] = result
            with done_lock:
                done_count += 1

        for index, spec, timeout_seconds in tasks:
            threading.Thread(
                target=worker,
                args=(index, spec, timeout_seconds),
                daemon=True,
                name=f"ai-usage-{spec.key}",
            ).start()

        while True:
            if stop_event.is_set():
                raise KeyboardInterrupt
            with done_lock:
                if done_count >= len(tasks):
                    break
            time.sleep(0.1)

        return [result for result in results if result is not None]

    def start_watch_input_thread(stop_event: threading.Event) -> tuple[threading.Thread | None, list[tuple[int, Any]]]:
        if not sys.stdin.isatty():
            return None, []

        def watcher() -> None:
            fd = sys.stdin.fileno()
            old_attrs = termios.tcgetattr(fd)
            try:
                tty.setcbreak(fd)
                while not stop_event.is_set():
                    readable, _, _ = select.select([fd], [], [], 0.2)
                    if not readable:
                        continue
                    try:
                        ch = os.read(fd, 1)
                    except OSError:
                        continue
                    if ch in {b"q", b"Q", b"\x03", b"\x04"}:
                        stop_event.set()
                        return
            finally:
                termios.tcsetattr(fd, termios.TCSADRAIN, old_attrs)

        thread = threading.Thread(target=watcher, name="ai-usage-watch", daemon=True)
        thread.start()
        return thread, []

    stop_event = threading.Event()
    watch_thread: threading.Thread | None = None
    if args.watch:
        watch_thread, _ = start_watch_input_thread(stop_event)

    try:
        while True:
            results = collect_results()

            if args.json:
                print(json.dumps(results, indent=2, ensure_ascii=True))
            else:
                if args.watch:
                    print("\033[2J\033[H", end="")
                render_text(results)

            if not args.watch:
                return 0

            if stop_event.wait(max(1.0, args.interval)):
                return 0
    except KeyboardInterrupt:
        return 130
    finally:
        stop_event.set()
        if watch_thread and watch_thread.is_alive():
            watch_thread.join(timeout=0.5)


if __name__ == "__main__":
    raise SystemExit(main())

def iso_now() -> str:
    return datetime.now(timezone.utc).isoformat()


def parse_iso(ts: str | None) -> datetime | None:
    if not ts or not isinstance(ts, str):
        return None
    try:
        return datetime.fromisoformat(ts.replace("Z", "+00:00"))
    except ValueError:
        return None


def minutes_until(ts: str | None) -> int | None:
    dt = parse_iso(ts)
    if dt is None:
        return None
    delta = dt - datetime.now(timezone.utc)
    return max(0, int(delta.total_seconds() // 60))


def normalize_window(raw: dict[str, Any] | None) -> dict[str, Any] | None:
    if not isinstance(raw, dict):
        return None
    used = raw.get("usedPercent")
    if used is None:
        return None
    try:
        used_val = float(used)
    except (TypeError, ValueError):
        return None
    remaining = max(0.0, 100.0 - used_val)
    resets_at = raw.get("resetsAt")
    return {
        "used_percent": used_val,
        "remaining_percent": remaining,
        "window_minutes": raw.get("windowMinutes"),
        "resets_at": resets_at,
        "minutes_to_reset": minutes_until(resets_at),
        "reset_description": raw.get("resetDescription"),
    }


def canonical_budget_kind(provider: str, usage: dict[str, Any], item: dict[str, Any]) -> str:
    if usage.get("openRouterUsage") is not None:
        return "credit_balance"
    if any(isinstance(usage.get(key), dict) for key in ("primary", "secondary", "tertiary")):
        return "windowed_quota"
    if provider == "deepinfra":
        return "token_cost"
    return "unknown"


def normalize_item(item: dict[str, Any]) -> dict[str, Any]:
    provider = str(item.get("provider") or "unknown")
    raw = item.get("raw") if isinstance(item.get("raw"), dict) else {}
    usage = raw.get("usage") if isinstance(raw.get("usage"), dict) else {}
    openrouter_usage = usage.get("openRouterUsage") if isinstance(usage.get("openRouterUsage"), dict) else None
    provider_cost = usage.get("providerCost") if isinstance(usage.get("providerCost"), dict) else None
    windows = {
        "short": normalize_window(usage.get("primary")),
        "long": normalize_window(usage.get("secondary")),
        "extra": normalize_window(usage.get("tertiary")),
    }
    windows = {k: v for k, v in windows.items() if v is not None}
    status = str(item.get("state") or "unknown")
    error = raw.get("error") if isinstance(raw.get("error"), dict) else None
    login_method = None
    identity = usage.get("identity") if isinstance(usage.get("identity"), dict) else {}
    if isinstance(identity, dict):
        login_method = identity.get("loginMethod")
    if not login_method:
        login_method = usage.get("loginMethod")
    budget_kind = canonical_budget_kind(provider, usage, item)

    burnable = False
    conserve = False
    healthy = status == "ok"

    short = windows.get("short")
    longw = windows.get("long")
    if provider == "codex" and short:
        mins = short.get("minutes_to_reset")
        rem = short.get("remaining_percent")
        if mins is not None and rem is not None and mins <= 480 and rem >= 20:
            burnable = True
        if longw:
            mins2 = longw.get("minutes_to_reset")
            rem2 = longw.get("remaining_percent")
            if mins2 is not None and rem2 is not None and mins2 <= 720 and rem2 >= 5:
                burnable = True
    if provider in {"openrouter", "deepinfra"}:
        conserve = True
    if provider == "gemini" and short and float(short.get("used_percent", 0)) >= 100:
        healthy = False
    if error is not None:
        healthy = False

    priority_score = 0
    if healthy:
        priority_score += 20
    if burnable:
        priority_score += 30
    if conserve:
        priority_score -= 20
    if provider == "openrouter" and openrouter_usage:
        if float(openrouter_usage.get("balance", 0) or 0) <= 0:
            priority_score -= 15
    return {
        "status": "unavailable" if status == "error" else status,
        "label": item.get("label"),
        "source": item.get("source"),
        "login_method": login_method,
        "budget": {"kind": budget_kind},
        "windows": windows,
        "credits": raw.get("credits") if isinstance(raw.get("credits"), dict) else None,
        "provider_cost": provider_cost,
        "openrouter_usage": openrouter_usage,
        "burnable": burnable,
        "conserve": conserve,
        "healthy": healthy,
        "priority_score": priority_score,
        "raw_summary": item.get("summary"),
        "error": error,
    }


def recommend_mode(providers: dict[str, Any]) -> str:
    codex = providers.get("codex") or {}
    short = ((codex.get("windows") or {}).get("short") or {})
    longw = ((codex.get("windows") or {}).get("long") or {})
    if codex.get("healthy"):
        mins = short.get("minutes_to_reset")
        rem = short.get("remaining_percent")
        if mins is not None and rem is not None and mins <= 480 and rem >= 20:
            return "burn"
        mins2 = longw.get("minutes_to_reset")
        rem2 = longw.get("remaining_percent")
        if mins2 is not None and rem2 is not None and mins2 <= 720 and rem2 >= 5:
            return "burn"
    openrouter = providers.get("openrouter") or {}
    gemini = providers.get("gemini") or {}
    nvidia_available = True
    if openrouter.get("healthy") is False and gemini.get("healthy") is not False and nvidia_available:
        return "conserve"
    return "normal"


def main() -> int:
    args = parse_args()
    cmd = [args.ai_usage, "--json", "--providers", args.providers]
    proc = subprocess.run(cmd, capture_output=True, text=True)
    if proc.returncode != 0:
        raise SystemExit(proc.stderr.strip() or proc.stdout.strip() or f"ai-usage failed with {proc.returncode}")
    payload = json.loads(proc.stdout)
    if not isinstance(payload, list):
        raise SystemExit("ai-usage JSON was not a list")
    providers = {}
    for item in payload:
        if not isinstance(item, dict):
            continue
        provider = str(item.get("provider") or "unknown")
        providers[provider] = normalize_item(item)
    mode = recommend_mode(providers) if args.mode == "auto" else args.mode
    state = {
        "generated_at": iso_now(),
        "source": "ai-usage --json",
        "ttl_seconds": 300,
        "mode_recommendation": mode,
        "providers": providers,
    }
    text = json.dumps(state, indent=2, sort_keys=True)
    print(text)
    if args.write_cache:
        CACHE_PATH.parent.mkdir(parents=True, exist_ok=True)
        CACHE_PATH.write_text(text + "\n", encoding="utf-8")
    return 0

    raise SystemExit(main())

        if key in seen:
            continue
        seen.add(key)
        score, detail = candidate_score(candidate, mode, usage_state)
        scored.append({
            **candidate,
            "score": score,
            "healthy": detail["healthy"],
            "provider_state": detail["provider_state"],
        })
    scored.sort(key=lambda item: item["score"], reverse=True)
    payload = {
        "task": args.task,
        "mode": mode,
        "generated_from": args.usage_state,
        "candidates": scored,
    }
    if args.json:
        print(json.dumps(payload, indent=2))
    else:
        print(f"task={args.task} mode={mode}")
        for item in scored:
            state = "healthy" if item["healthy"] else "degraded"
            print(f"- {item['provider']} / {item['model']} [{item['score']}] ({state}) — {item['why']}")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())

** cursor-agent - stable launcher for versioned Cursor installs
Cursor installs its CLI into a versioned directory under
=~/.local/share/cursor-agent/versions=. DotCortex keeps a stable wrapper in
=~/.local/bin/cursor-agent= so shell usage does not depend on whichever
versioned directory happens to exist.

#+BEGIN_SRC bash :tangle linux/.local/bin/cursor-agent :shebang "#!/usr/bin/env bash"
set -euo pipefail

root="${CURSOR_AGENT_ROOT:-$HOME/.local/share/cursor-agent/versions}"

if [ ! -d "$root" ]; then
  echo "cursor-agent: no install root at $root" >&2
  exit 1
fi

latest="$(
  find "$root" -mindepth 1 -maxdepth 1 -type d 2>/dev/null \
    | LC_ALL=C sort \
    | tail -n 1
)"

if [ -z "$latest" ]; then
  echo "cursor-agent: no installed versions under $root" >&2
  exit 1
fi

bin="$latest/cursor-agent"

if [ ! -x "$bin" ]; then
  echo "cursor-agent: latest install missing executable: $bin" >&2
  exit 1
fi

exec "$bin" "$@"

** dotcortex-clean-backups - quarantine backup files from DotCortex
This helper moves backup-like files out of =~/DotCortex= into a timestamped
folder under =~/Downloads=. It preserves the relative path so you can inspect
the damage without mixing it back into the live tree.

Usage:

- Preview only:
  #+BEGIN_EXAMPLE
  dotcortex-clean-backups
  #+END_EXAMPLE
- Apply the move:
  #+BEGIN_EXAMPLE
  dotcortex-clean-backups -y
  #+END_EXAMPLE

#+BEGIN_SRC python :tangle all/.local/bin/dotcortex-clean-backups :shebang "#!/usr/bin/env python3" :tangle-mode (identity #o755)
from __future__ import annotations

import argparse
import csv
import fnmatch
import os
import shutil
import sys
from collections import Counter
from datetime import datetime
from pathlib import Path

DEFAULT_ROOT = Path.home() / "DotCortex"
DEFAULT_DOWNLOADS = Path.home() / "Downloads"
DEST_PREFIX = "DotCortex-backups"
BACKUP_PATTERNS = (
    "*.bak",
    "*.bak.*",
    "*.backup-*",
    "*~",
    "#*#",
    ".#*",
    "*.orig",
    "*.rej",
)


def parse_args() -> argparse.Namespace:
    parser = argparse.ArgumentParser(
        description="Quarantine backup-like files out of ~/DotCortex into ~/Downloads."
    )
    parser.add_argument("--root", default=str(DEFAULT_ROOT), help="DotCortex repo root")
    parser.add_argument(
        "--downloads",
        default=str(DEFAULT_DOWNLOADS),
        help="Destination base directory (default: %(default)s)",
    )
    parser.add_argument(
        "--pattern",
        action="append",
        default=[],
        help="Extra filename glob to treat as a backup pattern",
    )
    parser.add_argument(
        "--timestamp",
        default=None,
        help="Override the timestamp used in the archive directory name",
    )
    parser.add_argument("-n", "--dry-run", action="store_true", help="Preview only")
    parser.add_argument("-y", "--yes", action="store_true", help="Apply the move")
    parser.add_argument(
        "--auto",
        action="store_true",
        help="Non-interactive mode (requires -n or -y)",
    )
    return parser.parse_args()


def eprint(*args: object) -> None:
    print(*args, file=sys.stderr)


def classify(name: str) -> str:
    if name.startswith(".#"):
        return "emacs-lock"
    if name.startswith("#") and name.endswith("#"):
        return "emacs-save"
    if name.endswith("~"):
        return "tilde"
    if name.endswith(".orig"):
        return "orig"
    if name.endswith(".rej"):
        return "rej"
    if ".bak" in name or ".backup-" in name:
        return "bak"
    return "backup"


def matches(name: str, patterns: tuple[str, ...]) -> bool:
    return any(fnmatch.fnmatchcase(name, pattern) for pattern in patterns)


def is_within(path: Path, parent: Path) -> bool:
    try:
        path.resolve().relative_to(parent.resolve())
        return True
    except ValueError:
        return False


def collect_candidates(root: Path, patterns: tuple[str, ...]) -> list[Path]:
    candidates: list[Path] = []
    for dirpath, dirnames, filenames in os.walk(root):
        dirnames[:] = [d for d in dirnames if d != ".git"]
        current = Path(dirpath)
        for name in filenames:
            if not matches(name, patterns):
                continue
            path = current / name
            if path.exists() and not path.is_dir():
                candidates.append(path)
    return sorted(set(candidates), key=lambda p: str(p.relative_to(root)))


def unique_dest_root(downloads: Path, stamp: str) -> Path:
    dest_root = downloads / f"{DEST_PREFIX}-{stamp}"
    suffix = 2
    while dest_root.exists():
        dest_root = downloads / f"{DEST_PREFIX}-{stamp}-{suffix}"
        suffix += 1
    return dest_root


def human_bytes(size: int) -> str:
    units = ["B", "KiB", "MiB", "GiB", "TiB"]
    value = float(size)
    for unit in units:
        if value < 1024.0 or unit == units[-1]:
            return f"{value:.1f} {unit}" if unit != "B" else f"{int(value)} {unit}"
        value /= 1024.0
    return f"{size} B"


def write_manifest(dest_root: Path, entries: list[dict[str, object]]) -> None:
    manifest = dest_root / "manifest.tsv"
    with manifest.open("w", encoding="utf-8", newline="") as handle:
        writer = csv.writer(handle, delimiter="\t")
        writer.writerow(["kind", "bytes", "source", "dest"])
        for entry in entries:
            writer.writerow(
                [
                    entry["kind"],
                    entry["bytes"],
                    entry["source"],
                    entry["dest"],
                ]
            )


def write_triage_notes(
    dest_root: Path,
    root: Path,
    entries: list[dict[str, object]],
    counts: Counter[str],
    bytes_by_kind: Counter[str],
) -> None:
    triage = dest_root / "TRIAGE.md"
    total_bytes = sum(int(entry["bytes"]) for entry in entries)
    with triage.open("w", encoding="utf-8") as handle:
        handle.write("# DotCortex backup quarantine\n\n")
        handle.write(f"- Created: {datetime.now().astimezone().isoformat()}\n")
        handle.write(f"- Source root: `{root}`\n")
        handle.write(f"- Items moved: {len(entries)}\n")
        handle.write(f"- Total size: {human_bytes(total_bytes)}\n")
        handle.write(f"- Manifest: `manifest.tsv`\n\n")
        handle.write("## By kind\n\n")
        for kind in sorted(counts):
            handle.write(
                f"- `{kind}`: {counts[kind]} items, {human_bytes(bytes_by_kind[kind])}\n"
            )
        handle.write("\n## Notes\n\n")
        handle.write(
            "- Paths mirror the original DotCortex layout so you can triage safely.\n"
        )
        handle.write(
            "- Review `manifest.tsv` first if you need to restore anything manually.\n"
        )


def preview(entries: list[dict[str, object]], dest_root: Path) -> None:
    total_bytes = sum(int(entry["bytes"]) for entry in entries)
    print(f"Archive root: {dest_root}")
    print(f"Items found: {len(entries)}")
    print(f"Total size: {human_bytes(total_bytes)}")
    print("Sample:")
    for entry in entries[:30]:
        print(f"  - {entry['source']}")


def main() -> int:
    args = parse_args()
    root = Path(args.root).expanduser().resolve()
    downloads = Path(args.downloads).expanduser()
    downloads.mkdir(parents=True, exist_ok=True)
    downloads = downloads.resolve()

    if not root.is_dir():
        eprint(f"dotcortex-clean-backups: root not found: {root}")
        return 2

    if is_within(downloads, root):
        eprint("dotcortex-clean-backups: refusing to archive inside DotCortex")
        return 2

    patterns = BACKUP_PATTERNS + tuple(args.pattern)
    candidates = collect_candidates(root, patterns)
    if not candidates:
        print(f"No backup-like files found under {root}")
        return 0

    if args.auto and not (args.dry_run or args.yes):
        eprint("dotcortex-clean-backups: --auto requires -n or -y")
        return 2

    stamp = args.timestamp or datetime.now().strftime("%Y%m%d-%H%M%S")
    dest_root = unique_dest_root(downloads, stamp)
    entries: list[dict[str, object]] = []
    counts: Counter[str] = Counter()
    bytes_by_kind: Counter[str] = Counter()

    for path in candidates:
        rel = path.relative_to(root)
        dest = dest_root / rel
        size = path.stat().st_size
        kind = classify(path.name)
        entries.append(
            {
                "kind": kind,
                "bytes": size,
                "source": str(path),
                "dest": str(dest),
            }
        )
        counts[kind] += 1
        bytes_by_kind[kind] += size

    preview(entries, dest_root)

    if args.dry_run:
        print("Dry run only; nothing moved.")
        return 0

    if not args.yes:
        try:
            answer = input(
                f"Type the archive root path to move these files, or 'd' for dry run: "
            ).strip()
        except EOFError:
            answer = "d"
        if answer == "d":
            print("Dry run only; nothing moved.")
            return 0
        if answer != str(dest_root):
            eprint("dotcortex-clean-backups: confirmation mismatch")
            return 2

    dest_root.mkdir(parents=True, exist_ok=False)

    moved_entries: list[dict[str, object]] = []
    moved_counts: Counter[str] = Counter()
    moved_bytes_by_kind: Counter[str] = Counter()
    failures: list[tuple[str, str]] = []
    for entry in entries:
        source = Path(str(entry["source"]))
        dest = Path(str(entry["dest"]))
        dest.parent.mkdir(parents=True, exist_ok=True)
        try:
            shutil.move(str(source), str(dest))
        except Exception as exc:  # pragma: no cover - surfaced to the user
            failures.append((str(source), str(exc)))
            continue
        moved_entries.append(entry)
        moved_counts[str(entry["kind"])] += 1
        moved_bytes_by_kind[str(entry["kind"])] += int(entry["bytes"])

    write_manifest(dest_root, moved_entries)
    write_triage_notes(dest_root, root, moved_entries, moved_counts, moved_bytes_by_kind)

    print(f"Archive created: {dest_root}")
    print(f"Moved: {len(moved_entries)}")
    if failures:
        print(f"Failures: {len(failures)}")
        for source, message in failures[:10]:
            print(f"  - {source}: {message}")
        return 1

    return 0


if __name__ == "__main__":
    raise SystemExit(main())
