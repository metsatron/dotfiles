* Purpose
Reproducible Flatpak state from your dotfiles, Guix style. One manifest file is the source of truth. Each line declares origin, id, branch, and scope. Scope is either =user= or =system=. The apply step is additive by default; exact enforcement with removals is opt in.

* Flatpak single manifest
Columns: ORIGIN, ID, BRANCH, SCOPE. Comments and blanks allowed.

* Remotes setup
Idempotent helper to ensure remotes exist before applying the manifest. Adjust the Pied URL if you use a custom one.
#+BEGIN_SRC bash :tangle linux/.local/bin/flatpak-remotes-setup :shebang "#!/usr/bin/env bash"
#!/usr/bin/env bash
set -euo pipefail

CLEAN_ENV=(env -i HOME="$HOME" PATH="/usr/sbin:/usr/bin:/bin"
  XDG_RUNTIME_DIR="${XDG_RUNTIME_DIR:-/run/user/$(id -u)}"
  DBUS_SESSION_BUS_ADDRESS="${DBUS_SESSION_BUS_ADDRESS:-}")

unset_ssl() { unset SSL_CERT_FILE SSL_CERT_DIR CURL_CA_BUNDLE GIT_SSL_CAINFO; }

add_remote() {
  local scope="$1" name="$2" url="$3"
  unset_ssl
  if [ "$scope" = "system" ]; then
    if flatpak remote-list --columns=name | grep -qx "$name"; then return 0; fi
    if ! "${CLEAN_ENV[@]}" flatpak remote-add --if-not-exists "$name" "$url"; then
      mkdir -p "$HOME/.cache/flatpak-remotes"
      local f="$HOME/.cache/flatpak-remotes/${name}.flatpakrepo"
      curl -fsSL "$url" -o "$f"
      "${CLEAN_ENV[@]}" flatpak remote-add --if-not-exists --from "$name" "$f"
    fi
  else
    if flatpak --user remote-list --columns=name | grep -qx "$name"; then return 0; fi
    if ! "${CLEAN_ENV[@]}" flatpak --user remote-add --if-not-exists "$name" "$url"; then
      mkdir -p "$HOME/.cache/flatpak-remotes"
      local f="$HOME/.cache/flatpak-remotes/${name}.flatpakrepo"
      curl -fsSL "$url" -o "$f"
      "${CLEAN_ENV[@]}" flatpak --user remote-add --if-not-exists --from "$name" "$f"
    fi
  fi
}

# Flathub for both scopes
add_remote system flathub https://dl.flathub.org/repo/flathub.flatpakrepo
add_remote user   flathub https://dl.flathub.org/repo/flathub.flatpakrepo

# Pied: set the real URL if you have one (placeholder keeps idempotence)
if ! flatpak remote-list --columns=name | grep -qx pied-origin; then
  add_remote system pied-origin https://dl.flathub.org/repo/flathub.flatpakrepo || true
fi

* Capture live state into the manifest
Merges your current user and system apps into =apps.ssv=. Keeps comments. Does not remove existing lines.
#+BEGIN_SRC bash :tangle linux/.local/bin/flatpak-capture :shebang "#!/usr/bin/env bash"
#!/usr/bin/env bash
set -euo pipefail
OUT="${HOME}/DotCortex/linux/.flatpak/manifest/apps.ssv"

# Run flatpak in a clean env to avoid SSL/Guix contamination
CLEAN_ENV=(env -i HOME="$HOME" PATH="/usr/sbin:/usr/bin:/bin"
  XDG_RUNTIME_DIR="${XDG_RUNTIME_DIR:-/run/user/$(id -u)}"
  DBUS_SESSION_BUS_ADDRESS="${DBUS_SESSION_BUS_ADDRESS:-}")

dump_scope() { # user|system
  local scope="$1"
  if [ "$scope" = user ]; then
    "${CLEAN_ENV[@]}" flatpak --user list --app --columns=origin,application,branch
  else
    "${CLEAN_ENV[@]}" flatpak --system list --app --columns=origin,application,branch
  fi | awk -v s="$scope" '
    NF>=3 { printf "%s %s %s %s\n", $1, $2, $3, s }
  '
}

tmp="$(mktemp)"; trap 'rm -f "$tmp"' EXIT

{
  # keep comments / blanks
  [ -f "$OUT" ] && grep -E '^\s*#|^\s*$' "$OUT" || true
  # merge existing data + fresh dumps, uniq
  {
    [ -f "$OUT" ] && grep -Ev '^\s*#|^\s*$' "$OUT" || true
    dump_scope user
    dump_scope system
  } | awk 'NF>=4 {print $1, $2, $3, $4}' | sort -u
} > "$tmp"

mv "$tmp" "$OUT"
echo "Wrote $OUT"
#+END_SRC

* Diff plan
Shows what would be installed and what would be removed to match the manifest.
#+BEGIN_SRC bash :tangle linux/.local/bin/flatpak-diff :shebang "#!/usr/bin/env bash"
set -euo pipefail
MAN="${HOME}/DotCortex/linux/.flatpak/manifest/apps.ssv"

desired_ids(){ awk '$0!~/^#/ && NF>=4{print $2, $4}' "$MAN" | sort -u; }
installed_ids(){
  { flatpak --user list --app --columns=application | awk 'NF{print $1, "user"}'
    flatpak --system list --app --columns=application | awk 'NF{print $1, "system"}'; } \
 \
  | sort -u
}

d="$(mktemp)"; i="$(mktemp)"; trap 'rm -f "$d" "$i"' EXIT
desired_ids >"$d"; installed_ids >"$i"

# desired minus installed -> to install
echo "[plan] to install:"
join -v1 <(cut -d' ' -f1 "$d") <(cut -d' ' -f1 "$i") \
  | while read -r id; do grep -F "$id" "$d" | sed 's/^/  + /'; done || true

# installed minus desired -> to uninstall
echo "[plan] to uninstall:"
join -v1 <(cut -d' ' -f1 "$i") <(cut -d' ' -f1 "$d") \
  | while read -r id; do grep -F "$id" "$i" | sed 's/^/  - /'; done || true
#+END_SRC

* Apply manifest with safe elevation
Default behavior: **enforce exact state** (install + update + uninstall).
System-scoped actions try Polkit first; if denied, retry the specific command with sudo.
#+BEGIN_SRC bash :tangle linux/.local/bin/flatpak-apply :shebang "#!/usr/bin/env bash"
# ~/.local/bin/flatpak-apply  (tangle this from flatpak.org)
#!/usr/bin/env bash
set -euo pipefail

MAN="${HOME}/DotCortex/linux/.flatpak/manifest/apps.ssv"

# defaults: installs + updates; removals only if UNINSTALL=1
ENFORCE="${ENFORCE:-1}"
UNINSTALL="${UNINSTALL:-1}"
UPDATE="${UPDATE:-1}"
FORCE_SUDO_SYSTEM="${FORCE_SUDO_SYSTEM:-0}"
DEBUG="${DEBUG:-0}"

[ "$DEBUG" = 1 ] && set -x

# clean env
CLEAN_LANG="${LANG:-C.UTF-8}"
CLEAN_TERM="${TERM:-xterm-256color}"
CLEAN_PATH="/usr/sbin:/usr/bin:/bin:${HOME}/.local/bin"
CLEAN_ENV=(env -i HOME="$HOME" PATH="$CLEAN_PATH"
  LANG="$CLEAN_LANG" LC_ALL="$CLEAN_LANG"
  TERM="$CLEAN_TERM"
  XDG_RUNTIME_DIR="${XDG_RUNTIME_DIR:-/run/user/$(id -u)}"
  DBUS_SESSION_BUS_ADDRESS="${DBUS_SESSION_BUS_ADDRESS:-}")
tab=$'\t'
FS='[[:space:]]+'
unset_ssl(){ unset SSL_CERT_FILE SSL_CERT_DIR CURL_CA_BUNDLE GIT_SSL_CAINFO; }

ensure_remotes(){
  local helper="$HOME/.local/bin/flatpak-remotes-setup"
  [ -x "$helper" ] && "${CLEAN_ENV[@]}" "$helper" || true
}

run_fp(){ local scope="$1"; shift; unset_ssl
  if [ "$scope" = system ] && [ "$FORCE_SUDO_SYSTEM" = 1 ]; then
    sudo "${CLEAN_ENV[@]}" flatpak "$@"; return
  fi
  if "${CLEAN_ENV[@]}" flatpak "$@"; then return; fi
  if [ "$scope" = system ]; then
    echo "Retrying with sudo: flatpak $*" >&2
    sudo "${CLEAN_ENV[@]}" flatpak "$@"
  fi
}

# ----- parsing & plan (normalize to TAB-delimited) -----
desired_pairs(){
  awk -v FS="$FS" -v OFS="$tab" '$0!~/^#/ && NF>=4{print $2,$4}' "$MAN" | sort -u
}
installed_pairs(){
  { "${CLEAN_ENV[@]}" flatpak --user   list --app --columns=application | awk -v OFS="$tab" 'NF{print $1,"user"}'
    "${CLEAN_ENV[@]}" flatpak --system list --app --columns=application | awk -v OFS="$tab" 'NF{print $1,"system"}'; } \
  | sort -u
}

print_plan(){
  local d i; d="$(mktemp)"; i="$(mktemp)"; trap 'rm -f "$d" "$i"' RETURN
  desired_pairs >"$d"; installed_pairs >"$i"
  echo "== Plan =="
  echo "[to install]"
  join -v1 -t "$tab" <(cut -f1 "$d") <(cut -f1 "$i") | sed 's/^/  + /' || true
  echo "[to uninstall]"
  join -v1 -t "$tab" <(cut -f1 "$i") <(cut -f1 "$d") | sed 's/^/  - /' || true
}

already_installed(){ # id scope
  if [ "$2" = user ]; then "${CLEAN_ENV[@]}" flatpak --user   list --app --columns=application | grep -qx "$1"
  else                           "${CLEAN_ENV[@]}" flatpak --system list --app --columns=application | grep -qx "$1"
  fi
}

apply_installs(){
  echo "== Installing =="
  awk -v FS="$FS" '$0!~/^#/ && NF>=4{printf "  manifest: origin=%s id=%s branch=%s scope=%s\n",$1,$2,$3,$4}' "$MAN" || true

  awk -v FS="$FS" -v OFS="$tab" '$0!~/^#/ && NF>=4{print $1,$2,$3,$4}' "$MAN" |
  while read -r origin id branch scope; do
    [ -n "$origin$id$branch$scope" ] || continue
    already_installed "$id" "$scope" && continue
    echo "  -> installing $id ($scope) from $origin:$branch"
    args=(install -y); [ "$scope" = user ] && args+=(--user) || args+=(--system)
    run_fp "$scope" "${args[@]}" "$origin" "$id" "$branch"
  done
}

update_manifested(){
  [ "$UPDATE" = 1 ] || return 0
  echo "== Updating =="
  mapfile -t IDS_USER < <(awk -v FS="$FS" '$0!~/^#/ && NF>=4 && $4=="user"{print $2}' "$MAN" | sort -u)
  if [ "${#IDS_USER[@]}" -gt 0 ]; then
    "${CLEAN_ENV[@]}" flatpak --user update -y "${IDS_USER[@]}" || true
  fi

  mapfile -t IDS_SYS  < <(awk -v FS="$FS" '$0!~/^#/ && NF>=4 && $4=="system"{print $2}' "$MAN" | sort -u)
  if [ "${#IDS_SYS[@]}" -gt 0 ]; then
    run_fp system update -y "${IDS_SYS[@]}" || true
  fi
}

compute_extras(){ # installed-not-desired (id<TAB>scope)
  local d i; d="$(mktemp)"; i="$(mktemp)"; trap 'rm -f "$d" "$i"' RETURN
  desired_pairs >"$d"; installed_pairs >"$i"
  join -v2 -t "$tab" <(cut -f1 "$d") "$i" || true
}

apply_uninstalls(){
  [ "$ENFORCE" = 1 ] && [ "$UNINSTALL" = 1 ] || return 0
  echo "== Uninstalling extras =="
  compute_extras | while read -r id scope; do
    [ -n "${id:-}" ] || continue
    echo "  -> uninstalling $id ($scope)"
    if [ "$scope" = user ]; then
      "${CLEAN_ENV[@]}" flatpak --user uninstall -y "$id" || true
    else
      "${CLEAN_ENV[@]}" flatpak --system uninstall -y "$id" || true
    fi
  done
}

ensure_remotes
print_plan
apply_installs
update_manifested
apply_uninstalls
echo "Done."
#+END_SRC

** Reproducible Flatpak permissions & overrides
Store exact per-app permissions from your machine and re-apply them anywhere.
#+BEGIN_SRC bash :tangle linux/.local/bin/flatpak-perms-capture :shebang "#!/usr/bin/env bash" :mkdirp yes
#!/usr/bin/env bash
set -euo pipefail

BASE="${HOME}/DotCortex/linux/.flatpak/overrides-perms"
mkdir -p "$BASE"

CLEAN_ENV=(env -i HOME="$HOME" PATH="/usr/sbin:/usr/bin:/bin"
  LANG="${LANG:-C.UTF-8}" LC_ALL="${LC_ALL:-C.UTF-8}" TERM="${TERM:-xterm-256color}"
  XDG_RUNTIME_DIR="${XDG_RUNTIME_DIR:-/run/user/$(id -u)}"
  DBUS_SESSION_BUS_ADDRESS="${DBUS_SESSION_BUS_ADDRESS:-}")

awk '$0!~/^#/ && NF>=4 {print $2}' "$HOME/DotCortex/linux/.flatpak/manifest/apps.ssv" \
  | sort -u \
  | while read -r app; do
      out="$BASE/$app.perms"

      # Do not let a single failure kill the whole capture run
      if ! "${CLEAN_ENV[@]}" flatpak info --show-permissions "$app" >"$out" 2>/dev/null; then
        rm -f "$out"
        continue
      fi

      # Some apps may have no overrides at all, skip empty files
      if [ ! -s "$out" ]; then
        rm -f "$out"
        continue
      fi

      echo "wrote $out"
    done
#+END_SRC

** Apply script (maps the common fields → flatpak override)
#+BEGIN_SRC bash :tangle linux/.local/bin/flatpak-perms-apply :shebang "#!/usr/bin/env bash" :mkdirp yes
set -euo pipefail
BASE="${HOME}/DotCortex/linux/.flatpak/overrides-perms"

CLEAN_ENV=(env -i HOME="$HOME" PATH="/usr/sbin:/usr/bin:/bin"
  LANG="${LANG:-C.UTF-8}" LC_ALL="${LC_ALL:-C.UTF-8}" TERM="${TERM:-xterm-256color}"
  XDG_RUNTIME_DIR="${XDG_RUNTIME_DIR:-/run/user/$(id -u)}"
  DBUS_SESSION_BUS_ADDRESS="${DBUS_SESSION_BUS_ADDRESS:-}")

apply_app(){
  local app="$1" f="$BASE/$1.perms"
  [ -f "$f" ] || { echo "skip $app (no $f)"; return 0; }

  # Build override flags from sections
  awk -v app="$app" '
    function splitcsv(s, arr){ gsub(/ /,"",s); n=split(s,arr,/[,;]/); return n }
    BEGIN{ sec=""; }
    /^\[/ { gsub(/[\[\]]/,""); sec=$0; next }
    /^[[:space:]]*$/ { next }
    { line=$0 }
    sec=="Context" {
      if (line ~ /^sockets=/) {
        sub(/^sockets=/,"",line); n=splitcsv(line,a); for(i=1;i<=n;i++) printf "--socket=%s\n", a[i];
      } else if (line ~ /^devices=/) {
        sub(/^devices=/,"",line); n=splitcsv(line,a); for(i=1;i<=n;i++) printf "--device=%s\n", a[i];
      } else if (line ~ /^filesystems=/) {
        sub(/^filesystems=/,"",line); n=splitcsv(line,a); for(i=1;i<=n;i++) printf "--filesystem=%s\n", a[i];
      } else if (line ~ /^persistent=/) {
        sub(/^persistent=/,"",line); n=splitcsv(line,a); for(i=1;i<=n;i++) printf "--persist=%s\n", a[i];
      } else if (line ~ /^features=/) {
        sub(/^features=/,"",line); n=splitcsv(line,a); for(i=1;i<=n;i++) printf "--enable-features=%s\n", a[i];
      } else if (line ~ /^shared=/) {
        sub(/^shared=/,"",line); n=splitcsv(line,a); for(i=1;i<=n;i++) printf "--share=%s\n", a[i];
      } else if (line ~ /^system-bus=/) {
        sub(/^system-bus=/,"",line); n=splitcsv(line,a); for(i=1;i<=n;i++) printf "--system-talk-name=%s\n", a[i];
      } else if (line ~ /^session-bus=/) {
        sub(/^session-bus=/,"",line); n=splitcsv(line,a); for(i=1;i<=n;i++) printf "--talk-name=%s\n", a[i];
      } else if (line ~ /^env=/) {
        sub(/^env=/,"",line); n=split(line,a,/[;,]/);
        for(i=1;i<=n;i++){ split(a[i],kv,/=/); if(kv[1]!="") printf "--env=%s=%s\n", kv[1], kv[2]; }
      }
    }
  ' "$f" | sed '/^$/d' > "/tmp/.fpflags.$app"

  # Apply at user scope
  if [ -s "/tmp/.fpflags.$app" ]; then
    mapfile -t FLAGS < "/tmp/.fpflags.$app"
    "${CLEAN_ENV[@]}" flatpak override --user "${FLAGS[@]}" "$app"
    echo "applied overrides → $app"
  else
    echo "no overrides to apply for $app"
  fi
  rm -f "/tmp/.fpflags.$app"
}

export -f apply_app
for p in "$BASE"/*.perms; do
  [ -e "$p" ] || { echo "no *.perms found in $BASE"; exit 0; }
  app="$(basename "$p" .perms)"
  apply_app "$app"
done

** Flatpak desktop bridge — repo-first, idempotent (script)
#+BEGIN_SRC sh :tangle linux/.local/bin/flatpak-desktop-bridge :tangle-mode (identity #o755)
#!/usr/bin/env bash
set -euo pipefail

HOME_FC="$HOME/.flatpak/fontconfig/fonts.conf"   # ← stowed symlink target

# Validate (non-fatal if xmllint missing)
file -b --mime-encoding "$HOME_FC" >/dev/null 2>&1 || true
xmllint --noout "$HOME_FC" >/dev/null 2>&1 || true

# Allow caches (global or per-app later)
flatpak override --user --persist=".cache/fontconfig"

# Read-only bindings the sandboxes need
flatpak override --user \
  --filesystem="$HOME/.local/share/fonts:ro" \
  --filesystem="$HOME/.config/fontconfig:ro" \
  --filesystem="/run/host/usr/share/fonts:ro" \
  --filesystem="/run/host/usr/local/share/fonts:ro" \
  --filesystem="/run/host/etc/fonts:ro" \
  --filesystem="$HOME/.icons:ro" \
  --filesystem="$HOME/.local/share/icons:ro"

# Point fontconfig at the **HOME** path (stowed symlink)
flatpak override --user --env=FONTCONFIG_FILE="$HOME_FC"

# Cursor search path:
# - If XCURSOR_PATH is already set in the host env, extend it with /run/host
# - Otherwise, fall back to a sane default based on HOME + host icon dirs
if [ "${XCURSOR_PATH-}" != "" ]; then
  XCURSOR_PATH_VALUE="${XCURSOR_PATH}:/run/host/usr/local/share/icons:/run/host/usr/share/icons"
else
  XCURSOR_PATH_VALUE="$HOME/.icons:$HOME/.local/share/icons:/run/host/usr/local/share/icons:/run/host/usr/share/icons"
fi

flatpak override --user --env=XCURSOR_PATH="$XCURSOR_PATH_VALUE"

# Hygiene
flatpak override --user --unset-env=LD_PRELOAD

echo "✅ Flatpak bridge bound to HOME symlink:"
echo "   FONTCONFIG_FILE=$HOME_FC"
echo "   XCURSOR_PATH=$XCURSOR_PATH_VALUE"

if [ -L "$SBX_CFG" ]; then

  # real dir or file exists, back it up once
  ts=$(date +%Y%m%d-%H%M%S)
  echo "Backing up $SBX_CFG -> ${SBX_CFG}.bak.$ts"
  mv "$SBX_CFG" "${SBX_CFG}.bak.$ts"

ln -sT "$REPO_CFG" "$SBX_CFG"
echo "Linked: $SBX_CFG -> $REPO_CFG"

** Ungoogled Chromium Flatpak -> host (AppImage) profile bridge
Reuse the host Chromium profile at ~/.config/chromium (XDG) instead of ~/.var/app/...

#+BEGIN_SRC bash :tangle linux/.local/bin/flatpak-ungoogled-chromium-profile-bridge :tangle-mode (identity #o755) :shebang "#!/usr/bin/env bash"
#!/usr/bin/env bash
set -euo pipefail

APP_ID="io.github.ungoogled_software.ungoogled_chromium"

# Host profile root. This is the folder that contains "Default/", "Local State", etc.
HOST_CFG_BASE="${XDG_CONFIG_HOME:-$HOME/.config}"
HOST_PROFILE="${HOST_PROFILE_DIR:-$HOST_CFG_BASE/chromium}"

SBX_CFG_BASE="$HOME/.var/app/$APP_ID/config"
SBX_PROFILE="$SBX_CFG_BASE/chromium"

if [ ! -d "$HOST_PROFILE" ]; then
  echo "Host profile dir not found: $HOST_PROFILE" >&2
  echo "Tip: confirm AppImage profile root, it should contain 'Default/' and 'Local State'." >&2
  exit 1
fi

if [ ! -d "$HOST_PROFILE/Default" ] && [ ! -f "$HOST_PROFILE/Local State" ]; then
  echo "Host profile dir exists but doesn't look like a Chromium profile root:" >&2
  echo "  $HOST_PROFILE" >&2
  echo "Expected either $HOST_PROFILE/Default/ or $HOST_PROFILE/Local State" >&2
  exit 1
fi

echo "Granting $APP_ID access to host profile:"
echo "  $HOST_PROFILE"

# Reproducible permission override: host XDG config chromium
flatpak override --user --filesystem="xdg-config/chromium:create" "$APP_ID"

# If you want cache/state on host too, uncomment:
# flatpak override --user --filesystem="xdg-cache/chromium:create" "$APP_ID"
# flatpak override --user --filesystem="xdg-state/chromium:create" "$APP_ID"

mkdir -p "$SBX_CFG_BASE"

# Resolve absolute target to reduce weirdness
HOST_PROFILE_REAL="$(readlink -f "$HOST_PROFILE")"

if [ -L "$SBX_PROFILE" ]; then
  tgt="$(readlink -f "$SBX_PROFILE" || true)"
  if [ "$tgt" = "$HOST_PROFILE_REAL" ]; then
    echo "Already linked:"
    echo "  $SBX_PROFILE -> $HOST_PROFILE_REAL"
    exit 0
  fi
  echo "Different symlink exists, refusing to overwrite:" >&2
  ls -ld "$SBX_PROFILE" >&2
  exit 1
fi

if [ -e "$SBX_PROFILE" ]; then
  ts="$(date +%Y%m%d-%H%M%S)"
  backup="${SBX_PROFILE}.bak.$ts"
  echo "Existing sandbox profile found, backing up:"
  echo "  $SBX_PROFILE -> $backup"
  mv "$SBX_PROFILE" "$backup"
fi

echo "Linking sandbox profile to host profile:"
echo "  $SBX_PROFILE -> $HOST_PROFILE_REAL"
ln -sT "$HOST_PROFILE_REAL" "$SBX_PROFILE"

echo
echo "Done."
echo "Next:"
echo "  1) Make sure the AppImage is not running (Chromium profiles lock)."
echo "  2) Run: flatpak run $APP_ID"
echo "  3) Visit chrome://version and confirm Profile Path."
#+END_SRC


* Usage
1. Tangle and stow:
#+BEGIN_SRC sh
make tangle && cd ~/DotCortex && stow all linux debian x230

2. See plan:
#+BEGIN_SRC sh
make -f all/.mk/flatpak.mk flatpak-diff
#+END_SRC

3. Additive sync:
#+BEGIN_SRC sh
make -f all/.mk/flatpak.mk flatpak-sync
#+END_SRC

4. Enforce exact state, allowing removals:
#+BEGIN_SRC sh
ENFORCE=1 UNINSTALL=1 make -f all/.mk/flatpak.mk flatpak-apply
# In a TTY without Polkit:
FORCE_SUDO_SYSTEM=1 ENFORCE=1 UNINSTALL=1 make -f all/.mk/flatpak.mk flatpak-apply
#+END_SRC

5. Capture any ad hoc installs back into git:
#+BEGIN_SRC sh
make -f all/.mk/flatpak.mk flatpak-capture
git add linux/.flatpak/manifest/apps.ssv && git commit -m "Update Flatpak manifest"
#+END_SRC
