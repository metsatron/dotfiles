* Purpose
Reproducible Flatpak state from your dotfiles, Guix style. One manifest file is the source of truth. Each line declares origin, id, branch, and scope. Scope is either =user= or =system=. The apply step is additive by default; exact enforcement with removals is opt in.

* Single manifest
Columns: ORIGIN, ID, BRANCH, SCOPE. Comments and blanks allowed.

#+BEGIN_SRC text :tangle linux/.flatpak/manifest/apps.tsv
# origin     id                              branch   scope
flathub      com.github.tchx84.Flatseal      stable   user
# Examples seeded. Edit freely, or run capture to import live state.
# flathub    com.vscodium.codium             stable   user
# pied-origin com.mikeasoft.pied             master   system
#+END_SRC

* Remotes setup
Idempotent helper to ensure remotes exist before applying the manifest. Adjust the Pied URL if you use a custom one.

#+BEGIN_SRC bash :tangle all/.local/bin/flatpak-remotes-setup :shebang "#!/usr/bin/env bash"
set -euo pipefail

add_remote(){ name="$1"; url="$2"; flatpak remote-add --if-not-exists "$name" "$url"; }
add_remote_user(){ name="$1"; url="$2"; flatpak --user remote-add --if-not-exists "$name" "$url"; }

add_remote flathub https://dl.flathub.org/repo/flathub.flatpakrepo || true
add_remote_user flathub https://dl.flathub.org/repo/flathub.flatpakrepo || true

# Replace with your real Pied remote if different
if ! flatpak remote-list --columns=name | grep -qx pied-origin; then
  add_remote pied-origin https://dl.flathub.org/repo/flathub.flatpakrepo || true
fi
#+END_SRC

* Capture live state into the manifest
Merges your current user and system apps into =apps.tsv=. Keeps comments. Does not remove existing lines.

#+BEGIN_SRC bash :tangle all/.local/bin/flatpak-capture :shebang "#!/usr/bin/env bash"
set -euo pipefail
OUT="${HOME}/.dotfiles/linux/.flatpak/manifest/apps.tsv"
tmp="$(mktemp)"; trap 'rm -f "$tmp"' EXIT

normalize(){
  scope="$1"
  while read -r appid; do
    [ -n "$appid" ] || continue
    origin="$(flatpak info --show-origin "$appid" 2>/dev/null || true)"
    branch="$(flatpak info --show-branch "$appid" 2>/dev/null || true)"
    [ -n "$origin" ] && [ -n "$branch" ] || continue
    printf "%s\t%s\t%s\t%s\n" "$origin" "$appid" "$branch" "$scope"
  done
}

{
  [ -f "$OUT" ] && grep -E '^\s*#|^\s*$' "$OUT" || true
  {
    [ -f "$OUT" ] && grep -Ev '^\s*#|^\s*$' "$OUT" || true
    flatpak --user   list --app --columns=application | normalize user
    flatpak --system list --app --columns=application | normalize system
  } | awk 'NF>=4{print $1"\t"$2"\t"$3"\t"$4}' | sort -u
} > "$tmp"

mv "$tmp" "$OUT"
echo "Wrote $OUT"
#+END_SRC

* Diff plan
Shows what would be installed and what would be removed to match the manifest.

#+BEGIN_SRC bash :tangle all/.local/bin/flatpak-diff :shebang "#!/usr/bin/env bash"
set -euo pipefail
MAN="${HOME}/.dotfiles/linux/.flatpak/manifest/apps.tsv"

desired_ids(){ awk '$0!~/^#/ && NF>=4{print $2"\t"$4}' "$MAN" | sort -u; }
installed_ids(){
  { flatpak --user list --app --columns=application | awk 'NF{print $1"\tuser"}'
    flatpak --system list --app --columns=application | awk 'NF{print $1"\tsystem"}'; } \
  | sort -u
}

d="$(mktemp)"; i="$(mktemp)"; trap 'rm -f "$d" "$i"' EXIT
desired_ids >"$d"; installed_ids >"$i"

echo "[plan] to install:"
join -v1 -t $'\t' <(cut -f1 "$i") <(cut -f1 "$d") \
  | while read -r id; do grep -F "$id" "$d" | sed 's/^/  + /'; done || true

echo "[plan] to uninstall:"
join -v1 -t $'\t' <(cut -f1 "$d") <(cut -f1 "$i") \
  | while read -r id; do grep -F "$id" "$i" | sed 's/^/  - /'; done || true
#+END_SRC

* Apply manifest with safe elevation
Default is additive only. Exact enforcement with removals requires =ENFORCE=1 UNINSTALL=1=. System scoped actions first try through Polkit. If Polkit cannot authorize, the script retries the specific command with =sudo flatpak ...=. User scoped actions never use sudo.

#+BEGIN_SRC bash :tangle all/.local/bin/flatpak-apply :shebang "#!/usr/bin/env bash"
set -euo pipefail
MAN="${HOME}/.dotfiles/linux/.flatpak/manifest/apps.tsv"
ENFORCE="${ENFORCE:-0}"     # 1 to enforce exact match
UNINSTALL="${UNINSTALL:-0}" # 1 to allow removals when ENFORCE=1
# If you always want sudo for system scope in TTY, set FORCE_SUDO_SYSTEM=1
FORCE_SUDO_SYSTEM="${FORCE_SUDO_SYSTEM:-0}"

ensure_remotes(){ command -v flatpak-remotes-setup >/dev/null && flatpak-remotes-setup || true; }

run_fp(){
  # run a flatpak command, retry with sudo only if scope=system and auth fails
  scope="$1"; shift
  if [ "$scope" = "system" ] && [ "$FORCE_SUDO_SYSTEM" = "1" ]; then
    sudo flatpak "$@"
    return
  fi
  if flatpak "$@"; then
    return
  fi
  if [ "$scope" = "system" ]; then
    echo "Retrying with sudo: flatpak $*" >&2
    sudo flatpak "$@"
  fi
}

already_installed(){ id="$1" scope="$2"
  if [ "$scope" = user ]; then flatpak --user list --app --columns=application | grep -qx "$id"
  else flatpak --system list --app --columns=application | grep -qx "$id"; fi
}

apply_installs(){
  awk '$0!~/^#/ && NF>=4{print $1"\t"$2"\t"$3"\t"$4}' "$MAN" \
  | while IFS=$'\t' read -r origin id branch scope; do
      [ -n "$origin" ] && [ -n "$id" ] && [ -n "$branch" ] && [ -n "$scope" ] || continue
      already_installed "$id" "$scope" && continue
      args=(install -y)
      [ "$scope" = user ] && args+=(--user) || args+=(--system)
      run_fp "$scope" "${args[@]}" "$origin" "$id" "$branch"
    done
}

compute_extras(){
  desired="$(mktemp)"; installed="$(mktemp)"; trap 'rm -f "$desired" "$installed"' RETURN
  awk '$0!~/^#/ && NF>=4{print $2"\t"$4}' "$MAN" | sort -u >"$desired"
  { flatpak --user list --app --columns=application | awk 'NF{print $1"\tuser"}'
    flatpak --system list --app --columns=application | awk 'NF{print $1"\tsystem"}'; } \
  | sort -u >"$installed"
  join -v2 -t $'\t' <(cut -f1 "$desired") "$installed"
}

apply_uninstalls(){
  [ "$ENFORCE" = 1 ] && [ "$UNINSTALL" = 1 ] || return 0
  compute_extras | while read -r id scope; do
    if [ "$scope" = user ]; then
      flatpak --user uninstall -y "$id" || true
    else
      run_fp system uninstall -y "$id" || true
    fi
  done
}

ensure_remotes
apply_installs
apply_uninstalls
echo "Done."
#+END_SRC

* Make targets
Include directly with =make -f all/.mk/flatpak.mk <target>= or add to your main Makefile with =-include all/.mk/flatpak.mk=.

#+BEGIN_SRC makefile :tangle all/.mk/flatpak.mk :padline no :comments none
.PHONY: flatpak-remotes flatpak-capture flatpak-diff flatpak-sync flatpak-apply

flatpak-remotes:
| @chmod +x $(HOME)/.local/bin/flatpak-remotes-setup 2>/dev/null || true
| $(HOME)/.local/bin/flatpak-remotes-setup

flatpak-capture:
| @chmod +x $(HOME)/.local/bin/flatpak-capture 2>/dev/null || true
| $(HOME)/.local/bin/flatpak-capture

flatpak-diff:
| @chmod +x $(HOME)/.local/bin/flatpak-diff 2>/dev/null || true
| $(HOME)/.local/bin/flatpak-diff

# Additive only by default
flatpak-sync: flatpak-remotes
| @chmod +x $(HOME)/.local/bin/flatpak-apply 2>/dev/null || true
| ENFORCE=0 UNINSTALL=0 $(HOME)/.local/bin/flatpak-apply

# Enforce exact match. Set both flags to permit removals.
# Examples:
#   ENFORCE=1 UNINSTALL=1 make -f all/.mk/flatpak.mk flatpak-apply
#   FORCE_SUDO_SYSTEM=1 ENFORCE=1 UNINSTALL=1 make -f all/.mk/flatpak.mk flatpak-apply   # TTY only
flatpak-apply: flatpak-remotes
| @chmod +x $(HOME)/.local/bin/flatpak-apply 2>/dev/null || true
| ENFORCE=${ENFORCE} UNINSTALL=${UNINSTALL} FORCE_SUDO_SYSTEM=${FORCE_SUDO_SYSTEM} $(HOME)/.local/bin/flatpak-apply
#+END_SRC

* Usage
1. Tangle and stow:
#+BEGIN_SRC sh
make tangle && cd ~/.dotfiles && stow all linux debian think
#+END_SRC

2. See plan:
#+BEGIN_SRC sh
make -f all/.mk/flatpak.mk flatpak-diff
#+END_SRC

3. Additive sync:
#+BEGIN_SRC sh
make -f all/.mk/flatpak.mk flatpak-sync
#+END_SRC

4. Enforce exact state, allowing removals:
#+BEGIN_SRC sh
ENFORCE=1 UNINSTALL=1 make -f all/.mk/flatpak.mk flatpak-apply
# In a TTY without Polkit:
FORCE_SUDO_SYSTEM=1 ENFORCE=1 UNINSTALL=1 make -f all/.mk/flatpak.mk flatpak-apply
#+END_SRC

5. Capture any ad hoc installs back into git:
#+BEGIN_SRC sh
make -f all/.mk/flatpak.mk flatpak-capture
git add linux/.flatpak/manifest/apps.tsv && git commit -m "Update Flatpak manifest"
#+END_SRC
