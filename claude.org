* Claude Code plugins manifest
Columns: PKG MARKETPLACE_NAME MARKETPLACE_REPO

The MARKETPLACE_NAME is the name Claude Code assigns when you run
=claude plugin marketplace add <MARKETPLACE_REPO>=.
Check it with: =claude plugin marketplace list --json=.

* Apply plugins
#+BEGIN_SRC bash :tangle all/.local/bin/claude-plugins-apply :shebang "#!/usr/bin/env bash"
# ~/DotCortex/all/.local/bin/claude-plugins-apply
set -euo pipefail

MAN="$HOME/DotCortex/all/.claude/manifest/plugins.ssv"

echo "== claude plugins apply =="

declare -A seen_marketplaces

while IFS=' ' read -r pkg marketplace_name marketplace_repo; do
  [[ "$pkg" =~ ^[[:space:]]*# ]] && continue
  [ -z "${pkg:-}" ] && continue

  if [ -z "${seen_marketplaces[$marketplace_name]:-}" ]; then
    echo "marketplace: $marketplace_name ($marketplace_repo)"
    claude plugin marketplace add "$marketplace_repo" 2>/dev/null || true
    seen_marketplaces["$marketplace_name"]=1
  fi

  installed_json="$HOME/.claude/plugins/installed_plugins.json"
  key="${pkg}@${marketplace_name}"

  if [ -f "$installed_json" ] && python3 -c "
import json, sys
data = json.load(open('$installed_json'))
sys.exit(0 if '$key' in data.get('plugins', {}) else 1)
" 2>/dev/null; then
    echo "ok: $key"
  else
    echo "install: $key"
    claude plugin install "${pkg}@${marketplace_name}"
  fi

done < <(awk '!/^[[:space:]]*#/ && NF' "$MAN")

echo "Done. Restart Claude Code or run /reload-plugins to activate."
#+END_SRC

* Health check
#+BEGIN_SRC bash :tangle all/.local/bin/claude-plugins-health :shebang "#!/usr/bin/env bash"
# ~/DotCortex/all/.local/bin/claude-plugins-health
set -euo pipefail

installed_json="$HOME/.claude/plugins/installed_plugins.json"
MAN="$HOME/DotCortex/all/.claude/manifest/plugins.ssv"

echo "== claude plugins health =="

if [ ! -f "$installed_json" ]; then
  echo "ERROR: $installed_json not found"
  exit 1
fi

echo ""
echo "Installed:"
python3 -c "
import json
data = json.load(open('$installed_json'))
plugins = data.get('plugins', {})
if not plugins:
    print('  (none)')
for key, installs in plugins.items():
    for inst in installs:
        print(f'  {key}  v{inst[\"version\"]}')
"

echo ""
echo "Manifest wants:"
awk '!/^[[:space:]]*#/ && NF { printf "  %s@%s\n", $1, $2 }' "$MAN"
#+END_SRC

* Hooks
Session lifecycle hooks. Scripts are stowed to =~/.local/bin/= and referenced
by name in =all/.claude/settings.json= (global settings).

The stop counter uses =$PWD/.claude/stop_count= so each workspace tracks
independently.  The session-start and pre-compact hooks derive the workspace
name from =basename $PWD= and use it to locate the matching Obsidian TODO page.

** Session start
#+BEGIN_SRC bash :tangle all/.local/bin/claude-hook-session-start :shebang "#!/usr/bin/env bash"
# ~/DotCortex/all/.local/bin/claude-hook-session-start
WORKSPACE=$(basename "$PWD")
echo "--- $WORKSPACE TODO ---"
flatpak run md.obsidian.Obsidian vault=HelmCortex read path="LOGS/TODO/$WORKSPACE.md" 2>/dev/null \
  || echo "(no $WORKSPACE TODO found)"
echo
HOST=$(hostname)
echo "--- Machine TODO ($HOST) ---"
flatpak run md.obsidian.Obsidian vault=HelmCortex read path="LOGS/TODO/Machines/$HOST.md" 2>/dev/null \
  || echo "(no machine TODO found)"
#+END_SRC

** Stop (reflect nudge)
#+BEGIN_SRC bash :tangle all/.local/bin/claude-hook-stop :shebang "#!/usr/bin/env bash"
# ~/DotCortex/all/.local/bin/claude-hook-stop
COUNTFILE="${PWD}/.claude/stop_count"
COUNT=$(cat "$COUNTFILE" 2>/dev/null || echo 0)
COUNT=$((COUNT + 1))
if [ "$COUNT" -ge 15 ]; then
  printf '\u27f3 15 responses logged \u2014 consider /reflect if this session produced durable discoveries.\n'
  COUNT=0
fi
echo "$COUNT" > "$COUNTFILE"
#+END_SRC

** Pre-compact (memory dump)
#+BEGIN_SRC bash :tangle all/.local/bin/claude-hook-pre-compact :shebang "#!/usr/bin/env bash"
# ~/DotCortex/all/.local/bin/claude-hook-pre-compact
WORKSPACE=$(basename "$PWD")
MEMFILE="$PWD/.claude/MEMORY.md"
CONTENT=$(tail -n +2 "$MEMFILE" 2>/dev/null | sed '/^$/d' | sed '/^## Session Facts$/d' | sed '/^## User Model.*$/d')
if [ -n "$CONTENT" ]; then
  DUMP=$(printf '## Memory Dump %s\n\n%s' "$(date +%Y-%m-%d)" "$CONTENT")
  flatpak run md.obsidian.Obsidian vault=HelmCortex append path="LOGS/TODO/$WORKSPACE.md" content="$DUMP" 2>/dev/null
  printf '# Session Memory (auto-updated by Opus, max 2000 chars)\n\n## Session Facts\n\n\n## User Model (max 1500 chars for this section)\n\n' > "$MEMFILE"
  echo "Memory saved to $WORKSPACE TODO via Obsidian CLI and MEMORY.md truncated."
else
  echo 'MEMORY.md empty, nothing to save.'
fi
#+END_SRC
