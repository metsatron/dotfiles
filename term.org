# See this wiki page for more info:

# info suports from ${c1} to ${c6} and do not support $(color n) syntax, use 'ascii_colors' color order

# This config file make use of some glyphs that are not available in the

# --- terminal font fallback (env → env → fc-match) ---
if [[ -z ${term_font-} ]]; then
  if [[ -n ${TERM_FONT-} ]]; then
    term_font="$TERM_FONT"
  elif [[ -n ${WEZTERM_FONT-} ]]; then
    term_font="$WEZTERM_FONT"
  elif command -v fc-match >/dev/null 2>&1; then
    term_font="$(fc-match -f '%{family}\n' 2>/dev/null | head -n1)"
  fi
fi

# Decorator for colors

** Ranger scope.sh
#+BEGIN_SRC bash :tangle all/.config/ranger/scope.sh :shebang "#!/usr/bin/env bash" :tangle-mode (identity #o755) :mkdirp yes
set -o noclobber -o nounset -o pipefail

file="$1"
w="${2:-$COLUMNS}"
h="${3:-$LINES}"
x="${4:-0}"
y="${5:-0}"
tty="/dev/tty"

mime="$(file --mime-type -Lb "$file")"

case "$mime" in
  image/*)
    # 1) Kitty graphics with exact placement (works in WezTerm), write to the TTY
    if command -v kitty >/dev/null 2>&1 && [ -z "${TMUX:-}" ]; then
      printf '\e7' >"$tty"                                 # save cursor
      printf '\e[%d;%dH' "$((y+1))" "$((x+1))" >"$tty"     # move to preview pane origin
      kitty +kitten icat \
        --place "${w}x${h}@${x}x${y}" \
        --transfer-mode=memory \
        --stdin=no \
        "$file" >"$tty" 2>/dev/null || true
      printf '\e8' >"$tty"                                 # restore cursor
      exit 6
    fi

    # 2) WezTerm's imgcat (width in cells), write to the TTY
    if command -v wezterm-imgcat >/dev/null 2>&1; then
      printf '\e7' >"$tty"
      printf '\e[%d;%dH' "$((y+1))" "$((x+1))" >"$tty"
      WEZTERM_IMGCAT_WIDTH="${w}cell" wezterm-imgcat "$file" >"$tty" 2>/dev/null || true
      printf '\e8' >"$tty"
      exit 6
    fi

    # 3) High-quality text fallback (sixel), goes to stdout inside the pane
    if command -v chafa >/dev/null 2>&1; then
      chafa --format=sixel --size "${w}x${h}" "$file"
      exit 0
    fi

    # 4) Simple text fallback
    if command -v viu >/dev/null 2>&1; then
      viu -w "$w" "$file"
      exit 0
    fi

    echo "Image preview available but no graphics backend found."
    exit 0
    ;;
esac

exit 0
#+END_SRC

** TPM bootstrap script
#+BEGIN_SRC bash :tangle all/.local/bin/tmux-bootstrap :shebang "#!/usr/bin/env bash" :tangle-mode (identity #o755)
#!/usr/bin/env bash
set -euo pipefail

TPM_DIR="${HOME}/.config/tmux/plugins/tpm"
RESURRECT_DIR="${HOME}/.local/share/tmux/resurrect"

echo "== tmux bootstrap =="

mkdir -p "$RESURRECT_DIR"

if [ ! -d "$TPM_DIR" ]; then
  echo ">> Installing TPM..."
  git clone https://github.com/tmux-plugins/tpm "$TPM_DIR"
  echo ">> TPM installed at $TPM_DIR"
else
  echo ">> TPM already present at $TPM_DIR"
fi

if [ -x "$TPM_DIR/bin/install_plugins" ]; then
  echo ">> Installing tmux plugins..."
  "$TPM_DIR/bin/install_plugins"
  echo ">> Plugins installed."
else
  echo ">> TPM install script not found. Start tmux and press prefix + I."
fi

echo "Done."
#+END_SRC

** Tmux health
#+BEGIN_SRC bash :tangle all/.local/bin/tmux-health :shebang "#!/usr/bin/env bash" :tangle-mode (identity #o755)
#!/usr/bin/env bash
set -euo pipefail

echo "== tmux health =="

if command -v tmux >/dev/null 2>&1; then
  echo "[tmux] $(tmux -V) at $(command -v tmux)"
else
  echo "[tmux] NOT FOUND"
fi

TPM_DIR="${HOME}/.config/tmux/plugins/tpm"
if [ -d "$TPM_DIR" ]; then
  echo "[tpm] installed at $TPM_DIR"
else
  echo "[tpm] NOT INSTALLED -- run tmux-bootstrap"
fi

for plugin in tmux-resurrect tmux-continuum tmux-yank tmux-sensible; do
  dir="${HOME}/.config/tmux/plugins/$plugin"
  if [ -d "$dir" ]; then
    echo "[$plugin] installed"
  else
    echo "[$plugin] NOT INSTALLED"
  fi
done

RESURRECT_DIR="${HOME}/.local/share/tmux/resurrect"
if [ -d "$RESURRECT_DIR" ]; then
  count=$(find "$RESURRECT_DIR" -name "*.txt" -type f 2>/dev/null | wc -l)
  latest=$(ls -1t "$RESURRECT_DIR"/*.txt 2>/dev/null | head -1 || true)
  echo "[resurrect] $count saves in $RESURRECT_DIR"
  if [ -n "$latest" ]; then
    echo "[resurrect] latest: $(basename "$latest")"
  fi
else
  echo "[resurrect] save dir not found"
fi

if tmux list-sessions 2>/dev/null; then
  echo "[server] tmux server running"
else
  echo "[server] no tmux server"
fi
#+END_SRC

** Tmux session launcher
Interactive entrypoint for tmux sessions.
Uses the fleet =fzf= picker by default and falls back to the legacy DotCortex
=Cortex= layout when =fzf= is unavailable or when =--legacy= is passed.
#+BEGIN_SRC bash :tangle all/.local/bin/tmux-session :shebang "#!/usr/bin/env bash" :tangle-mode (identity #o755)
#!/usr/bin/env bash
# tmux-session — primary tmux entrypoint
# Default: open the fleet session picker.
# Legacy mode: recreate the old fixed Cortex layout.

set -euo pipefail

if [ "${1:-}" = "--legacy" ]; then
  shift
elif command -v fzf >/dev/null 2>&1; then
  canonical_host() {
    local raw="${1:-}"
    local lower="${raw,,}"

    if [[ "$lower" == *x230* ]]; then
      printf '%s\n' "x230"
    elif [[ "$lower" == *t480s* ]]; then
      printf '%s\n' "t480s"
    else
      printf '%s\n' "${raw%%.*}"
    fi
  }

  trim() {
    local value="${1:-}"
    value="${value#"${value%%[![:space:]]*}"}"
    value="${value%"${value##*[![:space:]]}"}"
    printf '%s\n' "$value"
  }

  quote_sh() {
    printf "'%s'" "$(printf '%s' "$1" | sed "s/'/'\\\\''/g")"
  }

  attach_local() {
    local session_name="$1"
    if [ -n "${TMUX:-}" ]; then
      tmux new-window "TMUX='' exec tmux attach-session -t $(quote_sh "$session_name")"
      exit 0
    fi
    exec tmux attach-session -t "$session_name"
  }

  create_local() {
    local session_name="$1"
    if [ -n "${TMUX:-}" ]; then
      tmux has-session -t "$session_name" 2>/dev/null || tmux new-session -d -s "$session_name"
      tmux new-window "TMUX='' exec tmux attach-session -t $(quote_sh "$session_name")"
      exit 0
    fi
    exec tmux new-session -A -s "$session_name"
  }

  attach_remote() {
    local host="$1"
    local session_name="$2"
    ssh -t "$host" "tmux attach-session -t $(quote_sh "$session_name")"
    exit $?
  }

  create_remote() {
    local host="$1"
    local session_name="$2"
    ssh -t "$host" "tmux new-session -A -s $(quote_sh "$session_name")"
    exit $?
  }

  prompt_session_name() {
    local host="$1"
    local query="${2:-}"
    local session_name

    session_name="$(trim "$query")"
    if [ -n "$session_name" ]; then
      printf '%s\n' "$session_name"
      return 0
    fi

    printf 'New tmux session on %s: ' "$host" >&2
    read -r session_name
    session_name="$(trim "$session_name")"
    if [ -z "$session_name" ]; then
      echo "tmux-session: session name is required" >&2
      exit 1
    fi
    if [[ "$session_name" == *:* ]]; then
      echo "tmux-session: session names cannot contain ':'" >&2
      exit 1
    fi
    printf '%s\n' "$session_name"
  }

  list_local_sessions() {
    tmux list-sessions -F '#{session_name}|#{session_windows}|#{?session_attached,attached,detached}' 2>/dev/null || true
  }

  list_remote_sessions() {
    local host="$1"
    ssh "${SSH_OPTS[@]}" "$host" "tmux list-sessions -F '#{session_name}|#{session_windows}|#{?session_attached,attached,detached}' 2>/dev/null" 2>/dev/null || true
  }

  build_menu() {
    local host rows scope name windows attached

    for host in "${HOSTS[@]}"; do
      if [ "$host" = "$LOCAL_HOST" ]; then
        scope="local"
        rows="$(list_local_sessions)"
      else
        scope="ssh"
        rows="$(list_remote_sessions "$host")"
      fi

      if [ -n "$rows" ]; then
        while IFS='|' read -r name windows attached; do
          [ -n "${name:-}" ] || continue
          printf 'attach\t%s\t%s\tresume  %-5s %-10s %-24s %2sw %s\n' \
            "$host" "$name" "$scope" "$host" "$name" "$windows" "$attached"
        done <<< "$rows"
      fi

      printf 'create\t%s\t\tcreate  %-5s %-10s %s\n' \
        "$host" "$scope" "$host" "new session"
    done
  }

  LOCAL_HOST="$(canonical_host "$(hostname -s 2>/dev/null || hostname)")"
  declare -a HOSTS=()
  declare -A SEEN_HOSTS=()
  SSH_OPTS=(
    -o BatchMode=yes
    -o ConnectTimeout=2
    -o ConnectionAttempts=1
    -o RequestTTY=no
  )

  add_host() {
    local host
    host="$(canonical_host "$1")"
    [ -n "$host" ] || return 0
    [ -n "${SEEN_HOSTS[$host]:-}" ] && return 0
    SEEN_HOSTS["$host"]=1
    HOSTS+=("$host")
  }

  if [ -n "${TMUX_SESSION_HOSTS:-}" ]; then
    for host in ${TMUX_SESSION_HOSTS}; do
      add_host "$host"
    done
  else
    add_host "$LOCAL_HOST"
    add_host x230
    add_host t480s
  fi

  mapfile -t PICKED < <(
    build_menu | fzf \
      --height=70% \
      --layout=reverse \
      --border \
      --prompt='tmux> ' \
      --header='Enter: attach/create, type to filter or name a new session' \
      --delimiter=$'\t' \
      --with-nth=4 \
      --print-query
  )

  QUERY="${PICKED[0]:-}"
  SELECTION="${PICKED[1]:-}"

  [ -n "$SELECTION" ] || exit 0

  IFS=$'\t' read -r ACTION HOST SESSION_NAME _ <<< "$SELECTION"

  if [ "$ACTION" = "create" ]; then
    SESSION_NAME="$(prompt_session_name "$HOST" "$QUERY")"
  fi

  if [ "$HOST" = "$LOCAL_HOST" ]; then
    if [ "$ACTION" = "attach" ]; then
      attach_local "$SESSION_NAME"
    fi
    create_local "$SESSION_NAME"
  fi

  if [ "$ACTION" = "attach" ]; then
    attach_remote "$HOST" "$SESSION_NAME"
  fi

  create_remote "$HOST" "$SESSION_NAME"
fi

SESSION="Cortex"
HOST="$(hostname)"

# Attach if session exists
if tmux has-session -t "$SESSION" 2>/dev/null; then
  exec tmux attach-session -t "$SESSION"
fi

# --- T480s layout ---
if [[ "$HOST" == *T480s* || "$HOST" == *t480s* ]]; then
  tmux new-session -d -s "$SESSION" -n "t480s" -c "$HOME/DotCortex"
  tmux new-window  -t "$SESSION" -n "x230"
  # X230 window: ssh into x230 (will reconnect if host available)
  tmux send-keys -t "$SESSION:x230" "ssh x230 -t 'tmux attach -t Cortex 2>/dev/null || tmux new -s Cortex'" Enter

# --- X230 layout ---
elif [[ "$HOST" == *X230* || "$HOST" == *x230* ]]; then
  tmux new-session -d -s "$SESSION" -n "x230" -c "$HOME/DotCortex"
  tmux new-window  -t "$SESSION" -n "t480s"
  tmux send-keys -t "$SESSION:t480s" "ssh t480s -t 'tmux attach -t Cortex 2>/dev/null || tmux new -s Cortex'" Enter

# --- Generic fallback ---
else
  tmux new-session -d -s "$SESSION" -n "main" -c "$HOME/DotCortex"
fi

# Select the local window and attach
tmux select-window -t "$SESSION:1"
exec tmux attach-session -t "$SESSION"
#+END_SRC

** Tmux resurrect hooks for Claude Code / OpenCode
Resurrect saves and restores processes listed in =@resurrect-processes=.
Claude Code (=claude=) and OpenCode (=opencode=) are already in the list.
For Claude, resurrection works via =claude --resume <id>= which is what
the restored shell history will contain. For full automatic resume, we add
a resurrect hook that detects claude sessions and offers to resume them.
#+BEGIN_SRC bash :tangle all/.config/tmux/resurrect-post-restore.sh :shebang "#!/usr/bin/env bash" :tangle-mode (identity #o755)
#!/usr/bin/env bash
# Post-restore hook for tmux-resurrect
# Checks for recent claude/opencode sessions and notifies user

CLAUDE_SESSIONS="$HOME/.claude/projects"
if [ -d "$CLAUDE_SESSIONS" ]; then
  latest=$(find "$CLAUDE_SESSIONS" -name "*.jsonl" -mmin -720 -type f 2>/dev/null | head -1)
  if [ -n "$latest" ]; then
    tmux display-message "Claude Code sessions available. Use 'claude --resume' to restore."
  fi
fi
#+END_SRC
