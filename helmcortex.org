#+BEGIN_SRC bash :tangle all/.local/bin/helmcortex-anaconda :shebang "#!/usr/bin/env bash" :tangle-mode (identity #o755)
set -euo pipefail

# keep non-Guix runtime for this process
unset GUIX_PROFILE GUIX_LOCPATH GUIX_GTK3_PATH LD_LIBRARY_PATH

# activate env for this process
# use source so we do not spawn a subshell
source "$HOME/HelmCortex/FORGE/bin/helmcortex-anaconda-activate"

if [[ "${1-}" == "-c" ]]; then
  shift
  exec python -c "$@"
elif [[ "${1-}" == *.py ]]; then
  script="$1"; shift
  exec python "$script" "$@"
else
  exec python "$@"
fi
#+END_SRC

* HelmCortex Anaconda Activate
#+BEGIN_SRC bash :tangle all/HelmCortex/FORGE/bin/helmcortex-anaconda-activate :shebang "#!/usr/bin/env bash" :tangle-mode (identity #o755)
#!/usr/bin/env bash

# detect if sourced
sourced=0
if [ -n "${ZSH_VERSION:-}" ]; then
  case $ZSH_EVAL_CONTEXT in *:file) sourced=1;; esac
elif [ -n "${BASH_VERSION:-}" ]; then
  [[ "${BASH_SOURCE[0]}" != "$0" ]] && sourced=1 || true
else
  (return 0 2>/dev/null) && sourced=1 || true
fi

if [ "$sourced" -eq 0 ]; then
  set -euo pipefail
fi

export HELMCORTEX_CONDA_ROOT="$HOME/HelmCortex/CORE/anaconda3"
export PYTHONNOUSERSITE=1

# Guix sanitization
unset GUIX_PROFILE GUIX_LOCPATH GUIX_GTK3_PATH LD_LIBRARY_PATH

# prepend conda bin
export PATH="$HELMCORTEX_CONDA_ROOT/bin:$PATH"

# choose proper shell hook
shell_name="bash"
if [ -n "${ZSH_VERSION:-}" ]; then
  shell_name="zsh"
elif [ -n "${BASH_VERSION:-}" ]; then
  shell_name="bash"
elif [ -n "${SHELL:-}" ] && echo "$SHELL" | grep -q 'zsh'; then
  shell_name="zsh"
fi

# load conda into this shell
eval "$("$HELMCORTEX_CONDA_ROOT/bin/conda" "shell.$shell_name" hook)"

# activate project env (correct name)
conda activate helmcortex

# executed vs sourced behavior
if [ "$sourced" -eq 0 ]; then
  if [ "$#" -gt 0 ]; then
    exec "$@"
  else
    exec "${SHELL:-/bin/bash}" -i
  fi
else
  return 0 2>/dev/null || true
fi
#+END_SRC

* HelmCortex Conda Bootstrap
#+BEGIN_SRC bash :tangle all/HelmCortex/FORGE/bin/helmcortex-conda-ensure :shebang "#!/usr/bin/env bash" :tangle-mode (identity #o755)
set -euo pipefail

export HELMCORTEX_CONDA_ROOT="$HOME/HelmCortex/CORE/anaconda3"
export CONDARC="$HOME/HelmCortex/CORE/conda/condarc.yaml"

# ensure conda exists
if [ ! -x "$HELMCORTEX_CONDA_ROOT/bin/conda" ]; then
  echo "Conda not found at $HELMCORTEX_CONDA_ROOT" >&2
  exit 1
fi

# create env if missing
if ! "$HELMCORTEX_CONDA_ROOT/bin/conda" env list | grep -qE '/envs/helmcortex$'; then
  "$HELMCORTEX_CONDA_ROOT/bin/conda" create -y -n helmcortex python=3.10
fi

# install core packages with conda-forge available via CONDARC
"$HELMCORTEX_CONDA_ROOT/bin/conda" install -y -n helmcortex \
  pandas beautifulsoup4 markdown emoji typer requests lxml pytz

# export env spec for reproducibility
mkdir -p "$HOME/HelmCortex/ROOTS"
"$HELMCORTEX_CONDA_ROOT/bin/conda" env export -n helmcortex --from-history > "$HOME/HelmCortex/ROOTS/helmcortex-env.yml"
#+END_SRC
