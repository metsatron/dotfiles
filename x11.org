* X session init (Devuan/T480s)
#+BEGIN_SRC sh :tangle devuan/.xsessionrc :tangle-mode (identity #o644)
# Keep the distro gtk3-nocsd hook from exporting session-wide LD_PRELOAD.
# That leaks into Guix and Electron binaries and causes loader warnings.
export GTK3_NOCSD_IGNORE=1
export GTK_CSD=0

# Pull in per-session environment snippets before the desktop starts.
if [ -d "$HOME/.xsessionrc.d" ]; then
  for xsessionrc_snippet in "$HOME"/.xsessionrc.d/*.sh; do
    [ -r "$xsessionrc_snippet" ] || continue
    set +u
    . "$xsessionrc_snippet"
    set +u
  done
  unset xsessionrc_snippet
fi

# AppMenu must live in the real X session so GUI apps launched from menus
# can see both the module request and the Guix profile environment.
case ":${GTK_MODULES:-}:" in
  *:appmenu-gtk-module:*) ;;
  *) export GTK_MODULES="${GTK_MODULES:+$GTK_MODULES:}appmenu-gtk-module" ;;
esac
export UBUNTU_MENUPROXY="${UBUNTU_MENUPROXY:-1}"
export QT_QPA_PLATFORMTHEME="${QT_QPA_PLATFORMTHEME:-gtk2}"
#+END_SRC
