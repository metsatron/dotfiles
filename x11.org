* X session init (Devuan/T480s)
#+BEGIN_SRC sh :tangle devuan/.xsessionrc :tangle-mode (identity #o644)
# Keep the distro gtk3-nocsd hook from exporting session-wide LD_PRELOAD.
# That leaks into Guix and Electron binaries and causes loader warnings.
export GTK3_NOCSD_IGNORE=1
export GTK_CSD=0

# Pull in per-session environment snippets before the desktop starts.
if [ -d "$HOME/.xsessionrc.d" ]; then
  for xsessionrc_snippet in "$HOME"/.xsessionrc.d/*.sh; do
    [ -r "$xsessionrc_snippet" ] || continue
    set +u
    . "$xsessionrc_snippet"
    set +u
  done
  unset xsessionrc_snippet
fi

# AppMenu must live in the real X session so GUI apps launched from menus
# can see both the module request and the Guix profile environment.
case ":${GTK_MODULES:-}:" in
  *:appmenu-gtk-module:*) ;;
  *) export GTK_MODULES="${GTK_MODULES:+$GTK_MODULES:}appmenu-gtk-module" ;;
esac
export UBUNTU_MENUPROXY="${UBUNTU_MENUPROXY:-1}"
export QT_QPA_PLATFORMTHEME="${QT_QPA_PLATFORMTHEME:-gtk2}"
#+END_SRC

* AppMenu registrar bootstrap
#+BEGIN_SRC sh :tangle linux/.xsessionrc.d/30-appmenu-registrar.sh :shebang "#!/usr/bin/env sh"
# Start the registrar inside the real X session so exported menus and the
# panel plugin meet on the same session bus.
APPMENU_REGISTRAR=
if command -v appmenu-registrar >/dev/null 2>&1; then
  APPMENU_REGISTRAR="$(command -v appmenu-registrar)"
elif [ -x /usr/libexec/vala-panel/appmenu-registrar ]; then
  APPMENU_REGISTRAR=/usr/libexec/vala-panel/appmenu-registrar
fi

if [ -n "${APPMENU_REGISTRAR}" ]; then
  if ! pgrep -af '[a]ppmenu-registrar' >/dev/null 2>&1; then
    "$APPMENU_REGISTRAR" -r >/dev/null 2>&1 &
  fi
fi

unset APPMENU_REGISTRAR
#+END_SRC

* XFCE appmenu xsettings helper
#+BEGIN_SRC bash :tangle linux/.local/bin/xfce-appmenu-configure :shebang "#!/usr/bin/env bash"
set -euo pipefail

if ! command -v xfconf-query >/dev/null 2>&1; then
  echo "xfconf-query not found; XFCE xsettings cannot be configured" >&2
  exit 1
fi

set_bool() {
  local path="$1" value="$2"
  if xfconf-query -c xsettings -p "$path" >/dev/null 2>&1; then
    xfconf-query -c xsettings -p "$path" -t bool -s "$value"
  else
    xfconf-query -c xsettings -p "$path" -n -t bool -s "$value"
  fi
}

set_string() {
  local path="$1" value="$2"
  if xfconf-query -c xsettings -p "$path" >/dev/null 2>&1; then
    xfconf-query -c xsettings -p "$path" -t string -s "$value"
  else
    xfconf-query -c xsettings -p "$path" -n -t string -s "$value"
  fi
}

resolve_registrar() {
  if command -v appmenu-registrar >/dev/null 2>&1; then
    command -v appmenu-registrar
  elif [ -x /usr/libexec/vala-panel/appmenu-registrar ]; then
    printf '%s\n' /usr/libexec/vala-panel/appmenu-registrar
  else
    return 1
  fi
}

set_bool /Gtk/ShellShowsMenubar true
set_bool /Gtk/ShellShowsAppmenu true

gtk_modules="$(xfconf-query -c xsettings -p /Gtk/Modules 2>/dev/null || true)"
case ":${gtk_modules}:" in
  *:appmenu-gtk-module:*) ;;
  *) gtk_modules="${gtk_modules:+$gtk_modules:}appmenu-gtk-module" ;;
esac
set_string /Gtk/Modules "$gtk_modules"

registrar_bin="$(resolve_registrar || true)"
if [[ -n "$registrar_bin" ]]; then
  if ! pgrep -af '[a]ppmenu-registrar' >/dev/null 2>&1; then
    "$registrar_bin" -r >/dev/null 2>&1 &
  fi
fi

printf 'Gtk/ShellShowsMenubar=%s\n' "$(xfconf-query -c xsettings -p /Gtk/ShellShowsMenubar 2>/dev/null || echo unset)"
printf 'Gtk/ShellShowsAppmenu=%s\n' "$(xfconf-query -c xsettings -p /Gtk/ShellShowsAppmenu 2>/dev/null || echo unset)"
printf 'Gtk/Modules=%s\n' "$(xfconf-query -c xsettings -p /Gtk/Modules 2>/dev/null || echo unset)"
pgrep -af '[a]ppmenu-registrar' || true
#+END_SRC
