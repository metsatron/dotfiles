;;; ========== Color utils ==========
(defun lc--hex->rgb (hex)
  (let ((h (replace-regexp-in-string "#" "" hex)))
    (list (string-to-number (substring h 0 2) 16)
          (string-to-number (substring h 2 4) 16)
          (string-to-number (substring h 4 6) 16))))
(defun lc--rgb->hex (r g b)
  (format "#%02x%02x%02x" (max 0 (min 255 r)) (max 0 (min 255 g)) (max 0 (min 255 b))))
(defun lc--mix (h1 h2 a)
  (cl-destructuring-bind (r1 g1 b1) (lc--hex->rgb h1)
    (cl-destructuring-bind (r2 g2 b2) (lc--hex->rgb h2)
      (lc--rgb->hex
       (round (+ (* (- 1 a) r1) (* a r2)))
       (round (+ (* (- 1 a) g1) (* a g2)))
       (round (+ (* (- 1 a) b1) (* a b2)))))))

(defun lc/b (theme key)
  (let* ((cell (assoc theme lc/base16))
         (plist (and cell (cdr cell))))
    (plist-get plist key)))

;; Widget accessor — assoc-based + Base16 fallback
(defun lc/w (theme key)
  (let* ((cell (assoc theme lc/widget))
         (w (and cell (cdr cell))))
    (if w (plist-get w key)
      (pcase key
        (:white   (lc/b theme :base07))
        (:lighter (lc/b theme :base02))
        (:base    (lc/b theme :base01))
        (:darker  (lc/b theme :base00))
        (:black   "#000000")
        (:accent  (lc/b theme :base0C))
        (:selection-bg (lc/b theme :base0C))
        (:selection-fg (lc/b theme :base07))
        (:scrollbar-track (lc--mix (lc/b theme :base00) "#000000" 0.5))
        (:scrollbar-thumb (lc--mix (lc/b theme :base0C) "#000000" 0.6))

        (:button-hover (lc/b theme :base02))
        (:button-active(lc/b theme :base03))
        (:button-fg    (lc/b theme :base07))
        (_ nil)))))

(defun lc/derive-base16-from-seed (ui syntax)
  (let* ((b00 (plist-get syntax :bg))
         (b05 (plist-get syntax :fg))
         (b03 (plist-get syntax :comment))
         (b07 (or (plist-get ui :white) "#ffffff"))
         (b01 (lc--mix b00 b03 0.50))
         (b02 (lc--mix b00 b03 0.80))
         (b04 (lc--mix b03 b05 0.60))
         (b06 (lc--mix b05 b07 0.50))
         (b08 (or (plist-get syntax :error)   "#ff6b6b"))
         (b09 (or (plist-get syntax :const)   "#d19a66"))
         (b0A (or (plist-get syntax :func)    "#e5c07b"))
         (b0B (or (plist-get syntax :string)  "#98c379"))
         (b0C (or (plist-get ui :ui-accent) (plist-get ui :accent) "#00bcd4"))
         (b0D (or (plist-get syntax :type)    "#61afef"))
         (b0E (or (plist-get syntax :keyword) "#c678dd"))
         (b0F (lc--mix b0E b08 0.40)))
    (list :base00 b00 :base01 b01 :base02 b02 :base03 b03
          :base04 b04 :base05 b05 :base06 b06 :base07 b07
          :base08 b08 :base09 b09 :base0A b0A :base0B b0B
          :base0C b0C :base0D b0D :base0E b0E :base0F b0F)))

(defun lc/add-derived-b16 (name ui syntax)
  (push (cons name (lc/derive-base16-from-seed ui syntax)) lc/base16))

;; --- safety guard: if the shared defs aren't loaded yet, define no-ops ---

    (concat "{"
            (mapconcat (lambda (s) (format "%S" s)) fonts ", ")
            "}")))

    (insert "local wezterm = require 'wezterm'\n")
    (insert "local config = {}\n\n")
    (insert "config.font_locator = 'FontConfig'\n\n")

    ;; helper: keep only families actually visible in the (Flatpak) sandbox
    (insert "local function _norm(s) return (s or \"\"):lower():gsub(\"%s+\",\"\"):gsub(\"[%p_%-%/]\",\"\") end\n")
    (insert "local function _strip_nf_variants(n)\n")
    (insert "  -- Collapse ONLY Nerd Font suffix forms; DO NOT strip 'mono' from real families.\n")
    (insert "  return n\n")
    (insert "    :gsub(\"nerdfontmono$\",\"nerdfont\")\n")
    (insert "    :gsub(\"nerdfontpropo$\",\"nerdfont\")\n")
    (insert "    :gsub(\"nerdfont$\", \"\")\n")
    (insert "end\n")
    (insert "local ALIASES = {\n")
    (insert "  sfmononerdfont = \"SF Mono\",\n")
    (insert "  sfmononerdfontmono = \"SF Mono\",\n")
    (insert "  sanfranciscomono = \"SF Mono\",\n")
    (insert "}\n")
    (insert "local function filter_available_families(candidates)\n")
    (insert "  local out = {}\n")
    (insert "  if not (wezterm.gui and wezterm.gui.enumerate_fonts) then return out end\n")
    (insert "  local have_by_norm, have_by_base, have_norm_list = {}, {}, {}\n")
    (insert "  for _, f in ipairs(wezterm.gui.enumerate_fonts()) do\n")
    (insert "    local n = _norm(f.family)\n")
    (insert "    have_by_norm[n] = f.family\n")
    (insert "    table.insert(have_norm_list, n)\n")
    (insert "    local base = _strip_nf_variants(n)\n")
    (insert "    have_by_base[base] = have_by_base[base] or {}\n")
    (insert "    table.insert(have_by_base[base], f.family)\n")
    (insert "  end\n")
    (insert "  local function pick(name)\n")
    (insert "    local n = _norm(name)\n")
    (insert "    -- alias preference first (e.g. prefer 'SF Mono' over NF variants)\n")
    (insert "    local alias = ALIASES[n]\n")
    (insert "    if alias and have_by_norm[_norm(alias)] then return have_by_norm[_norm(alias)] end\n")
    (insert "    -- exact family present?\n")
    (insert "    if have_by_norm[n] then return have_by_norm[n] end\n")
    (insert "    -- base match with NF suffixes collapsed\n")
    (insert "    local base = _strip_nf_variants(n)\n")
    (insert "    local opts = have_by_base[base]\n")
    (insert "    if opts and #opts > 0 then\n")
    (insert "      for _, fam in ipairs(opts) do if _norm(fam):sub(1, #n) == n then return fam end end\n")
    (insert "      return opts[1]\n")
    (insert "    end\n")
    (insert "  end\n")
    (insert "  for _, name in ipairs(candidates) do\n")
    (insert "    local fam = pick(name)\n")
    (insert "    if fam then table.insert(out, fam) end\n")
    (insert "  end\n")
    (insert "  return out\n")
    (insert "end\n\n")

                    (funcall b :base00) (funcall b :base01) (funcall b :base02) (funcall b :base03)

                    (funcall b :base08) (funcall b :base09) (funcall b :base0A) (funcall b :base0B)
                    (funcall b :base0C) (funcall b :base0D) (funcall b :base0E) (funcall b :base0F)))

    ;; terminal fonts (SFMono first, then your declared fallbacks, then wide coverage)
    (let* ((propo (or (lc/font-fallbacks 'mono-propo) '()))
           (mono  (or (lc/font-fallbacks 'mono) '()))
           (emoji (or (lc/font-fallbacks 'emoji) '()))
           ;; optional CJK (define in your lc-font-fallbacks if you want JP glyphs)
           (cjk   (or (lc/font-fallbacks 'cjk) '()))
           (wanted (append (list mono-fam) propo mono emoji cjk))
           (fonts  (delete-dups wanted)))
      (insert (format "local _fonts = %s\n" (lc/fonts->lua-array fonts)))
      (insert "local _fonts_filtered = filter_available_families(_fonts)\n")
      (insert "local function ensured(tbl, fallback) return (tbl and #tbl > 0) and tbl or fallback end\n")
      (insert "local _term_fonts = ensured(_fonts_filtered, _fonts)\n")

(insert "  end)\n")
(insert "end)\n")

    (insert "return config\n")))
