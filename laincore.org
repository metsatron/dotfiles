* Switchboard (central control)
#+BEGIN_SRC emacs-lisp :results silent
;;; --- CONFIG ---
;; Global defaults
(defvar lc/ui     'adobe-cc-dark-blue)   ;; UI chrome
(defvar lc/syntax 'adobe-cc-dark-blue)   ;; code highlighting

;; Per-app overrides (optional).
;; (app lane theme) — lane is 'ui or 'syntax
(defvar lc/overrides
  '(
    ;; examples:
    ;; (obsidian ui     adobe-cc-dark-blue)
    ;; (emacs    syntax monokai-pro)   ; keep chrome = global, syntax ≠ global
    ))

;;; --- RESOLUTION ---
(defun lc/get (app lane)
  "Pick theme for APP and LANE ('ui or 'syntax), falling back to global."
  (or (caddr (assoc app (cl-remove-if-not (lambda (x) (eq (cadr x) lane)) lc/overrides)))
      (pcase lane
        ('ui     lc/ui)
        ('syntax lc/syntax))))

(defun lc/emit (app lane path)
  "Return PATH only if (app,lane) is active for this tangle. Otherwise \"no\"."
  ;; For now: always allow. If you later want a dry-run gate, add checks here.
  path)

;;; --- PALETTE REGISTRY (UI) ---
;; Minimal canonical slots. Add more as needed (success/warn/err, neutrals, etc.)
(defvar lc/palettes
  '((adobe-cc-dark-blue
     :white "#ffffff" :lighter "#5b5b5b" :base "#454545" :darker "#303030" :black "#000000"
     :accent "#4b6983" :accent-l "#5b89b1" :accent-d "#324657")))

(defun lc/p (theme key)
  (plist-get (alist-get theme lc/palettes) key))

;;; --- SYNTAX REGISTRY (code colors) ---
;; Start by reusing the same palette; you can add real syntax schemes later.
(defvar lc/syntax-palettes
  '((adobe-cc-dark-blue
     :bg "#1f1f1f" :fg "#dbdbdb" :ui-accent "#4b6983"
     :string "#98c379" :keyword "#c678dd" :type "#61afef" :func "#e5c07b"
     :comment "#7f848e" :const "#d19a66" :error "#e06c75")))

(defun lc/s (theme key)
  (plist-get (alist-get theme lc/syntax-palettes) key))
#+END_SRC

* Emacs — Syntax (tiny faces demo; Doom/Spacemacs can load this)
#+BEGIN_SRC emacs-lisp :tangle (lc/emit 'emacs 'syntax "linux/.local/share/laincore/emacs-laincore.el")
;; Load with (load-file "~/.local/share/laincore/emacs-laincore.el")
(let* ((tname (lc/get 'emacs 'syntax))
       (fg   (lc/s tname :fg))
       (bg   (lc/s tname :bg))
       (kw   (lc/s tname :keyword))
       (str  (lc/s tname :string))
       (fnc  (lc/s tname :func))
       (typ  (lc/s tname :type))
       (com  (lc/s tname :comment))
       (cst  (lc/s tname :const))
       (err  (lc/s tname :error)))
  (custom-set-faces
   `(default ((t (:foreground ,fg :background ,bg))))
   `(font-lock-comment-face ((t (:foreground ,com :slant italic))))
   `(font-lock-string-face  ((t (:foreground ,str))))
   `(font-lock-keyword-face ((t (:foreground ,kw))))
   `(font-lock-function-name-face ((t (:foreground ,fnc))))
   `(font-lock-type-face ((t (:foreground ,typ))))
   `(font-lock-constant-face ((t (:foreground ,cst))))
   `(error ((t (:foreground ,err :weight bold))))))
#+END_SRC
