(defun dotspacemacs/user-init ()
  (let* ((cache (expand-file-name "~/.spacemacs.d.claude/.cache/"))
         (elpa  (expand-file-name "elpa" cache))
         (cust  (expand-file-name "custom.el" (expand-file-name "~/.spacemacs.d.claude/"))))
    (unless (file-exists-p cache) (make-directory cache t))
    (setq spacemacs-cache-directory cache
          package-user-dir          elpa
          custom-file               cust)
    (setq recentf-save-file (expand-file-name "recentf" cache)
          recentf-max-saved-items 200
          recentf-auto-cleanup 'never)
    (require 'recentf nil t)
    (when (fboundp 'recentf-mode) (recentf-mode 1)))
  (unless (boundp 'treemacs--buffer-name-prefix)
    (defvar treemacs--buffer-name-prefix " *Treemacs-")))
#+END_SRC

;; Window moves on NEIO (leader + C-w)
(spacemacs/set-leader-keys

(with-eval-after-load 'evil
  ;; C-w n/e/i/o for windows too
  (define-key evil-window-map "n" 'evil-window-left)
  (define-key evil-window-map "e" 'evil-window-down)
  (define-key evil-window-map "i" 'evil-window-up)
  (define-key evil-window-map "o" 'evil-window-right)

  ;; ---- Pin NEIO & open-line on l/L ----
  (defun metsatron/pin-neio ()
    ;; make sure default o/O don’t leak through
    (dolist (m '(evil-normal-state-map evil-motion-state-map evil-visual-state-map))
      (when (boundp m)
        (define-key (symbol-value m) (kbd "o") nil)
        (define-key (symbol-value m) (kbd "O") nil)))
    ;; movement on n/e/i/o (overriding map)
    (evil-define-key* '(normal motion visual) 'global
      (kbd "n") #'evil-backward-char
      (kbd "e") #'evil-next-line
      (kbd "i") #'evil-previous-line
      (kbd "o") #'evil-forward-char)
    ;; your “open line” on l / L in NORMAL
    (evil-define-key* 'normal 'global
      (kbd "l") #'evil-open-below
      (kbd "L") #'evil-open-above))

  ;; apply now that Evil is loaded
  (metsatron/pin-neio)

  ;; --- Org: NEIO + l/L open-line, and keep it pinned ---
  (with-eval-after-load 'evil-org
    (defun metsatron/org-neio ()
      ;; kill any lingering org local o/O
      (define-key org-mode-map (kbd "o") nil)
      (define-key org-mode-map (kbd "O") nil)
      ;; NEIO movement in Normal/Motion
      (evil-define-key* '(normal motion) org-mode-map
        (kbd "n") #'evil-backward-char
        (kbd "e") #'evil-next-line
        (kbd "i") #'evil-previous-line
        (kbd "o") #'evil-forward-char)
      ;; open-line lives on l/L in Normal
      (evil-define-key* 'normal org-mode-map
        (kbd "l") #'evil-open-below
        (kbd "L") #'evil-open-above)))

  ;; run when evil-org loads
  (with-eval-after-load 'evil-org
    (add-hook 'org-mode-hook #'metsatron/org-neio))

  ;; keyboard-layout can rewrite keys later; pin again after it loads
  (with-eval-after-load 'keyboard-layout
    (with-eval-after-load 'evil-org
      (add-hook 'org-mode-hook #'metsatron/org-neio)
      ;; also re-pin in already-open Org buffers
      (dolist (b (buffer-list))
        (with-current-buffer b
          (when (derived-mode-p 'org-mode) (metsatron/org-neio))))))

  ;; quick one-shot command if needed: M-x metsatron/org-neio on a buffer

  ;; ---- Visual i must not be a prefix; make it move up ----
  (evil-define-key* 'visual 'global (kbd "i") #'evil-previous-line)
  (evil-define-key* 'visual 'global (kbd "I") nil)

  ;; ---- Keep selection after shift left/right ----
  (setq evil-shift-round nil)

  (defun metsatron/visual--exact-range ()
    "Return (BEG END TYPE) of current visual region, normalizing linewise ends."
    (let* ((type evil-visual-type)
           (range (evil-visual-range))
           (beg (copy-marker (nth 0 range)))
           (end (copy-marker (nth 1 range))))
      (when (eq type 'line)
        (save-excursion
          (goto-char beg) (set-marker beg (line-beginning-position))
          (goto-char end) (set-marker end (line-end-position))))
      (list beg end type)))

  (defun metsatron/visual--reselect (beg end type)
    (evil-visual-select beg end type)
    (evil-normalize-visual-state))

  (defun metsatron/visual-shift-right (_beg _end count)
    (interactive "<r><c>")
    (pcase-let ((`(,rb ,re ,rt) (metsatron/visual--exact-range)))
      (evil-shift-right rb re count)
      (metsatron/visual--reselect rb re rt)))

  (defun metsatron/visual-shift-left (_beg _end count)
    (interactive "<r><c>")
    (pcase-let ((`(,rb ,re ,rt) (metsatron/visual--exact-range)))
      (evil-shift-left rb re count)
      (metsatron/visual--reselect rb re rt)))

  (define-key evil-visual-state-map (kbd ">") #'metsatron/visual-shift-right)
  (define-key evil-visual-state-map (kbd "<") #'metsatron/visual-shift-left)

  ;; ---- Optional: translate plain "i" to <up> only while in Visual ----

    (if (and (boundp 'evil-state) (eq evil-state 'visual)) [up] (vector ?i)))
  (define-key key-translation-map (kbd "i") #'metsatron/translate-i-if-visual))

(defun metsatron/persp--save (file)
  (cond ((fboundp 'persp-state-save)           (persp-state-save file))
        ((fboundp 'persp-save-state-to-file)   (persp-save-state-to-file file))
        (t (with-temp-file file
             (prin1 (list :type 'window-state :state (window-state-get nil t)) (current-buffer))))))
(defun metsatron/persp--load (file)
  (cond ((fboundp 'persp-state-load)           (persp-state-load file))
        ((fboundp 'persp-load-state-from-file) (persp-load-state-from-file file))
        (t
         (let ((data (with-temp-buffer (insert-file-contents file) (read (current-buffer)))))
           (when (eq (plist-get data :type) 'window-state)
             (when (and (fboundp 'vterm)
                        (save-excursion (goto-char (point-min))
                                        (search-forward "\"*vterm*\"" nil t)))
               (ignore-errors (vterm)))
             (let ((ignore-window-parameters t))
               (dolist (w (window-list))
                 (when (window-parameter w 'window-side) (delete-window w))))
             (window-state-put (plist-get data :state) (frame-root-window) 'safe))))))

  (interactive)
  (delete-other-windows)
  (let* ((left  (selected-window))
         (right (split-window-right)))
    (select-window left)
    (let* ((term (split-window-below)))
      (select-window term)
      (if (fboundp 'vterm) (vterm) (ansi-term (or (getenv "SHELL") "/bin/bash")))
      (let ((desired 8))
        (window-resize term (- desired (window-total-height term)))))
    (select-window right))
  (treemacs)
  (when (fboundp 'treemacs-set-width) (treemacs-set-width 20)))

#+BEGIN_SRC emacs-lisp
;; Idempotent Centaur Tabs setup: safe to include in multiple profiles
(require 'centaur-tabs nil t)
(eval-when-compile (defvar metsatron/ct--setup-done nil))

(with-eval-after-load 'centaur-tabs
  (unless (bound-and-true-p metsatron/ct--setup-done)
    ;; Core options
    (setq centaur-tabs-style "rounded"
          centaur-tabs-height 24
          centaur-tabs-set-icons nil
          centaur-tabs-set-close-button nil
          centaur-tabs-set-modified-marker t
          centaur-tabs-modified-marker "*"
          centaur-tabs-cycle-scope 'tabs
          centaur-tabs-show-new-tab-button nil
          centaur-tabs-show-navigation-buttons t
          centaur-tabs-group-by-projectile-project t)

    ;; Use our own grouping
    (setq centaur-tabs-group-by-projectile-project nil)

    ;; Groups
    (defun metsatron/ct-buffer-groups ()
      (list
        (cond
        ((or (string-prefix-p " *" (buffer-name))
              (member (buffer-name)
                      '("*spacemacs*" "*Messages*" "*Compile-Log*" "*Help*"
                        "*Warnings*" "*Backtrace*" "*Which Key*")))
          "Service")
        ((memq major-mode '(treemacs-mode vterm-mode term-mode
                                          magit-status-mode magit-process-mode
                                          package-menu-mode))
          "Tools")
        ((buffer-file-name) "All")
        (t "Buffers"))))
    (setq centaur-tabs-buffer-groups-function #'metsatron/ct-buffer-groups)

    ;; Hide tab function (accepts 1-arg or 2-arg call sites)
    (setq centaur-tabs-hide-tab-function
          (lambda (&rest args)
            (let* ((buffer (if (= (length args) 1) (car args) (cadr args)))
                    (name   (if (= (length args) 1) (buffer-name buffer) (car args)))
                    (mode   (buffer-local-value 'major-mode buffer)))
              (or (string-prefix-p " *Treemacs" name)
                  (string-prefix-p "*helm" name)
                  (member name '("*Which Key*" "*spacemacs*" "*Messages*" "*scratch*" "*Completions*" "*info*" "*Info*"))
                  (memq mode '(helm-major-mode spacemacs-buffer-mode which-key-mode
                                                treemacs-mode help-mode helpful-mode Info-mode))))))

    ;; Extra excludes (cheap, fast)
    (setq centaur-tabs-excluded-regexps
          '("^\\*helm" "^\\*scratch\\*$" "^\\*info\\*$" "^\\*Messages\\*$" "^\\*Which Key\\*$" "^ \\*Treemacs"))

    ;; Hard filter after list build (keeps noise low)
    (require 'seq)
    (defun metsatron/ct-should-hide (buf)
      (let* ((name (buffer-name buf))
              (mode (buffer-local-value 'major-mode buf)))
        (or (string-prefix-p " *Treemacs" name)
            (string-prefix-p "*helm" name)
            (member name '("*Which Key*" "*spacemacs*" "*Messages*" "*scratch*" "*Completions*" "*info*" "*Info*"))
            (memq mode '(helm-major-mode spacemacs-buffer-mode which-key-mode
                                          treemacs-mode help-mode helpful-mode Info-mode)))))
    (advice-add 'centaur-tabs-buffer-list :filter-return
                (lambda (buffers) (seq-remove #'metsatron/ct-should-hide buffers)))

    ;; Keys
    (global-set-key (kbd "<C-prior>")   #'centaur-tabs-backward)
    (global-set-key (kbd "<C-next>")    #'centaur-tabs-forward)
    (global-set-key (kbd "<C-S-prior>") #'centaur-tabs-move-current-tab-to-left)
    (global-set-key (kbd "<C-S-next>")  #'centaur-tabs-move-current-tab-to-right)
    (global-set-key (kbd "C-S-w")       #'kill-current-buffer)

    (spacemacs/declare-prefix "ot" "tabs")
    (spacemacs/set-leader-keys
      "ot[" #'centaur-tabs-backward-group
      "ot]" #'centaur-tabs-forward-group)
    (global-set-key (kbd "M-<prior>") #'centaur-tabs-backward-group)
    (global-set-key (kbd "M-<next>")  #'centaur-tabs-forward-group)

    ;; Reopen last closed file
    (defvar metsatron/killed-file-stack nil)
    (add-hook 'kill-buffer-hook (lambda () (when buffer-file-name
                                              (push buffer-file-name metsatron/killed-file-stack))))
    (defun metsatron/reopen-last-closed-tab () (interactive)
      (let ((f (seq-find #'file-exists-p metsatron/killed-file-stack)))
        (if f (progn (find-file f)
                      (setq metsatron/killed-file-stack (delete f metsatron/killed-file-stack)))
          (message "No recently killed file buffer found."))))
    (global-set-key (kbd "C-S-t") #'metsatron/reopen-last-closed-tab)

    ;; Home buffer key (global + emulation layer)
    (defun metsatron/go-home ()
      (interactive)
      (if (get-buffer "*spacemacs*")
          (switch-to-buffer "*spacemacs*")
        (spacemacs-buffer/goto-buffer)))
    (global-set-key (kbd "C-t") #'metsatron/go-home)
    (defvar metsatron/keys-emulation-mode t)
    (defvar metsatron/emulation-map (make-sparse-keymap))
    (define-key metsatron/emulation-map (kbd "C-t") #'metsatron/go-home)
    (add-to-list 'emulation-mode-map-alists
                  `((metsatron/keys-emulation-mode . ,metsatron/emulation-map)))

    ;; One-time mode refresh (no duplicates)
    (when (fboundp 'centaur-tabs-mode)
      (centaur-tabs-mode -1)
      (centaur-tabs-mode +1))

    (setq metsatron/ct--setup-done t)))

;; Ensure centaur-tabs is loaded & enabled at startup
(add-hook 'emacs-startup-hook
          (lambda ()
            (require 'centaur-tabs nil t)
            (when (featurep 'centaur-tabs)
              (centaur-tabs-mode 1))))

(centaur-tabs-mode 1))
