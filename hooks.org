* Claude statusline
#+BEGIN_SRC bash :tangle all/.local/bin/claude-statusline :shebang "#!/usr/bin/env bash" :comments no
input=$(cat)

# User and host
user=$(whoami)
host=$(hostname -s)

# Directory — current dir and parent only (2 segments)
cwd=$(echo "$input" | jq -r '.workspace.current_dir // .cwd')
short_cwd=$(echo "$cwd" | awk -F'/' '{
  n=NF; if(n<=2){print $0}
  else {printf "…/"; for(i=n-1;i<=n;i++){printf $i; if(i<n) printf "/"} printf "\n"}
}')

# Git branch from worktree or workspace
git_branch=$(echo "$input" | jq -r '.worktree.branch // .workspace.git_worktree // empty')
if [ -z "$git_branch" ] && command -v git >/dev/null 2>&1; then
  git_branch=$(git -C "$cwd" symbolic-ref --short HEAD 2>/dev/null || git -C "$cwd" rev-parse --short HEAD 2>/dev/null)
fi

# Time
time_str=$(date +%H:%M)

# Model short name
model=$(echo "$input" | jq -r '.model.display_name // empty' | sed 's/Claude //')

# Context USED (not remaining)
used=$(echo "$input" | jq -r '.context_window.used_percentage // empty')

# Token count as XK or X.XK
tokens=$(echo "$input" | jq -r '(.context_window.total_input_tokens // 0) + (.context_window.total_output_tokens // 0)')
if [ -n "$tokens" ] && [ "$tokens" != "null" ] && [ "$tokens" -gt 0 ] 2>/dev/null; then
  tk=$(echo "$tokens" | awk '{if($1>=1000) printf "%.1fK",$1/1000; else printf "%d",$1}')
else
  tk=""
fi

# 5-hour rolling window — remaining (prefer dedicated field, fall back to 100-used)
five_hr=$(echo "$input" | jq -r '
  if .rate_limits.five_hour.remaining_percentage != null
  then .rate_limits.five_hour.remaining_percentage
  elif .rate_limits.five_hour.used_percentage != null
  then (100 - .rate_limits.five_hour.used_percentage)
  else empty
  end')

printf "\033[38;5;166m %s\033[0m" "$user@$host"
printf " \033[33m %s\033[0m" "$short_cwd"
if [ -n "$git_branch" ]; then
  printf " \033[34m %s\033[0m" "$git_branch"
fi
printf "\033[38;5;59m  %s\033[0m" "$time_str"
if [ -n "$model" ]; then
  printf " \033[36m✦ %s\033[0m" "$model"
fi
if [ -n "$tk" ] && [ -n "$used" ]; then
  printf " \033[38;5;59m%s (%s%%)\033[0m" "$tk" "$(printf '%.0f' "$used")"
elif [ -n "$used" ]; then
  printf " \033[38;5;59m(%s%%)\033[0m" "$(printf '%.0f' "$used")"
fi
if [ -n "$five_hr" ]; then
  printf " \033[38;5;59m5h %s%%\033[0m" "$(printf '%.0f' "$five_hr")"
fi
#+END_SRC

#+BEGIN_SRC bash :tangle all/.local/bin/dotcortex-opencode-config-merge-guard :shebang "#!/usr/bin/env bash"
# ~/DotCortex/all/.local/bin/dotcortex-opencode-config-merge-guard
set -euo pipefail

mode="${1:-check}"
if [ "$mode" != "check" ] && [ "$mode" != "apply" ]; then
  echo "usage: dotcortex-opencode-config-merge-guard [check|apply]" >&2
  exit 2
fi

repo="${DOTCORTEX_REPO:-$HOME/DotCortex}"
desired="${DOTCORTEX_OPENCODE_DESIRED:-$repo/all/.config/opencode/opencode.json}"
live="${DOTCORTEX_OPENCODE_LIVE:-$HOME/.config/opencode/opencode.json}"
state_home="${XDG_STATE_HOME:-$HOME/.local/state}"
marker="${DOTCORTEX_OPENCODE_MERGE_MARKER:-$state_home/dotcortex/opencode-config-merge.json}"

python3 - "$mode" "$repo" "$desired" "$live" "$marker" <<'PYEOF'
import json
import os
import shutil
import subprocess
import sys
from datetime import datetime

mode, repo, desired_path, live_path, marker_path = sys.argv[1:6]


def same_path(a, b):
    try:
        return os.path.samefile(a, b)
    except OSError:
        return False


def path_join(base, key):
    return f"{base}.{key}" if base else key


def live_only_paths(live_value, desired_value, path=""):
    if isinstance(live_value, dict):
        if not isinstance(desired_value, dict):
            return [path or "$"]
        paths = []
        for key, value in live_value.items():
            child = path_join(path, key)
            if key not in desired_value:
                paths.append(child)
                continue
            paths.extend(live_only_paths(value, desired_value[key], child))
        return paths

    if isinstance(live_value, list):
        if not isinstance(desired_value, list):
            return [path or "$"]
        return [path or "$"] if any(item not in desired_value for item in live_value) else []

    return [path or "$"] if live_value != desired_value else []


def safe_remove(path):
    if os.path.islink(path) or os.path.isfile(path):
        os.remove(path)
        return
    if os.path.isdir(path):
        shutil.rmtree(path)


def clear_marker():
    if os.path.exists(marker_path):
        os.remove(marker_path)


if not os.path.isfile(desired_path):
    print(f"error: desired config missing: {desired_path}", file=sys.stderr)
    sys.exit(1)

with open(desired_path, "r", encoding="utf-8") as handle:
    try:
        desired_value = json.load(handle)
    except json.JSONDecodeError as exc:
        print(f"error: desired config is invalid JSON: {exc}", file=sys.stderr)
        sys.exit(1)

if same_path(desired_path, live_path):
    clear_marker()
    print("status=managed")
    sys.exit(0)

if not (os.path.exists(live_path) or os.path.islink(live_path)):
    clear_marker()
    print("status=missing")
    sys.exit(0)

extra_paths = []
try:
    with open(live_path, "r", encoding="utf-8") as handle:
        live_value = json.load(handle)
    extra_paths = sorted(set(live_only_paths(live_value, desired_value)))
except json.JSONDecodeError as exc:
    extra_paths = [f"$ (invalid JSON: {exc})"]

if mode == "check":
    if extra_paths:
        print("status=backup")
        for item in extra_paths:
            print(f"extra={item}")
    else:
        print("status=overwrite")
    sys.exit(0)

os.makedirs(os.path.dirname(marker_path), exist_ok=True)

if extra_paths:
    timestamp = datetime.now().strftime("%Y%m%d-%H%M%S")
    backup_path = f"{live_path}.bak.{timestamp}"
    subprocess.run(["cp", "-a", live_path, backup_path], check=True)
    safe_remove(live_path)
    payload = {
        "repo": os.path.realpath(repo),
        "live": live_path,
        "desired": desired_path,
        "backup": backup_path,
        "created_at": datetime.now().isoformat(timespec="seconds"),
        "extra_paths": extra_paths,
    }
    with open(marker_path, "w", encoding="utf-8") as handle:
        json.dump(payload, handle, indent=2)
        handle.write("\n")
    print(f"status=backup backup={backup_path}")
    for item in extra_paths:
        print(f"extra={item}")
    sys.exit(0)

clear_marker()
safe_remove(live_path)
print("status=overwrite")
PYEOF

* Git commit guard
Block the exact failure mode that caused the recent regression:

- refuse commits from a dirty tracked worktree
- reject backup/runtime spillover files
- reject generated output commits that do not include canonical Org source
- reject giant sweep commits unless explicitly overridden

Escape hatches exist for intentional exceptional cases:

- =DOTCORTEX_GUARD_ALLOW_DIRTY=1=
- =DOTCORTEX_GUARD_ALLOW_BULK=1=
- =DOTCORTEX_GUARD_ALLOW_JUNK=1=

#+BEGIN_SRC bash :tangle all/.local/bin/dotcortex-git-guard :shebang "#!/usr/bin/env bash"
# ~/DotCortex/all/.local/bin/dotcortex-git-guard
set -euo pipefail

note() {
  printf 'dotcortex-guard: %s\n' "$*" >&2
}

die() {
  note "$1"
  exit 1
}

repo="$(git rev-parse --show-toplevel 2>/dev/null || true)"
[ -n "$repo" ] || exit 0
cd "$repo"

allow_dirty="${DOTCORTEX_GUARD_ALLOW_DIRTY:-0}"
allow_bulk="${DOTCORTEX_GUARD_ALLOW_BULK:-0}"
allow_junk="${DOTCORTEX_GUARD_ALLOW_JUNK:-0}"

staged=()
while IFS= read -r -d '' path; do
  staged+=("$path")
done < <(git diff --cached --name-only -z --diff-filter=ACMRD)

[ "${#staged[@]}" -gt 0 ] || exit 0

unstaged=()
while IFS= read -r -d '' path; do
  unstaged+=("$path")
done < <(git diff --name-only -z --diff-filter=ACMRD)

if [ "${#unstaged[@]}" -gt 0 ] && [ "$allow_dirty" != "1" ]; then
  note "refusing commit from a dirty tracked worktree"
  printf '%s\n' "${unstaged[@]:0:20}" | sed 's/^/  - /' >&2
  if [ "${#unstaged[@]}" -gt 20 ]; then
    note "... and $((${#unstaged[@]} - 20)) more"
  fi
  note "stash or stage intentionally, then retry"
  note "override once with: DOTCORTEX_GUARD_ALLOW_DIRTY=1 git commit ..."
  exit 1
fi

declare -A roots=()
bad_paths=()
has_org_source=0
generated_paths=0

for path in "${staged[@]}"; do
  root="${path%%/*}"
  roots["$root"]=1

  case "$path" in
    *.org)
      has_org_source=1
      ;;
  esac

  case "$path" in
    all/*|linux/*|debian/*|devuan/*|x230/*|t480s/*|arch/*|osx/*|be/*|navi/*|think/*|.claude/*|.opencode/*|.agents/*|AGENTS.md|CLAUDE.md|Makefile)
      generated_paths=$((generated_paths + 1))
      ;;
  esac

  case "$path" in
    *.bak|*.bak.*|*.backup-*|*~|*.orig|*.rej|*.swp|*.swo|.#*|[#]*[#]|.vorg.db|.claude/stop_count|.codex|all/.hermes/channel_directory.json|all/.hermes/cron/*)
      bad_paths+=("$path")
      ;;
  esac
done

if [ "${#bad_paths[@]}" -gt 0 ] && [ "$allow_junk" != "1" ]; then
  note "refusing commit with backup/runtime spillover paths"
  printf '%s\n' "${bad_paths[@]}" | sed 's/^/  - /' >&2
  note "clean or ignore them first"
  note "override once with: DOTCORTEX_GUARD_ALLOW_JUNK=1 git commit ..."
  exit 1
fi

if [ "$generated_paths" -gt 0 ] && [ "$has_org_source" -eq 0 ]; then
  die "generated output is staged without a canonical .org source change"
fi

root_count="${#roots[@]}"
staged_count="${#staged[@]}"
if [ "$staged_count" -gt 80 ] && [ "$root_count" -gt 10 ] && [ "$allow_bulk" != "1" ]; then
  note "refusing giant sweep commit ($staged_count staged paths across $root_count top-level roots)"
  note "split it into reviewed chunks"
  note "override once with: DOTCORTEX_GUARD_ALLOW_BULK=1 git commit ..."
  exit 1
fi

if [ "$has_org_source" -eq 1 ] && [ "$generated_paths" -eq 0 ]; then
  note "source-only commit detected; if this change should affect generated output, run tangle first"
fi

exit 0
#+END_SRC

while IFS=' ' read -r pkg marketplace_name marketplace_repo; do

  [ -z "${pkg:-}" ] && continue

  if [ -z "${seen_marketplaces[$marketplace_name]:-}" ]; then
    echo "marketplace: $marketplace_name ($marketplace_repo)"
    claude plugin marketplace add "$marketplace_repo" 2>/dev/null || true
    seen_marketplaces["$marketplace_name"]=1
  fi

  installed_json="$HOME/.claude/plugins/installed_plugins.json"
  key="${pkg}@${marketplace_name}"

  if [ -f "$installed_json" ] && python3 -c "
import json, sys
data = json.load(open('$installed_json'))
sys.exit(0 if '$key' in data.get('plugins', {}) else 1)
" 2>/dev/null; then
    echo "ok: $key"
  else
    echo "install: $key"
    claude plugin install "${pkg}@${marketplace_name}"
  fi

done < <(awk '!/^[[:space:]]*#/ && NF' "$MAN")

echo "Done. Restart Claude Code or run /reload-plugins to activate."

# Extract the ## LightCone section from a TODO file (falls back to full file if no LightCone section)
extract_lightcone() {
  local content="$1"
  local extracted
  extracted=$(printf '%s' "$content" | awk '/^## LightCone/{p=1} p && /^## / && !/^## LightCone/{exit} p')
  if [ -n "$extracted" ]; then
    printf '%s\n' "$extracted"
  else
    printf '%s\n' "$content"
  fi
}

WORKSPACE=$(basename "$PWD")
echo "--- $WORKSPACE LightCone ---"
todo_content=$(flatpak run md.obsidian.Obsidian vault=HelmCortex read path="LOGS/TODO/$WORKSPACE.md" 2>/dev/null)
if [ -n "$todo_content" ]; then
  extract_lightcone "$todo_content"
else
  echo "(no $WORKSPACE TODO found)"
fi

echo
HOST=$(hostname)
echo "--- Machine LightCone ($HOST) ---"
machine_content=$(flatpak run md.obsidian.Obsidian vault=HelmCortex read path="LOGS/TODO/Machines/$HOST.md" 2>/dev/null)
if [ -n "$machine_content" ]; then
  extract_lightcone "$machine_content"
else
  echo "(no machine TODO found)"

* Skill forced eval
Force explicit YES/NO skill evaluation before every user prompt. Lifts skill activation
from opportunistic to reliable. Only fires when =.claude/skills/= exists in the working
directory (project-scoped -- no-op outside DotCortex).

#+BEGIN_SRC bash :tangle all/.local/bin/claude-hook-user-prompt-submit :shebang "#!/usr/bin/env bash"
# ~/DotCortex/all/.local/bin/claude-hook-user-prompt-submit
# Inject DotCortex reminders before every task.
# Fires for Claude Code (UserPromptSubmit) and OpenCode.
# No-op outside the DotCortex repo.
# Emits skill evaluation context when .claude/skills/ exists in $PWD.
# Also surfaces pending OpenCode config merge reminders from stow-time drift detection.
set -euo pipefail

python3 - <<'PYEOF'
import os, sys, json

blocks = []
state_home = os.environ.get('XDG_STATE_HOME', os.path.join(os.path.expanduser('~'), '.local', 'state'))
marker_path = os.path.join(state_home, 'dotcortex', 'opencode-config-merge.json')
cwd = os.path.realpath(os.getcwd())
repo = os.path.realpath(os.environ.get('DOTCORTEX_REPO', os.path.join(os.path.expanduser('~'), 'DotCortex')))

if cwd != repo and not cwd.startswith(repo + os.sep):
    sys.exit(0)

if os.path.isfile(marker_path):
    try:
        with open(marker_path) as f:
            marker = json.load(f)
    except Exception:
        marker = None

    repo = os.path.realpath(marker.get('repo', '')) if marker else ''
    if marker and repo and (cwd == repo or cwd.startswith(repo + os.sep)):
        extra_paths = marker.get('extra_paths', [])
        bullets = ['- `{}`'.format(path) for path in extra_paths[:10]]
        if len(extra_paths) > 10:
            bullets.append('- ... and {} more'.format(len(extra_paths) - 10))
        if not bullets:
            bullets = ['- (paths unavailable)']
        blocks.append(
            '## OPENCODE CONFIG MERGE PENDING\n\n'
            'DotCortex backed up a live OpenCode config because it had settings '
            'not represented in `hooks.org`. Merge them back into source before '
            'deleting the backup.\n\n'
            '- Live config: `{}`\n'
            '- Backup: `{}`\n'
            '- Unmerged paths:\n{}'.format(
                marker.get('live', '(unknown live config)'),
                marker.get('backup', '(unknown backup path)'),
                '\n'.join(bullets),
            )
        )

skills_dir = os.path.join(os.getcwd(), '.claude', 'skills')
if os.path.isdir(skills_dir):
    skill_list = []
    for slug in sorted(os.listdir(skills_dir)):
        skill_path = os.path.join(skills_dir, slug, 'SKILL.md')
        description = ''
        if os.path.isfile(skill_path):
            with open(skill_path) as f:
                for line in f:
                    if line.startswith('description:'):
                        description = line.split(':', 1)[1].strip()
                        break
        skill_list.append('- {}: {}'.format(slug, description))

    if skill_list:
        skills_text = '\n'.join(skill_list)
        blocks.append(
            '## SKILL EVALUATION REQUIRED\n\n'
            'Before proceeding, explicitly evaluate each skill YES or NO '
            'with one-line reasoning. Once you write YES, you are committed '
            'to loading it.\n\n### Available skills:\n' + skills_text +
            '\n\nComplete the evaluation now, then proceed.'
        )

if not blocks:
    sys.exit(0)

output = {
    'hookSpecificOutput': {
        'hookEventName': 'UserPromptSubmit',
        'additionalContext': '\n\n'.join(blocks)
    }
}
print(json.dumps(output))
PYEOF
#+END_SRC

# ~/DotCortex/all/.local/bin/claude-hook-pre-compact
WORKSPACE=$(basename "$PWD")
MEMFILE="$PWD/.claude/MEMORY.md"
CONTENT=$(tail -n +2 "$MEMFILE" 2>/dev/null | sed '/^$/d' | sed '/^## Session Facts$/d' | sed '/^## User Model.*$/d')
if [ -n "$CONTENT" ]; then
  DUMP=$(printf '## Memory Dump %s\n\n%s' "$(date +%Y-%m-%d)" "$CONTENT")
  flatpak run md.obsidian.Obsidian vault=HelmCortex append path="LOGS/TODO/$WORKSPACE.md" content="$DUMP" 2>/dev/null
  printf '# Session Memory (auto-updated by Opus, max 2000 chars)\n\n## Session Facts\n\n\n## User Model (max 1500 chars for this section)\n\n' > "$MEMFILE"
  echo "Memory saved to $WORKSPACE TODO via Obsidian CLI and MEMORY.md truncated."
else
  echo 'MEMORY.md empty, nothing to save.'
fi
#+END_SRC
