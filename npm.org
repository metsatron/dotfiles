* Capture current global npm → manifest
# Append any new global npm packages into DotCortex manifest, do not clobber.
#+BEGIN_SRC bash :tangle all/.local/bin/npm-capture :shebang "#!/usr/bin/env bash"
# ~/DotCortex/all/.local/bin/npm-capture
set -euo pipefail

MAN="$HOME/DotCortex/all/.npm/manifest/global.ssv"
RUN="$HOME/.local/bin/dotcortex-node-env"

mkdir -p "$(dirname "$MAN")"
touch "$MAN"

tmp_existing="$(mktemp)"
tmp_have="$(mktemp)"
tmp_new="$(mktemp)"

# Existing packages already tracked in manifest
awk '!/^[[:space:]]*#/ && NF { print $1 }' "$MAN" | sort -u >"$tmp_existing"

# Current global packages (names only) from npm list -g --depth=0
"$RUN" npm list -g --depth=0 2>/dev/null \
  | awk 'NR>1 && /@/ {
      for (i = 1; i <= NF; i++) {
        if ($i ~ /@/) {
          split($i, a, "@");
          if (a[1] != "") print a[1];
          break;
        }
      }
    }' \
  | sort -u >"$tmp_have"

# new = have - existing
comm -13 "$tmp_existing" "$tmp_have" >"$tmp_new"

rm -f "$tmp_existing" "$tmp_have"

if [ ! -s "$tmp_new" ]; then
  rm -f "$tmp_new"
  echo "No new npm globals to capture into $MAN"
  exit 0
fi

while read -r pkg; do
  [ -n "$pkg" ] || continue
  [ "$pkg" = "npm" ] && continue
  # PKG VERSION FLAGS GLOBAL SOURCE REGISTRY TARBALL EXTRA
  printf '%s "" "" "global" "registry" "" "" ""\n' "$pkg" >>"$MAN"
done <"$tmp_new"

rm -f "$tmp_new"

echo "Updated $MAN"
#+END_SRC

set -euo pipefail

MAN="$HOME/DotCortex/all/.npm/manifest/global.ssv"
RUN="$HOME/.local/bin/dotcortex-node-env"

tmp_want="$(mktemp)"
tmp_have="$(mktemp)"

# Desired packages from manifest
awk '!/^[[:space:]]*#/ && NF { print $1 }' "$MAN" | sort -u >"$tmp_want"

# Live global packages from npm
"$RUN" npm list -g --depth=0 2>/dev/null \
  | awk 'NR>1 && /@/ {
      for (i = 1; i <= NF; i++) {
        if ($i ~ /@/) {
          split($i, a, "@");
          if (a[1] != "") print a[1];
          break;
        }
      }
    }' \
  | sort -u >"$tmp_have"

echo "== npm plan =="
echo "[to install]"
comm -23 "$tmp_want" "$tmp_have" | sed 's/^/  + /'

echo "[to uninstall]"
comm -13 "$tmp_want" "$tmp_have" \
  | grep -v '^npm$' \
  | sed 's/^/  - /'

rm -f "$tmp_want" "$tmp_have"

* Apply manifest inside DotCortex env
#+BEGIN_SRC bash :tangle all/.local/bin/npm-apply :shebang "#!/usr/bin/env bash"
# ~/DotCortex/all/.local/bin/npm-apply
set -euo pipefail

MAN="$HOME/DotCortex/all/.npm/manifest/global.ssv"
RUN="$HOME/.local/bin/dotcortex-node-env"

ENFORCE="${ENFORCE:-1}"   # 1 = enforce manifest membership, 0 = installs only
UNINSTALL="${UNINSTALL:-1}"
UPDATE="${UPDATE:-0}"     # 1 = allow upgrades, 0 = leave existing versions alone

echo "== npm apply =="

# Normalize manifest rows to 8 pipe-separated fields, honoring "" as empty.
_emit_norm() {
  awk -v FS='[ \t]+' '
    function dequote(s){ return (s=="\"\"" ? "" : s) }
    $0 ~ /^[ \t]*#/ || NF==0 { next }
    {
      printf "%s|%s|%s|%s|%s|%s|%s|%s\n",
        dequote($1),dequote($2),dequote($3),dequote($4),
        dequote($5),dequote($6),dequote($7),dequote($8)
    }
  ' "$MAN"
}

# Build map of live package versions from npm list -g --depth=0
declare -A live_ver

while read -r name ver; do
  [ -n "${name:-}" ] || continue
  live_ver["$name"]="$ver"
done < <(
  "$RUN" npm list -g --depth=0 2>/dev/null \
    | awk 'NR>1 && /@/ {
        for (i = 1; i <= NF; i++) {
          if ($i ~ /@/) {
            split($i, a, "@");
            if (a[1] != "") print a[1], a[2];
            break;
          }
        }
      }'
)

declare -A want_names

# Apply manifest: install / upgrade as needed
while IFS='|' read -r pkg version flags global_flag source registry tarball extra
do
  [ -n "$pkg" ] || continue
  want_names["$pkg"]=1

  # npm itself is usually managed with Node, not via this loom
  if [ "$pkg" = "npm" ]; then
    continue
  fi

  installed_ver="${live_ver[$pkg]:-}"

  if [ -n "$installed_ver" ]; then
    # Already installed
    if [ "$UPDATE" != "1" ]; then
      # Do not touch existing packages in default mode
      continue
    fi

    if [ -z "$version" ]; then
      # No explicit version in manifest, treat installed version as canonical
      continue
    fi

    if [ "$version" = "$installed_ver" ]; then
      # Already at requested version
      continue
    fi
  fi

  # If we get here, pkg is missing or we are explicitly upgrading.
  spec="$pkg"
  if [ -n "$version" ]; then
    spec="$pkg@$version"
  fi

  args=(install -g "$spec")

  # Allow FLAGS column to carry extra npm args, comma-separated.
  if [ -n "$flags" ]; then
    IFS=',' read -ra FF <<<"$flags"
    for f in "${FF[@]}"; do
      [ -n "$f" ] && args+=("$f")
    done
  fi

  "$RUN" npm "${args[@]}"
done < <(_emit_norm)

# Optional uninstall of extras
if [ "$ENFORCE" = "1" ] && [ "$UNINSTALL" = "1" ]; then
  echo "== uninstall extras =="

  # Sorted iteration for stable output; npm itself is skipped.
  for pkg in $(printf '%s\n' "${!live_ver[@]}" | sort); do
    [ "$pkg" = "npm" ] && continue
    if [ -z "${want_names[$pkg]:-}" ]; then
      "$RUN" npm uninstall -g "$pkg" || true
    fi
  done
fi

echo "Done."
#+END_SRC

* Makefile fragment for npm
#+BEGIN_SRC makefile :tangle all/.mk/npm.mk :padline no
# ~/DotCortex/all/.mk/npm.mk
.RECIPEPREFIX := |
SHELL := /bin/bash

.PHONY: npm-capture npm-diff npm-sync npm-apply npm-health

npm-capture:
| @chmod +x $(HOME)/.local/bin/npm-capture 2>/dev/null || true
| $(HOME)/.local/bin/npm-capture

npm-diff:
| @chmod +x $(HOME)/.local/bin/npm-diff 2>/dev/null || true
| $(HOME)/.local/bin/npm-diff

# Install anything new from the manifest, do not uninstall extras, do not touch existing versions.
npm-sync:
| @chmod +x $(HOME)/.local/bin/npm-apply 2>/dev/null || true
| ENFORCE=0 UNINSTALL=0 UPDATE=0 $(HOME)/.local/bin/npm-apply

# Strict apply, install and enforce manifest membership, but do not auto upgrade versions.
npm-apply:
| @chmod +x $(HOME)/.local/bin/npm-apply 2>/dev/null || true
| ENFORCE=1 UNINSTALL=1 UPDATE=0 $(HOME)/.local/bin/npm-apply

npm-health:
| @chmod +x $(HOME)/.local/bin/npm-health 2>/dev/null || true
| $(HOME)/.local/bin/npm-health
#+END_SRC
