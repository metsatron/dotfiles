* Home sync helper
Keep icon and theme payloads in DotCortex while making the live home roots real
directories. Icon themes become real directories with repo-backed contents so
only cache files live in =$HOME=. GTK/XFWM theme entries stay repo-backed as
top-level symlinks inside real home theme roots so Stow never tries to replace
the whole tree.

#+BEGIN_SRC python :tangle all/.local/bin/icons-home-sync :shebang "#!/usr/bin/env python3" :tangle-mode (identity #o755)
from __future__ import annotations

import argparse
import shutil
import subprocess
from pathlib import Path


ICON_SYNC_MAP = (
    ("all/.icons", ".icons"),
    ("linux/.icons", ".icons"),
    ("all/.local/share/icons", ".local/share/icons"),
    ("linux/.local/share/icons", ".local/share/icons"),
)

THEME_SYNC_MAP = (
    ("all/.themes", ".themes"),
    ("linux/.themes", ".themes"),
    ("all/.local/share/themes", ".local/share/themes"),
    ("linux/.local/share/themes", ".local/share/themes"),
)

CACHE_NAMES = {"icon-theme.cache", ".icon-theme.cache"}
ROOT_SKIP_NAMES = {".git"}
ICON_SKIP_NAMES = CACHE_NAMES | ROOT_SKIP_NAMES


def remove_path(path: Path) -> None:
    if path.is_dir() and not path.is_symlink():
        shutil.rmtree(path)
    elif path.exists() or path.is_symlink():
        path.unlink()


def ensure_real_dir(path: Path, dry_run: bool) -> None:
    if dry_run:
        if path.is_symlink():
            print(f"plan detach root {path}")
        elif not path.exists():
            print(f"plan mkdir {path}")
        return
    if path.is_symlink():
        path.unlink()
    elif path.exists() and not path.is_dir():
        path.unlink()
    path.mkdir(parents=True, exist_ok=True)


def sync_icon_dir(srcs: list[Path], dst: Path, dry_run: bool) -> None:
    if dry_run:
        if dst.is_symlink():
            print(f"plan detach theme {dst}")
        elif dst.exists() and not dst.is_dir():
            print(f"plan replace non-dir {dst}")
        elif not dst.exists():
            print(f"plan mkdir {dst}")
    else:
        if dst.is_symlink() or (dst.exists() and not dst.is_dir()):
            remove_path(dst)
        dst.mkdir(parents=True, exist_ok=True)

    desired: dict[str, Path] = {}
    for src in srcs:
        for child in sorted(src.iterdir()):
            if child.name in ICON_SKIP_NAMES:
                continue
            desired[child.name] = child

    if dst.exists() and dst.is_dir() and not dst.is_symlink():
        for existing in sorted(dst.iterdir()):
            if existing.name in CACHE_NAMES:
                continue
            if existing.name not in desired:
                if dry_run:
                    print(f"plan remove stale {existing}")
                else:
                    remove_path(existing)

    for name, child in sorted(desired.items()):
        target = dst / name
        child_resolved = child.resolve(strict=False)
        if target.is_symlink() and target.resolve(strict=False) == child_resolved:
            continue
        if dry_run:
            print(f"plan link {target} -> {child}")
            continue
        if target.exists() or target.is_symlink():
            remove_path(target)
        target.symlink_to(child)


def remove_repo_caches(repo_root: Path, dry_run: bool) -> int:
    removed = 0
    seen: set[Path] = set()
    for src_rel, _dst_rel in ICON_SYNC_MAP:
        src_root = repo_root / src_rel
        if not src_root.exists():
            continue
        for cache_name in CACHE_NAMES:
            for cache_path in src_root.rglob(cache_name):
                if cache_path in seen:
                    continue
                seen.add(cache_path)
                if dry_run:
                    print(f"plan remove repo cache {cache_path}")
                else:
                    remove_path(cache_path)
                removed += 1
    return removed


def sync_icon_entry(srcs: list[Path], dst_root: Path, dry_run: bool) -> bool:
    src = srcs[-1]
    dst = dst_root / src.name
    if src.is_dir():
        sync_icon_dir(srcs, dst, dry_run)
    else:
        src_resolved = src.resolve(strict=False)
        if dst.is_symlink() and dst.resolve(strict=False) == src_resolved:
            return False
        if dry_run:
            print(f"plan link {dst} -> {src}")
        else:
            dst.parent.mkdir(parents=True, exist_ok=True)
            if dst.exists() or dst.is_symlink():
                remove_path(dst)
            dst.symlink_to(src)
    return True


def is_under_any(path: Path, roots: list[Path]) -> bool:
    resolved = path.resolve(strict=False)
    for root in roots:
        root_resolved = root.resolve(strict=False)
        try:
            resolved.relative_to(root_resolved)
            return True
        except ValueError:
            continue
    return False


def sync_theme_entries(src_roots: list[Path], dst_root: Path, dry_run: bool) -> int:
    desired: dict[str, Path] = {}
    for src_root in src_roots:
        for src in sorted(src_root.iterdir()):
            if src.name in ROOT_SKIP_NAMES:
                continue
            desired[src.name] = src

    synced = 0
    if dst_root.exists() and dst_root.is_dir() and not dst_root.is_symlink():
        for existing in sorted(dst_root.iterdir()):
            if existing.name in desired:
                continue
            if existing.is_symlink() and is_under_any(existing, src_roots):
                if dry_run:
                    print(f"plan remove stale {existing}")
                else:
                    existing.unlink()

    for name, src in sorted(desired.items()):
        dst = dst_root / name
        src_resolved = src.resolve(strict=False)
        if dst.is_symlink() and dst.resolve(strict=False) == src_resolved:
            continue
        if dry_run:
            action = "relink" if dst.exists() or dst.is_symlink() else "link"
            print(f"plan {action} {dst} -> {src}")
        else:
            if dst.exists() or dst.is_symlink():
                remove_path(dst)
            dst.symlink_to(src)
        synced += 1
    return synced


def rebuild_caches(themes: set[Path], dry_run: bool) -> int:
    gtk_cache = shutil.which("gtk-update-icon-cache")
    if not gtk_cache:
        return 0
    count = 0
    for theme in sorted(themes):
        if not theme.exists() or not theme.is_dir() or theme.is_symlink():
            continue
        if not (theme / "index.theme").is_file():
            continue
        for stale in (theme / ".icon-theme.cache", theme / "icon-theme.cache"):
            if stale.exists() or stale.is_symlink():
                if dry_run:
                    print(f"plan remove stale cache {stale}")
                else:
                    stale.unlink()
        if dry_run:
            print(f"plan cache {theme}")
        else:
            subprocess.run(
                [gtk_cache, "-f", "-t", str(theme)],
                check=False,
                stdout=subprocess.DEVNULL,
                stderr=subprocess.DEVNULL,
            )
        count += 1
    return count


def main() -> int:
    parser = argparse.ArgumentParser(
        description="Keep repo-backed icons and themes while rebuilding only local icon caches in home roots."
    )
    parser.add_argument(
        "--repo-root",
        default=str(Path.home() / "DotCortex"),
        help="DotCortex repo root",
    )
    parser.add_argument("--dry-run", action="store_true", help="show planned actions only")
    args = parser.parse_args()

    repo_root = Path(args.repo_root).expanduser().resolve()
    home = Path.home()
    repo_caches_removed = remove_repo_caches(repo_root, args.dry_run)
    synced = 0
    icon_themes_to_rebuild: set[Path] = set()

    sync_roots: dict[str, list[Path]] = {}
    for src_rel, dst_rel in ICON_SYNC_MAP:
        src_root = repo_root / src_rel
        if not src_root.exists():
            continue
        sync_roots.setdefault(dst_rel, []).append(src_root)

    for dst_rel, src_roots in sync_roots.items():
        dst_root = home / dst_rel
        ensure_real_dir(dst_root, args.dry_run)
        entries: dict[str, list[Path]] = {}
        for src_root in src_roots:
            for src in sorted(src_root.iterdir()):
                if src.name in ICON_SKIP_NAMES:
                    continue
                entries.setdefault(src.name, []).append(src)
        for name in sorted(entries):
            if sync_icon_entry(entries[name], dst_root, args.dry_run):
                synced += 1
                dst = dst_root / name
                if dst.exists() and dst.is_dir() and not dst.is_symlink():
                    icon_themes_to_rebuild.add(dst)

    theme_roots: dict[str, list[Path]] = {}
    for src_rel, dst_rel in THEME_SYNC_MAP:
        src_root = repo_root / src_rel
        if not src_root.exists():
            continue
        theme_roots.setdefault(dst_rel, []).append(src_root)

    for dst_rel, src_roots in theme_roots.items():
        dst_root = home / dst_rel
        ensure_real_dir(dst_root, args.dry_run)
        synced += sync_theme_entries(src_roots, dst_root, args.dry_run)

    cached = rebuild_caches(icon_themes_to_rebuild, args.dry_run)

    mode = "planned" if args.dry_run else "linked"
    print(
        f"icons-home-sync: {mode}_entries={synced} "
        f"repo_caches_removed={repo_caches_removed} cache_roots={cached}"
    )
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
