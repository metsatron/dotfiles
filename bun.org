* Bun manifest
Global Bun packages installed with =bun add -g=.
The manifest lives in a DotCortex-owned data path so =~/.bun= and =~/.bunx=
can stay real Bun homes instead of repo symlinks.
Columns: PKG VERSION FLAGS REGISTRY EXTRA
Leave VERSION empty to treat the currently installed version as canonical.

* Capture current global Bun -> manifest
#+BEGIN_SRC bash :tangle all/.local/bin/bun-capture :shebang "#!/usr/bin/env bash"
# ~/DotCortex/all/.local/bin/bun-capture
set -euo pipefail

MAN="$HOME/DotCortex/all/.local/share/dotcortex/manifests/bun/global.ssv"

mkdir -p "$(dirname "$MAN")"
touch "$MAN"

tmp_existing="$(mktemp)"
tmp_have="$(mktemp)"
tmp_new="$(mktemp)"

awk '!/^[[:space:]]*#/ && NF { print $1 }' "$MAN" | sort -u >"$tmp_existing"

"$RUN" python3 - <<'PY' | sort -u >"$tmp_have"
import json
import os
import sys

root = os.path.expanduser(os.environ.get("BUN_GLOBAL_DIR", "~/.bun/install/global"))
node_modules = os.path.join(root, "node_modules")

def emit(pkg_dir):
    pkg_json = os.path.join(pkg_dir, "package.json")
    if not os.path.isfile(pkg_json):
        return
    try:
        with open(pkg_json, "r", encoding="utf-8") as fh:
            data = json.load(fh)
    except Exception:
        return
    name = data.get("name")
    if name:
        print(name)

if os.path.isdir(node_modules):
    for entry in sorted(os.listdir(node_modules)):
        path = os.path.join(node_modules, entry)
        if entry.startswith("@") and os.path.isdir(path):
            for sub in sorted(os.listdir(path)):
                emit(os.path.join(path, sub))
        elif os.path.isdir(path):
            emit(path)
PY

comm -13 "$tmp_existing" "$tmp_have" >"$tmp_new"

rm -f "$tmp_existing" "$tmp_have"

if [ ! -s "$tmp_new" ]; then
  rm -f "$tmp_new"
  echo "No new Bun globals to capture into $MAN"
  exit 0
fi

while read -r pkg; do
  [ -n "$pkg" ] || continue
  printf '%s "" "" "" ""\n' "$pkg" >>"$MAN"
done <"$tmp_new"

rm -f "$tmp_new"
echo "Updated $MAN"
#+END_SRC

ENFORCE="${ENFORCE:-1}"

echo "== bun apply =="

_emit_norm() {
  awk -v FS='[ \t]+' '
    function dequote(s){ return (s=="\"\"" ? "" : s) }
    $0 ~ /^[ \t]*#/ || NF==0 { next }
    {
      printf "%s|%s|%s|%s|%s\n",
        dequote($1),dequote($2),dequote($3),dequote($4),dequote($5)
    }
  ' "$MAN"
}

declare -A live_ver

while read -r name ver; do
  [ -n "${name:-}" ] || continue
  live_ver["$name"]="$ver"
done < <("$RUN" python3 - <<'PY'
import json
import os

root = os.path.expanduser(os.environ.get("BUN_GLOBAL_DIR", "~/.bun/install/global"))
node_modules = os.path.join(root, "node_modules")

def emit(pkg_dir):
    pkg_json = os.path.join(pkg_dir, "package.json")
    if not os.path.isfile(pkg_json):
        return
    try:
        with open(pkg_json, "r", encoding="utf-8") as fh:
            data = json.load(fh)
    except Exception:
        return
    name = data.get("name")
    version = data.get("version", "")
    if name:
        print(f"{name} {version}")

if os.path.isdir(node_modules):
    for entry in sorted(os.listdir(node_modules)):
        path = os.path.join(node_modules, entry)
        if entry.startswith("@") and os.path.isdir(path):
            for sub in sorted(os.listdir(path)):
                emit(os.path.join(path, sub))
        elif os.path.isdir(path):
            emit(path)
PY
)

declare -A want_names

while IFS='|' read -r pkg version flags registry extra
do
  [ -n "$pkg" ] || continue
  want_names["$pkg"]=1

  installed_ver="${live_ver[$pkg]:-}"
  if [ -n "$installed_ver" ] && [ "$UPDATE" != "1" ]; then
    continue
  fi

  if [ -n "$installed_ver" ] && [ -n "$version" ] && [ "$version" = "$installed_ver" ]; then
    continue
  fi

  spec="$pkg"
  if [ -n "$version" ]; then
    spec="$pkg@$version"
  fi

  args=(add -g "$spec")

  if [ -n "$flags" ]; then
    IFS=',' read -ra FF <<<"$flags"
    for f in "${FF[@]}"; do
      [ -n "$f" ] && args+=("$f")
    done
  fi

  if [ -n "$registry" ]; then
    args+=(--registry "$registry")
  fi

  "$RUN" bun "${args[@]}"
done < <(_emit_norm)

if [ "$ENFORCE" = "1" ] && [ "$UNINSTALL" = "1" ]; then
  echo "== uninstall extras =="
  for pkg in $(printf '%s\n' "${!live_ver[@]}" | sort); do
    if [ -z "${want_names[$pkg]:-}" ]; then
      "$RUN" bun remove -g "$pkg" || true
    fi
  done
fi

if [ -x "$HOME/.local/bin/dotcortex-bun-build-lib" ]; then
  # Source-build rows are processed in the same pass so updates and installs stay sequential.
  # The build fingerprints suppress reruns when the checkout and commands are unchanged.
  source "$HOME/.local/bin/dotcortex-bun-build-lib"

  echo "== bun build pass =="

  seen_build=0
  while IFS=$'\t' read -r name mode root repo ref subdir install_cmd build_cmd health_cmd extra
  do
    [ -n "$name" ] || continue
    if [ -n "$BUN_BUILD_ONLY" ] && [ "$name" != "$BUN_BUILD_ONLY" ]; then
      continue
    fi

    seen_build=1
    bun_build_ensure_checkout "$UPDATE" "$name" "$mode" "$root" "$repo" "$ref"

    root_abs="$(bun_build_root_abs "$root")"
    workdir_abs="$(bun_build_workdir_abs "$root" "$subdir")"
    state_dir="$(bun_build_state_dir "$name")"
    install_stamp="$state_dir/install.fp"
    build_stamp="$state_dir/build.fp"

    install_effective="$(bun_build_effective_install_cmd "$root_abs" "$workdir_abs" "$install_cmd")"
    build_effective="$(bun_build_effective_build_cmd "$build_cmd")"

    if [ -n "$install_effective" ]; then
      install_fp="$(bun_build_install_fingerprint "$root_abs" "$workdir_abs" "$install_effective")"
      if bun_build_fingerprint_matches "$install_stamp" "$install_fp"; then
        echo ">> [$name] install current"
      else
        bun_build_run_step "$name" install "$workdir_abs" "$install_effective"
        bun_build_write_fingerprint "$install_stamp" "$install_fp"
      fi
    else
      echo ">> [$name] install disabled"
    fi

    if [ -n "$build_effective" ]; then
      build_fp="$(bun_build_build_fingerprint "$root_abs" "$workdir_abs" "$install_effective" "$build_effective")"
      if bun_build_fingerprint_matches "$build_stamp" "$build_fp"; then
        echo ">> [$name] build current"
      else
        bun_build_run_step "$name" build "$workdir_abs" "$build_effective"
        bun_build_write_fingerprint "$build_stamp" "$build_fp"
      fi
    else
      echo ">> [$name] build disabled"
    fi
  done < <(bun_build_rows)

  if [ "$seen_build" = "0" ]; then
    echo "No Bun builds defined in $BUN_BUILD_MANIFEST"
  fi
fi

echo "Done."

* Bun build manifest
Use this lane for Bun-managed source trees that need a checkout plus conditional
install/build steps rather than a simple =bun add -g=. These rows are applied
sequentially by =bun:apply= after the global Bun package pass.

- =MODE=submodule= expects =ROOT= to already exist in =.gitmodules=
- =MODE=git= clones or syncs a standalone checkout at =ROOT=
- =MODE=local= uses an already-present tree and never pulls it forward
- =INSTALL=: empty = auto (=bun install --frozen-lockfile= when a Bun lockfile exists,
  else =bun install=); =none= disables dependency install
- =BUILD=: empty or =none= skips builds
- =HEALTH=: optional command, only executed when =RUN_CHECKS=1 bun-health=
- Quote commands that contain spaces

# ~/DotCortex/all/.local/bin/dotcortex-bun-build-lib
set -euo pipefail

DOTCORTEX_ROOT="${DOTCORTEX_ROOT:-$HOME/DotCortex}"
BUN_BUILD_MANIFEST="${BUN_BUILD_MANIFEST:-$DOTCORTEX_ROOT/all/.local/share/dotcortex/manifests/bun/build.ssv}"
BUN_BUILD_STATE_ROOT="${BUN_BUILD_STATE_ROOT:-${XDG_STATE_HOME:-$HOME/.local/state}/dotcortex/bun-build}"

DOTCORTEX_BUN_ENV="${DOTCORTEX_BUN_ENV:-$HOME/.local/bin/dotcortex-bun-env}"

sha256_file() {
  local path="$1"

  if command -v sha256sum >/dev/null 2>&1; then
    sha256sum "$path" | awk '{print $1}'
  else
    shasum -a 256 "$path" | awk '{print $1}'
  fi
}

sha256_stdin() {
  if command -v sha256sum >/dev/null 2>&1; then
    sha256sum | awk '{print $1}'
  else
    shasum -a 256 | awk '{print $1}'
  fi
}

bun_build_rows() {
  python3 - "$BUN_BUILD_MANIFEST" <<'PY'
import os
import shlex
import sys

manifest = os.path.expanduser(sys.argv[1])
if not os.path.exists(manifest):
    raise SystemExit(0)

with open(manifest, encoding="utf-8") as fh:
    for lineno, raw in enumerate(fh, 1):
        line = raw.strip()
        if not line or line.startswith("#"):
            continue
        fields = shlex.split(line, posix=True)
        if len(fields) != 9:
            sys.stderr.write(f"{manifest}:{lineno}: expected 9 fields, got {len(fields)}\n")
            raise SystemExit(1)
        sys.stdout.write("\t".join(fields) + "\n")
PY
}

bun_build_safe_name() {
  printf '%s\n' "$1" | tr '/: ' '___'
}

bun_build_state_dir() {
  printf '%s/%s\n' "$BUN_BUILD_STATE_ROOT" "$(bun_build_safe_name "$1")"
}

bun_build_root_abs() {
  local root="$1"

  case "$root" in
    /*)
      printf '%s\n' "$root"
      ;;
    ~*)
      printf '%s\n' "${root/#\~/$HOME}"
      ;;
    *)
      printf '%s\n' "$DOTCORTEX_ROOT/$root"
      ;;
  esac
}

bun_build_workdir_abs() {
  local root="$1" subdir="$2" root_abs

  root_abs="$(bun_build_root_abs "$root")"
  if [ -z "$subdir" ] || [ "$subdir" = "." ]; then
    printf '%s\n' "$root_abs"
  else
    printf '%s\n' "$root_abs/$subdir"
  fi
}

bun_build_bun_version_or_missing() {
  local bun_home bin_path

  if command -v bun >/dev/null 2>&1; then
    bun --version
    return 0
  fi

  bun_home="${BUN_INSTALL:-$HOME/.bun}"
  bin_path="$bun_home/bin/bun"
  if [ -x "$bin_path" ]; then
    "$bin_path" --version
  else
    printf 'missing\n'
  fi
}

bun_build_hash_existing_files() {
  local path
  declare -A seen=()

  for path in "$@"; do
    [ -n "$path" ] || continue
    if [ -n "${seen[$path]:-}" ]; then
      continue
    fi
    seen["$path"]=1
    if [ -f "$path" ]; then
      printf '%s=%s\n' "$path" "$(sha256_file "$path")"
    fi
  done
}

bun_build_effective_install_cmd() {
  local root_abs="$1" workdir_abs="$2" install_cmd="$3"

  case "$install_cmd" in
    none)
      printf '%s\n' ""
      ;;
    "")
      if [ -f "$root_abs/bun.lock" ] || [ -f "$root_abs/bun.lockb" ] || [ -f "$workdir_abs/bun.lock" ] || [ -f "$workdir_abs/bun.lockb" ]; then
        printf '%s\n' 'bun install --frozen-lockfile'
      else
        printf '%s\n' 'bun install'
      fi
      ;;
    *)
      printf '%s\n' "$install_cmd"
      ;;
  esac
}

bun_build_effective_build_cmd() {
  local build_cmd="$1"

  case "$build_cmd" in
    ""|none)
      printf '%s\n' ""
      ;;
    *)
      printf '%s\n' "$build_cmd"
      ;;
  esac
}

bun_build_effective_health_cmd() {
  local health_cmd="$1"

  case "$health_cmd" in
    ""|none)
      printf '%s\n' ""
      ;;
    *)
      printf '%s\n' "$health_cmd"
      ;;
  esac
}

bun_build_is_git_repo() {
  local root_abs="$1"
  git -C "$root_abs" rev-parse --is-inside-work-tree >/dev/null 2>&1
}

bun_build_source_fingerprint() {
  local root_abs="$1" status head worktree_diff cached_diff submodules

  if [ ! -d "$root_abs" ]; then
    printf '%s\n' 'missing'
    return 0
  fi

  if bun_build_is_git_repo "$root_abs"; then
    head="$(git -C "$root_abs" rev-parse HEAD 2>/dev/null || printf 'unknown')"
    status="$(git -C "$root_abs" status --porcelain=v1 --untracked-files=normal 2>/dev/null || true)"
    submodules="$(git -C "$root_abs" submodule status --recursive 2>/dev/null || true)"
    worktree_diff=""
    cached_diff=""

    if [ -n "$status" ]; then
      worktree_diff="$(git -C "$root_abs" diff --no-ext-diff HEAD 2>/dev/null | sha256_stdin)"
      cached_diff="$(git -C "$root_abs" diff --cached --no-ext-diff 2>/dev/null | sha256_stdin)"
    fi

    {
      printf 'head=%s\n' "$head"
      printf 'status=%s\n' "$status"
      printf 'worktree_diff=%s\n' "$worktree_diff"
      printf 'cached_diff=%s\n' "$cached_diff"
      printf 'submodules=%s\n' "$submodules"
    } | sha256_stdin
    return 0
  fi

  {
    printf 'root=%s\n' "$root_abs"
      bun_build_hash_existing_files \
      "$root_abs/package.json" \
      "$root_abs/bun.lock" \
      "$root_abs/bun.lockb" \
      "$root_abs/bunfig.toml"
  } | sha256_stdin
}

bun_build_install_fingerprint() {
  local root_abs="$1" workdir_abs="$2" install_cmd="$3"

  {
    printf 'bun=%s\n' "$(bun_build_bun_version_or_missing)"
    printf 'system=%s|%s\n' "$(uname -s)" "$(uname -m)"
    printf 'install=%s\n' "$install_cmd"
    bun_build_hash_existing_files \
      "$root_abs/package.json" \
      "$root_abs/bun.lock" \
      "$root_abs/bun.lockb" \
      "$root_abs/bunfig.toml" \
      "$workdir_abs/package.json" \
      "$workdir_abs/bun.lock" \
      "$workdir_abs/bun.lockb" \
      "$workdir_abs/bunfig.toml"
  } | sha256_stdin
}

bun_build_build_fingerprint() {
  local root_abs="$1" workdir_abs="$2" install_cmd="$3" build_cmd="$4"

  {
    printf 'build=%s\n' "$build_cmd"
    printf 'source=%s\n' "$(bun_build_source_fingerprint "$root_abs")"
    printf 'install=%s\n' "$(bun_build_install_fingerprint "$root_abs" "$workdir_abs" "$install_cmd")"
  } | sha256_stdin
}

bun_build_fingerprint_matches() {
  local stamp_path="$1" fingerprint="$2"

  [ -f "$stamp_path" ] || return 1
  [ "$(<"$stamp_path")" = "$fingerprint" ]
}

bun_build_write_fingerprint() {
  local stamp_path="$1" fingerprint="$2"

  mkdir -p "$(dirname "$stamp_path")"
  printf '%s\n' "$fingerprint" >"$stamp_path"
}

bun_build_submodule_declared() {
  local root="$1" gitmodules="$DOTCORTEX_ROOT/.gitmodules"

  [ -f "$gitmodules" ] || return 1
  git -C "$DOTCORTEX_ROOT" config -f "$gitmodules" --get-regexp '^submodule\..*\.path$' 2>/dev/null \
    | awk '{print $2}' \
    | grep -Fxq "$root"
}

bun_build_ensure_checkout() {
  local update="$1" name="$2" mode="$3" root="$4" repo="$5" ref="$6"
  local root_abs parent origin branch

  root_abs="$(bun_build_root_abs "$root")"

  case "$mode" in
    submodule)
      if [[ "$root" = /* || "$root" = ~* ]]; then
        echo "!! [$name] submodule ROOT must be repo-relative: $root" >&2
        return 1
      fi
      if ! bun_build_submodule_declared "$root"; then
        echo "!! [$name] submodule path is not declared in $DOTCORTEX_ROOT/.gitmodules: $root" >&2
        return 1
      fi
      echo ">> [$name] sync submodule: $root"
      git -C "$DOTCORTEX_ROOT" submodule update --init --recursive -- "$root"
      if [ "$update" = "1" ]; then
        echo ">> [$name] update submodule from remote"
        git -C "$DOTCORTEX_ROOT" submodule update --remote --recursive -- "$root"
      fi
      if [ -n "$repo" ]; then
        origin="$(git -C "$root_abs" remote get-url origin 2>/dev/null || true)"
        if [ -n "$origin" ] && [ "$origin" != "$repo" ]; then
          echo "!! [$name] submodule origin mismatch: expected $repo, got $origin" >&2
          return 1
        fi
      fi
      ;;

    git)
      parent="$(dirname "$root_abs")"
      mkdir -p "$parent"
      if [ ! -d "$root_abs/.git" ]; then
        if [ -z "$repo" ]; then
          echo "!! [$name] MODE=git requires REPO" >&2
          return 1
        fi
        echo ">> [$name] clone git source: $repo -> $root_abs"
        git clone "$repo" "$root_abs"
      fi

      if ! bun_build_is_git_repo "$root_abs"; then
        echo "!! [$name] expected a git checkout at $root_abs" >&2
        return 1
      fi

      if [ -n "$repo" ]; then
        origin="$(git -C "$root_abs" remote get-url origin 2>/dev/null || true)"
        if [ -z "$origin" ]; then
          git -C "$root_abs" remote add origin "$repo"
        elif [ "$origin" != "$repo" ]; then
          echo "!! [$name] git origin mismatch: expected $repo, got $origin" >&2
          return 1
        fi
      fi

      if [ "$update" = "1" ]; then
        echo ">> [$name] fetch latest git refs"
        git -C "$root_abs" fetch --all --tags --prune
      fi

      if [ -n "$ref" ]; then
        git -C "$root_abs" checkout "$ref"
        if [ "$update" = "1" ] && git -C "$root_abs" rev-parse --verify "origin/$ref" >/dev/null 2>&1; then
          git -C "$root_abs" merge --ff-only "origin/$ref"
        fi
      elif [ "$update" = "1" ]; then
        branch="$(git -C "$root_abs" symbolic-ref --quiet --short HEAD 2>/dev/null || true)"
        if [ -n "$branch" ]; then
          git -C "$root_abs" pull --ff-only origin "$branch"
        fi
      fi
      ;;

    local)
      if [ ! -d "$root_abs" ]; then
        echo "!! [$name] local source tree missing: $root_abs" >&2
        return 1
      fi
      ;;

    *)
      echo "!! [$name] unsupported MODE: $mode" >&2
      return 1
      ;;
  esac
}

bun_build_run_step() {
  local name="$1" phase="$2" workdir_abs="$3" cmd="$4"

  [ -n "$cmd" ] || return 0

  if [ ! -d "$workdir_abs" ]; then
    echo "!! [$name] workdir missing for $phase: $workdir_abs" >&2
    return 1
  fi

  echo ">> [$name] $phase: $cmd"
  "$DOTCORTEX_BUN_ENV" bash -lc 'cd "$1" && eval "$2"' bash "$workdir_abs" "$cmd"
}

_emit_norm() {
  awk -v FS='[ \t]+' '
    function dequote(s){
      if (s == "\"\"") return ""
      if (s ~ /^".*"$/) {
        sub(/^"/, "", s)
        sub(/"$/, "", s)
      }
      return s
    }
    $0 ~ /^[ \t]*#/ || NF==0 { next }
    {
      printf "%s|%s|%s|%s|%s|%s\n",
        dequote($1),dequote($2),dequote($3),dequote($4),dequote($5),dequote($6)
    }
  ' "$MAN"
}

declare -A want_names

while IFS='|' read -r name pkg exe version flags extra
do
  [ -n "$name" ] || continue
  want_names["$name"]=1

  pkg="${pkg:-$name}"
  exe="${exe:-$name}"
  pkgspec="$pkg"
  [ -n "$version" ] && pkgspec="$pkg@$version"

  target="$BIN_DIR/$name"
  tmp="$(mktemp)"

  cat >"$tmp" <<EOF
#!/usr/bin/env bash
# dotcortex:bunx-managed
# name: $name
# package: $pkg
# executable: $exe
# version: $version
# flags: $flags

export PATH="\$BUN_INSTALL/bin:\$PATH"
args=()
pkgspec='$pkgspec'
exe='$exe'
flags_csv='$flags'

  done

exec bunx "\${args[@]}" -p "\$pkgspec" "\$exe" "\$@"
EOF

  mv "$tmp" "$target"
  echo ">> wrote $target"

* Detach live Bun homes from DotCortex
#+BEGIN_SRC python :tangle all/.local/bin/bun-home-detach :shebang "#!/usr/bin/env python3" :tangle-mode (identity #o755)
from __future__ import annotations

import argparse
import shutil
import sys
from pathlib import Path


SKIP = {"manifest"}


def copy_entry(src: Path, dst: Path) -> None:
    if src.is_dir() and not src.is_symlink():
        shutil.copytree(src, dst, symlinks=True, dirs_exist_ok=True)
    else:
        dst.parent.mkdir(parents=True, exist_ok=True)
        if dst.exists() or dst.is_symlink():
            if dst.is_dir() and not dst.is_symlink():
                shutil.rmtree(dst)
            else:
                dst.unlink()
        shutil.copy2(src, dst, follow_symlinks=False)


def remove_entry(path: Path) -> None:
    if path.is_dir() and not path.is_symlink():
        shutil.rmtree(path)
    elif path.exists() or path.is_symlink():
        path.unlink()


def detach_pair(repo_home: Path, home: Path, keep_repo_copy: bool) -> int:
    if not repo_home.exists():
        return 0
    migrated = 0
    if home.is_symlink():
        home.unlink()
    home.mkdir(parents=True, exist_ok=True)
    for src in sorted(repo_home.iterdir()):
        if src.name in SKIP:
            continue
        copy_entry(src, home / src.name)
        migrated += 1
        if not keep_repo_copy:
            remove_entry(src)
    return migrated


def main() -> int:
    parser = argparse.ArgumentParser(
        description="Detach ~/.bun and ~/.bunx from DotCortex and keep live Bun state in $HOME."
    )
    parser.add_argument(
        "--repo-bun",
        default=str(Path.home() / "DotCortex/all/.bun"),
        help="repo-owned bun path to migrate away from",
    )
    parser.add_argument(
        "--repo-bunx",
        default=str(Path.home() / "DotCortex/all/.bunx"),
        help="repo-owned bunx path to migrate away from",
    )
    parser.add_argument(
        "--bun-home",
        default=str(Path.home() / ".bun"),
        help="real bun home destination",
    )
    parser.add_argument(
        "--bunx-home",
        default=str(Path.home() / ".bunx"),
        help="real bunx home destination",
    )
    parser.add_argument(
        "--keep-repo-copy",
        action="store_true",
        help="do not remove migrated live state from the repo paths",
    )
    args = parser.parse_args()

    migrated_bun = detach_pair(
        Path(args.repo_bun).expanduser().resolve(),
        Path(args.bun_home).expanduser(),
        args.keep_repo_copy,
    )
    migrated_bunx = detach_pair(
        Path(args.repo_bunx).expanduser().resolve(),
        Path(args.bunx_home).expanduser(),
        args.keep_repo_copy,
    )

    print(
        "detached bun homes: "
        f"bun={migrated_bun} bunx={migrated_bunx} "
        f"homes={Path(args.bun_home).expanduser()},{Path(args.bunx_home).expanduser()}"
    )
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
