* Mount helpers

Shared helper function sourced by all backup scripts to check mount
points directly (no symlink indirection).

#+BEGIN_SRC bash :tangle x230/.local/bin/backup-mount-check :shebang "#!/usr/bin/env bash" :mkdirp yes
# Shared mount-point checker for backup scripts.
# Source this file, then call: require_mount /mnt/ironwolf01 [/mnt/ironwolf02 ...]

require_mount() {
  for mp in "$@"; do
    if ! mountpoint -q "$mp" 2>/dev/null; then
      echo "ERROR: $mp is not mounted. Aborting." >&2
      exit 1
    fi
  done
}

* backup-system – System mirror to ironwolf02

Mirrors =/= to =/mnt/ironwolf02/backup-xubuntu/backup-xubuntu/= using
rsync with filter rules. Guard file prevents writing to wrong target.

#+BEGIN_SRC bash :tangle x230/.local/bin/backup-system :shebang "#!/usr/bin/env bash" :mkdirp yes
set -Eeuo pipefail

# ===== CONFIG =====
DEST_BASE="/mnt/ironwolf02/backup-xubuntu"
DEST="${DEST_BASE}/backup-xubuntu"               # final mirror dir
GUARD_FILE=""
OWNER_FILE=""
LOG="/tmp/backup_system_$(date +%Y%m%d_%H%M%S).log"

# Resolve the real backup owner even when the script is launched via sudo.
BACKUP_USER="${SUDO_USER:-$(id -un)}"
BACKUP_GROUP="${SUDO_GID:+$(getent group "$SUDO_GID" | cut -d: -f1)}"
BACKUP_GROUP="${BACKUP_GROUP:-$(id -gn "$BACKUP_USER")}"
USER_HOME="$(getent passwd "$BACKUP_USER" | cut -d: -f6)"
USER_HOME="${USER_HOME:-$HOME}"

refresh_dest_paths() {
  local base="${DEST%/}"
  GUARD_FILE="${base}/.backup_guard"
  OWNER_FILE="${base}/.backup_owner"
}
refresh_dest_paths

# Filters: use env FILTERS_FILE if provided, else your user default
FILTERS_FILE="${FILTERS_FILE:-$USER_HOME/.config/backup/system-filters.rsync}"

# Keep backup trees owned by the real user even when rsync runs as root.
RSYNC_OWNERSHIP=()
if [ -n "${SUDO_USER:-}" ]; then
  RSYNC_OWNERSHIP=( "--chown=${BACKUP_USER}:${BACKUP_GROUP}" )
fi

# Default rsync arg arrays (real vs dry)
RSYNC_RUN=( -aAX --delete --numeric-ids --info=stats2,progress2 --no-owner --no-group "${RSYNC_OWNERSHIP[@]}" --filter="merge $FILTERS_FILE" )
RSYNC_DRY=( -aAXn --delete --itemize-changes --numeric-ids --no-owner --no-group "${RSYNC_OWNERSHIP[@]}" --filter="merge $FILTERS_FILE" )

# ===== ARGS =====
MODE="" AUTO=0
while (($#)); do
  case "$1" in
    -y|--yes) MODE="yes";;
    -n|--dry|--dry-run) MODE="dry";;
    --auto) AUTO=1;;
    --dest) DEST="${2:?}"; shift;;                       # optional override
    --filters|--filter-file) FILTERS_FILE="${2:?}"; shift;;
    -h|--help)
      cat <<EOF
Usage: $(basename "$0") [-n|--dry-run | -y|--yes] [--auto]
       [--dest PATH] [--filters FILE]
Mirrors "/" into ${DEST} using rsync + your filter rules.
Guards:
  * Requires /mnt/ironwolf02 mounted; DEST must be under backup-xubuntu
  * Must-type DEST confirmation (unless -n|-y provided)
  * Creates/uses ${GUARD_FILE} sentinel
EOF
      exit 0;;
    *) echo "Unknown option: $1" >&2; exit 1;;
  esac; shift
done
refresh_dest_paths

# ===== Utils =====
die(){ echo "ERROR: $*" >&2; exit 1; }
hr(){ command -v numfmt >/dev/null 2>&1 && numfmt --to=iec --suffix=B "$1" || echo "$1 B"; }
avail_bytes(){ df -P -B1 "$1" | awk 'NR==2{print $4}'; }
canon(){ readlink -f -- "$1" 2>/dev/null || printf '%s' "$1"; }
dest_writable() {
  local path="$1"
  if [ -n "${SUDO_USER:-}" ]; then
    sudo -u "$BACKUP_USER" test -w "$path"
  else
    [ -w "$path" ]
  fi
}
repair_dest_ownership() {
  local path="$1" want="${BACKUP_USER}:${BACKUP_GROUP}"
  [ -n "${SUDO_USER:-}" ] || return 0
  if [ -f "$OWNER_FILE" ] && [ "$(cat "$OWNER_FILE" 2>/dev/null || true)" = "$want" ]; then
    chown "$want" "$path" "$GUARD_FILE" "$OWNER_FILE" 2>/dev/null || true
    return 0
  fi
  echo "[*] Repairing destination ownership to $want"
  chown -R "$want" "$path"
  printf '%s\n' "$want" > "$OWNER_FILE"
  chown "$want" "$OWNER_FILE"
  chmod 644 "$OWNER_FILE"
}

ensure_filters() {
  [ -f "$FILTERS_FILE" ] || die "Missing filters file: $FILTERS_FILE"
  [ -r "$FILTERS_FILE" ] || die "Filters not readable: $FILTERS_FILE"
}

dest_sanity() {
  # Mount check — direct, no symlinks
  mountpoint -q /mnt/ironwolf02 || die "/mnt/ironwolf02 not mounted"

  # DEST must live under backup-xubuntu, not be the mount root or /
  local cdest; cdest="$(canon "$DEST")"
  [[ "$cdest" == /mnt/ironwolf02/backup-xubuntu/* ]] || die "Refusing DEST outside /mnt/ironwolf02/backup-xubuntu: $cdest"
  [[ "$cdest" != "/" && "$cdest" != "/mnt/ironwolf02" && "$cdest" != "/mnt/ironwolf02/" ]] || die "Refusing dangerous DEST: $cdest"

  mkdir -p "$cdest"
  DEST="$(canon "$cdest")"
  refresh_dest_paths

  # Guard/sentinel: ensure this target *is* a system mirror target
  if [ ! -f "$GUARD_FILE" ]; then
    {
      echo "system mirror target (rsync from /)"
      echo "DEST=$(canon "$cdest")"
      echo "FSTYPE=$(findmnt -no FSTYPE --target "$cdest")"
      echo "DEV=$(findmnt -no SOURCE --target "$cdest")"
      echo "CREATED=$(date -Is)"
    } > "$GUARD_FILE"
    chmod 644 "$GUARD_FILE"
  else
    grep -q "system mirror target (rsync from /)" "$GUARD_FILE" \
      || die "Guard file exists but does not look like a system mirror target: $GUARD_FILE"
  fi

  repair_dest_ownership "$DEST"

  # Ensure writable as your user (so non-root runs don't silently fail)
  dest_writable "$DEST" || die "Destination not writable by $BACKUP_USER. Try: sudo chown -R $BACKUP_USER:$BACKUP_GROUP '$DEST'"
}

stats() {
  # Estimate with the SAME filters
  rsync -aAXn --delete --stats --numeric-ids --no-owner --no-group \
    --filter="merge $FILTERS_FILE" / "${DEST%/}/" \
  | awk -F': *' '
      /Total file size:/ {gsub(/,| bytes/,"",$2); t=$2}
      /Total transferred file size:/ {gsub(/,| bytes/,"",$2); x=$2}
      END {print t? t:0, x? x:0}'
}

prompt() {
  local d dev; d="$(canon "$DEST")"
  dev="$(findmnt -no SOURCE,TARGET,FSTYPE --target "$d" 2>/dev/null || true)"
  read -r t x < <(stats); f="$(avail_bytes "$d")"

  echo "================ SYSTEM MIRROR ================"
  echo "SRC : /"
  echo "DEST: $d  ($dev)"
  echo "CMD : rsync ${RSYNC_RUN[*]} / \"$d/\""
  echo "Size total=$(hr "$t")  xfer~=$(hr "$x")  free=$(hr "$f")"
  echo "Filters: $FILTERS_FILE"
  echo "=============================================="

  # If mode preselected, skip asking.
  case "${MODE:-}" in
    dry) echo "[DRY MODE] Skipping confirmation."; return ;;
    yes) echo "[AUTO YES] Proceeding without prompt."; return ;;
  esac
  (( AUTO )) && { [ -n "$MODE" ] || die "--auto requires -n or -y"; return; }

  # Must-type confirmation
  read -rp "Type EXACT DEST path to confirm, or 'd' for dry-run, or 'n' to abort: " ans
  case "$ans" in
    n|N) exit 0;;
    d|D) MODE="dry";;
    *)   [[ "$(canon "$ans")" = "$d" ]] || die "Confirmation mismatch"; MODE="yes";;
  esac
}

run() {
  local d; d="$(canon "$DEST")"
  if [ "${MODE:-}" = "dry" ]; then
    rsync "${RSYNC_DRY[@]}" / "$d/" | tee "$LOG"
  else
    ionice -c2 -n7 nice -n 19 rsync "${RSYNC_RUN[@]}" / "$d/" | tee "$LOG"
  fi
  echo "Log: $LOG"
}

main() {
  ensure_filters
  dest_sanity
  prompt
  run
}
main "$@"
#+END_SRC

* backup-system-clean – Prune system mirror

Removes cached / excluded files from the system mirror on ironwolf02
using a cleanup filter list.

#+BEGIN_SRC bash :tangle x230/.local/bin/backup-system-clean :shebang "#!/usr/bin/env bash" :mkdirp yes
set -Eeuo pipefail

ROOT_DEFAULT="/mnt/ironwolf02/backup-xubuntu/backup-xubuntu"
USER_HOME="${SUDO_USER:+$(getent passwd "$SUDO_USER" | cut -d: -f6)}"
USER_HOME="${USER_HOME:-$HOME}"
CLEAN_FILTER_DEFAULT="$USER_HOME/.config/backup/system-clean.rsync"
CLEAN_FILTER="${CLEAN_FILTER:-$CLEAN_FILTER_DEFAULT}"
LOG="/tmp/backup_system_clean_$(date +%Y%m%d_%H%M%S).log"

MODE="" AUTO=0 ROOT="$ROOT_DEFAULT" DO_RM=0

while (($#)); do
  case "$1" in
    -y|--yes) MODE="yes";;
    -n|--dry|--dry-run) MODE="dry";;
    --auto) AUTO=1;;
    --root) ROOT="${2:?}"; shift;;
    --rm) DO_RM=1;;
    --filter) CLEAN_FILTER="${2:?}"; shift;;
    -h|--help)
      cat <<EOF
Usage: $(basename "$0") [-n|--dry-run|-y|--yes] [--auto]
       [--root PATH] [--filter FILE] [--rm]
  --root PATH    Snapshot root to clean (default: $ROOT_DEFAULT)
  --filter FILE  Cleanup filter file (default: $CLEAN_FILTER_DEFAULT)
  --rm           Permanently delete instead of 'gio trash'
EOF
      exit 0;;
    *) echo "Unknown: $1" >&2; exit 1;;
  esac; shift
done

die(){ echo "ERROR: $*" >&2; exit 1; }
hr(){ command -v numfmt >/dev/null 2>&1 && numfmt --to=iec --suffix=B "$1" || echo "$1 B"; }
need(){ command -v "$1" >/dev/null 2>&1 || die "Missing dependency: $1"; }

trash_bucket() {
  local mount="/mnt/ironwolf02"
  local uid; uid="$(id -u)"
  if [[ -d "$mount/.Trash-$uid" ]]; then
    echo "$mount/.Trash-$uid"
  elif [[ -d "$mount/.Trash/$uid" ]]; then
    echo "$mount/.Trash/$uid"
  else
    echo "(no trash dir found on $mount)"
  fi
}

safety() {
  if [ ! -d "$ROOT" ]; then
    echo "SKIP: $ROOT does not exist yet (first system backup not run?)"
    exit 0
  fi
  [[ "$ROOT" == /mnt/ironwolf02/* ]] || die "Refusing to run outside /mnt/ironwolf02"
  mountpoint -q /mnt/ironwolf02 || die "/mnt/ironwolf02 not mounted"
  [ -f "$CLEAN_FILTER" ] || die "Missing cleanup filter: $CLEAN_FILTER"
  [ -r "$CLEAN_FILTER" ] || die "Cleanup filter not readable"
  (( DO_RM )) || need gio
}

# Build candidate list by expanding each '- pattern' as a glob under $ROOT.
build_candidates() {
  local tmp=/tmp/clean_candidates_all_$$.txt
  : > "$tmp"
  shopt -s nullglob dotglob globstar

  while IFS= read -r line; do
    [[ "$line" =~ ^[[:space:]]*# ]] && continue
    [[ "$line" =~ ^[[:space:]]*$ ]] && continue
    local pat="${line#*- }"; pat="${pat#"${pat%%[![:space:]]*}"}"
    [[ -z "$pat" ]] && continue
    local glob
    if [[ "$pat" == /* ]]; then
      glob="$ROOT$pat"
    else
      glob="$ROOT/$pat"
    fi
    while IFS= read -r m; do
      [[ -z "$m" ]] && continue
      [[ "$m" == "$ROOT" || "$m" != "$ROOT"* ]] && continue
      printf '%s\n' "$m" >> "$tmp"
    done < <(compgen -G "$glob" || true)
  done < "$CLEAN_FILTER"

  sort -u "$tmp" |
  awk '{
    paths[NR]=$0; lens[NR]=length($0)
  } END {
    n=NR
    for(i=1;i<=n;i++) idx[i]=i
    for(i=1;i<=n;i++) for(j=i+1;j<=n;j++) if(lens[idx[i]]>lens[idx[j]]) {t=idx[i]; idx[i]=idx[j]; idx[j]=t}
    keepc=0
    for(k=1;k<=n;k++){
      p=paths[idx[k]]
      keep=1
      for(m=1;m<=keepc;m++){
        if (p==keepv[m] || index(p, keepv[m] "/")==1){ keep=0; break }
      }
      if(keep){ keepv[++keepc]=p }
    }
    for(m=1;m<=keepc;m++) print keepv[m]
  }' > /tmp/clean_candidates_$$.txt

  tr '\n' '\0' < /tmp/clean_candidates_$$.txt > /tmp/clean_candidates_$$.nul
}

summary() {
  local total=0 count=0 path sz
  while IFS= read -r -d '' path; do
    if [ -d "$path" ]; then
      sz=$(du -sb --apparent-size -- "$path" 2>/dev/null | awk '{print $1}')
    else
      sz=$(stat -c %s -- "$path" 2>/dev/null || echo 0)
    fi
    total=$(( total + sz ))
    count=$(( count + 1 ))
  done < /tmp/clean_candidates_$$.nul
  echo "$count" "$total"
}

prompt() {
  canon() { printf '%s' "$1" | sed 's:/*$::'; }
  local dev; dev="$(findmnt -no SOURCE,TARGET,FSTYPE --target "$ROOT" 2>/dev/null || true)"
  local count bytes; read -r count bytes < <(summary)
  echo "================ CLEAN PREVIEW ================"
  echo "ROOT : $ROOT  ($dev)"
  echo "RULE : $CLEAN_FILTER"
  echo "Would purge: $count items  reclaim ~$(hr "$bytes")"
  if (( DO_RM )); then
    echo "Delete mode : PERMANENT (no Trash)"
  else
    echo "Trash bucket: $(trash_bucket)"
  fi
  echo "Sample:"
  tr '\0' '\n' < /tmp/clean_candidates_$$.nul | head -n 30 | sed 's/^/  - /'
  echo "=============================================="

  case "${MODE:-}" in
    dry) echo "[DRY MODE] Skipping confirmation."; return ;;
    yes) echo "[AUTO YES] Proceeding without prompt."; return ;;
  esac

  if (( AUTO )); then
    [ -n "$MODE" ] || die "--auto requires -n or -y"
    return
  fi

  read -rp "Type EXACT ROOT path to confirm, or 'd' dry, 'n' abort: " ans
  case "$ans" in
    n|N) exit 0;;
    d|D) MODE="dry";;
    *)
      [ "$(canon "$ans")" = "$(canon "$ROOT")" ] || die "Confirmation mismatch"
      MODE="yes"
      ;;
  esac
}

do_clean() {
  if [ "$MODE" = "dry" ]; then
    echo "[DRY] Nothing deleted. See list above."
    return
  fi
  local path ok=0 fail=0
  while IFS= read -r -d '' path; do
    if (( DO_RM )); then
      if rm -rf --one-file-system -- "$path"; then ok=$((ok+1)); else fail=$((fail+1)); fi
    else
      if gio trash -- "$path"; then ok=$((ok+1)); else fail=$((fail+1)); fi
    fi
  done < /tmp/clean_candidates_$$.nul
  echo "[✓] Clean finished. OK=$ok FAIL=$fail"
}

main() {
  safety
  build_candidates
  prompt
  do_clean
  [ -n "${LOG:-}" ] && tr '\0' '\n' < /tmp/clean_candidates_$$.nul > "$LOG" && echo "Log: $LOG"
  rm -f /tmp/clean_candidates_$$.nul /tmp/clean_candidates_$$.txt /tmp/clean_candidates_all_$$.txt
  echo "Log: $LOG"
}
main "$@"
#+END_SRC

* backup-dotcortex (T480s) – Push DotCortex to X230 via rsync+SSH

Runs on T480s. Pushes =~/DotCortex= over SSH to X230's ironwolf02 backup volume,
creating a timestamped snapshot: =DotCortex-T480s-{TIMESTAMP}/=.
This helper stays source-owned here so the T480s checkout can tangle it back
whenever the overlay is refreshed.

Requires =x230= to be reachable (SSH alias in =~/.ssh/config=) and
=/mnt/ironwolf02/backup-dotfiles= to be mounted on X230.

#+BEGIN_SRC bash :tangle t480s/.local/bin/backup-dotcortex :shebang "#!/usr/bin/env bash" :mkdirp yes
set -Eeuo pipefail

SRC="$HOME/DotCortex"
REMOTE_HOST="x230"
REMOTE_BACKUP_DIR="/mnt/ironwolf02/backup-dotfiles"
TIMESTAMP="$(date +%Y%m%d-%H%M%S)"
REMOTE_DEST="${REMOTE_BACKUP_DIR}/DotCortex-T480s-${TIMESTAMP}"

MODE="" AUTO=0
while (($#)); do
  case "$1" in
    -y|--yes) MODE="yes";;
    -n|--dry|--dry-run) MODE="dry";;
    --auto) AUTO=1;;
    -h|--help)
      echo "Usage: $(basename "$0") [-n|--dry-run] [-y|--yes] [--auto]"
      echo "Pushes ~/DotCortex via rsync+ssh to ${REMOTE_HOST}:${REMOTE_DEST}/"
      exit 0;;
    *) echo "Unknown: $1" >&2; exit 1;;
  esac; shift
done

die(){ echo "ERROR: $*" >&2; exit 1; }

[ -d "$SRC/.git" ] || die "Not a git repo: $SRC"

if (( AUTO )); then
  [ -n "$MODE" ] || die "--auto requires -n or -y"

  echo "SRC : $SRC"
  echo "DEST: ${REMOTE_HOST}:${REMOTE_DEST}"

  case "$ans" in
    y|Y) MODE="yes";;
    d|D) MODE="dry";;
    *)   exit 0;;
  esac
fi

if [ "$MODE" = "dry" ]; then
  echo "[dry] Checking connectivity to ${REMOTE_HOST}..."
  ssh -o BatchMode=yes -o ConnectTimeout=5 "$REMOTE_HOST" true \
    || die "${REMOTE_HOST} unreachable"
  echo "[dry] Would create: ${REMOTE_HOST}:${REMOTE_DEST}"
  echo "[dry] Would run:    rsync -aH --no-owner --no-group $SRC/ ${REMOTE_HOST}:${REMOTE_DEST}/"
  echo "(dry run — no files written)"
  exit 0
fi

ssh -o BatchMode=yes -o ConnectTimeout=5 "$REMOTE_HOST" \
  "[ -d '${REMOTE_BACKUP_DIR}' ]" \
  || die "${REMOTE_HOST} unreachable or ${REMOTE_BACKUP_DIR} missing — is ironwolf02 mounted on X230?"

ssh "$REMOTE_HOST" "mkdir -p '${REMOTE_DEST}'"
echo "[*] Syncing: $SRC -> ${REMOTE_HOST}:${REMOTE_DEST}"
rsync -aH --no-owner --no-group --info=stats2,progress2 "$SRC/" "${REMOTE_HOST}:${REMOTE_DEST}/"
echo "[✓] Done: ${REMOTE_HOST}:${REMOTE_DEST}"

* backup-nextcloud – Dual sync (ZFS pool + ironwolf02)

Mirrors =$HOME/Nextcloud= to both the ZFS pool and ironwolf02.

#+BEGIN_SRC bash :tangle x230/.local/bin/backup-nextcloud :shebang "#!/usr/bin/env bash" :mkdirp yes
set -Eeuo pipefail

NAME="nextcloud"

# SUDO-safe HOME
USER_HOME="${SUDO_USER:+$(getent passwd "$SUDO_USER" | cut -d: -f6)}"
USER_HOME="${USER_HOME:-$HOME}"

# === CONFIG ===
SRC="${USER_HOME}/Nextcloud"
DEST_POOL="/mnt/fideco-pool/Nextcloud"                   # ZFS (local pool)
DEST_BACK="/mnt/ironwolf02/backup-nextcloud/Nextcloud"   # ext4 (backup disk)
GUARD_POOL="${DEST_POOL}/.backup_guard"
GUARD_BACK="${DEST_BACK}/.backup_guard"
LOG="/tmp/backup_${NAME}_$(date +%Y%m%d_%H%M%S).log"
LOCK="/tmp/backup_${NAME}.lock"

RSYNC_RUN=( -aH --delete --info=stats2,progress2 --no-owner --no-group --exclude='.backup_guard' )
RSYNC_DRY=( -anv --delete --itemize-changes --no-owner --no-group --exclude='.backup_guard' )

# === ARGS ===
MODE="" AUTO=0 USE_LOCK=1
while (($#)); do
  case "$1" in
    -y|--yes) MODE="yes" ;;
    -n|--dry|--dry-run) MODE="dry" ;;
    --auto) AUTO=1 ;;
    --no-lock) USE_LOCK=0 ;;
    -h|--help)
      cat <<EOF
Usage: $(basename "$0") [-n|--dry-run] [-y|--yes] [--auto] [--no-lock]
  -n, --dry-run   Dry-run only (no writes)
  -y, --yes       Perform real sync
  --auto          Non-interactive (must be combined with -n or -y)
  --no-lock       Skip single-instance lock (not recommended)
EOF
      exit 0;;
    *) echo "ERROR: Unknown arg: $1" >&2; exit 1;;
  esac; shift
done

# === UTILS ===
die(){ echo "ERROR: $*" >&2; exit 1; }
hr(){ command -v numfmt >/dev/null 2>&1 && numfmt --to=iec --suffix=B "$1" || echo "$1 B"; }
avail_bytes(){ df -P -B1 "$1" | awk 'NR==2{print $4}'; }
realp(){ readlink -f "$1" 2>/dev/null || echo "$1"; }

lock(){
  (( USE_LOCK )) || return 0
  exec 9>"$LOCK" || true
  flock -n 9 || die "Another ${NAME} run appears active (lock: $LOCK). Use --no-lock to override."
  echo "$$" 1>&9 || true
}

prep_dest() {
  local d="$1" g="$2" label="$3"
  mkdir -p "$d"
  local D; D="$(realp "$d")"
  [ -w "$D" ] || die "Destination not writable: $D (chown to you)"
  if [ ! -f "$g" ]; then
    {
      echo "mirror target for \$HOME/Nextcloud"
      echo "SRC=$(realp "$SRC")"
      echo "DEST=$D"
      echo "FSTYPE=$(findmnt -no FSTYPE --target "$D")"
      echo "DEV=$(findmnt -no SOURCE --target "$D")"
      echo "CREATED=$(date -Is)"
    } > "$g"
    chmod 644 "$g"
  fi
}

stats_pair(){ # prints: total xfer dels free dev_str
  local s="$1" d="$2"
  local total xfer dels free dev
  read -r total xfer < <( rsync -an --delete --stats --exclude='.backup_guard' --no-owner --no-group "$s/" "$d/" 2>/dev/null \

run_dry(){

  hl_before=$(count_hardlink_files)
  sb_before=$(saved_bytes)
  log "[DRY] existing hardlinked files : $hl_before"
  log "[DRY] existing logical saved    : $(fmt_bytes "$sb_before")"
  case "$BACKEND" in
    czkawka) "$BACKEND_BIN" dup -d "$ROOT" -t "$HASH" --thread-number "$THREADS" | tee -a "$LOG" ;;
    jdupes)  jdupes -r "$ROOT" | tee -a "$LOG" ;;
    rdfind)  rdfind -makeresultsfile false -dryrun true "$ROOT" | tee -a "$LOG" ;;
  esac
  log "[DRY] No changes."
}

run_real(){
  if need_sudo; then start_sudo_keepalive; SUDO=sudo; fi
  log "[run] hashing & linking… (log: $LOG)"

  local hl_before sb_before
  hl_before=$(count_hardlink_files)
  sb_before=$(saved_bytes)

  local rc=0
  case "$BACKEND" in
    czkawka)
      if [ "$CZK_MODE" = "v10+" ]; then
        set +e
        $SUDO ionice -c2 -n7 nice -n 19 "$BACKEND_BIN" dup -d "$ROOT" -t "$HASH" --thread-number "$THREADS" \
             --allow-hard-links --delete-method HARD | tee -a "$LOG"
        rc=${PIPESTATUS[0]}; set -e
      else
        set +e
        $SUDO ionice -c2 -n7 nice -n 19 "$BACKEND_BIN" dup -d "$ROOT" -t "$HASH" --thread-number "$THREADS" \
             --hardlink | tee -a "$LOG"
        rc=${PIPESTATUS[0]}; set -e
      fi
      ;;
    jdupes) set +e; $SUDO jdupes -rL "$ROOT" | tee -a "$LOG"; rc=${PIPESTATUS[0]}; set -e ;;
    rdfind) set +e; $SUDO rdfind -makeresultsfile false -makehardlinks true "$ROOT" | tee -a "$LOG"; rc=${PIPESTATUS[0]}; set -e ;;
  esac
  [ $rc -eq 0 ] || log "[warn] backend exited with code $rc"

  # ----- FINAL SUMMARY (always) -----
  local hl_after sb_after delta_hl delta_sb
  hl_after=$(count_hardlink_files)
  sb_after=$(saved_bytes)
  delta_hl=$(( hl_after - hl_before ))
  delta_sb=$(( sb_after - sb_before ))

  log "-------------- RESULT SUMMARY --------------"
  log "Hardlinked files before : $hl_before"
  log "Hardlinked files after  : $hl_after"
  log "Delta (files)           : $delta_hl"
  log "Logical saved before    : $(fmt_bytes "$sb_before")"
  log "Logical saved after     : $(fmt_bytes "$sb_after")"
  log "Delta (saved)           : $(fmt_bytes "$delta_sb")"

  if [ "$delta_hl" -eq 0 ] && [ "$delta_sb" -eq 0 ]; then
    log "No new duplicates were hard-linked (either none found or already deduped)."
  else
    log ""
    log "Top hard-linked sets (by saved size):"
    show_top_sets 20 | tee -a "$LOG"
  fi

  log "[✓] Completed. Log: $LOG"
}

confirm(){
  if (( AUTO )); then
    [ -n "$MODE" ] || die "--auto requires -n or -y"; return
  fi
  read -rp "Type EXACT ROOT to confirm, or 'd' dry, 'n' abort: " ans
  case "$ans" in
    n|N) exit 0;;
    d|D) MODE="dry";;
    *) [ "$ans" = "$ROOT" ] || die "Confirmation mismatch"; MODE="yes";;
  esac
}

main(){
  pick_backend
  safety
  preview
  confirm
  log "[log] writing to: $LOG"
  if [ "${MODE:-}" = "dry" ]; then run_dry; else run_real; fi
}

main "$@"
